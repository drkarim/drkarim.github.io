Introduction

The program outputs normalized pixel intensities labelled with a mask to file. It does the following:

  • Computes Blood pool mean intensity and standard deviation (specified by la mask)
  • For each pixel in scar binary mask, outputs the normalized intensity to a text file 
Usage: normalizepixel2file.exe <demri> <scar_mask> <la_mask> <text_output>

Code 

#include <irtkImage.h>
#include <fstream>
#include <vector>
#include <string> 

void getMeanVariance(vector<double> val, double& mean, double& variance)
{
	double sum=0.0, N, sum_var=0.0; 

	N = val.size(); 

	for (int i=0;i<N;i++)
	{
		sum += val[i]; 
	}

	mean = sum/N;

	for (int i=0;i<N;i++)
	{
		sum_var += (val[i] - mean)*(val[i] - mean);
	}

	variance = sum_var/(N-1);
}

int main(int argc, char **argv)
{

		ofstream output; 
       char *filename1, *filename2, *filename3, *fileout;
       irtkGreyPixel *demri_ip, *scar_bin_ip, *la_bin_ip;
	   vector<double> val; 
	   double la_mean, la_var, norm;
		
	   if (argc != 5) 
	   {
		   cout << "\nUsage: NormalizePix2File.exe <demri> <scar_mask> <la_mask> <txt_output>\n\n";  
			exit(0); 
	   }
       // get files to subtract
       filename1 = argv[1];
       argv++;
       argc--;
       filename2 = argv[1];
       argv++;
       argc--;
	   filename3 = argv[1];
       argv++;
       argc--;
	   fileout = argv[1];
	   output.open(fileout);
       // read image
       irtkGreyImage demri_i;
       irtkGreyImage scar_bin_i;
	   irtkGreyImage la_bin_i;

       double d;
       demri_i.Read(filename1);
       scar_bin_i.Read(filename2);
	   la_bin_i.Read(filename3); 

       demri_ip = demri_i.GetPointerToVoxels();
       scar_bin_ip = scar_bin_i.GetPointerToVoxels();
	   la_bin_ip = la_bin_i.GetPointerToVoxels(); 


	   // get the mean and std of atrial blood pool 
       for (int i=0;i<demri_i.GetNumberOfVoxels();i++)
       {
				if (*la_bin_ip > 0)
					val.push_back(*demri_ip);  
              
               demri_ip++;
               la_bin_ip++;
       }
	   getMeanVariance(val, la_mean, la_var);
	   cout << "LA blood pool statistics\n===================\nMean\t=\t"<<la_mean<<"\nStd. dev =\t"<<sqrt(la_var) << endl;
	   // reset pointers 
	   demri_ip = demri_i.GetPointerToVoxels();
       scar_bin_ip = scar_bin_i.GetPointerToVoxels();
	   la_bin_ip = la_bin_i.GetPointerToVoxels(); 

	   for (int i=0;i<demri_i.GetNumberOfVoxels();i++)
       {
		   if (*scar_bin_ip > 0) {
			    norm  = ((double)abs(*demri_ip-la_mean))/sqrt(la_var);
				output << norm << endl; 
		   }
			demri_ip++; 
			scar_bin_ip++; 
	   }

	   output.close();
}

Semi automatic scar segmentation from Utah group

From Rashed Karim Wiki

Jump to: navigation, search

Steps:

  • Obtain atrial wall using euclidean distance transform of LA segmentation wall, i.e. 3 pixels in and out of the shell boundary.
  • Perform an Otsu thresholding with 5 classes, and pick the 4th of 5th clas thresholds.

Batch file 

@echo off
setLocal EnableDelayedExpansion
for /f "tokens=* delims= " %%a in (paths_to_3dheart_seg.txt) do (
	set /a N+=1
	set v!N!=%%a
)

for /f "tokens=* delims= " %%a in (paths_to_le.txt) do (
	set /a M+=1
	set k!M!=%%a
)

for /f "tokens=* delims= " %%a in (patient_names_abbr.txt) do (
	set /a K+=1
	set l!K!=%%a
)

for /l %%x in (1,1,11) do (
	prescar3d !k%%x! !v%%x! 1 1 
	mymask !k%%x! _marrouche_aw.gipl _marrouche_aw_masked.gipl 
	 magic -i _marrouche_aw_masked.gipl  -o _marrouche_aw_masked_trimmed.gipl --tm x 
	MyOtsuFilter _marrouche_aw_masked_trimmed.gipl 2 gipl 5 output.txt

	for /f "tokens=* delims= " %%a in (output.txt) do (
	set /a K+=1
	set b!K!=%%a
	)
	
	threshold _marrouche_aw_masked.gipl C:\PhD\experiments\kcl_data\chronic_scans\utah_datasets_redoScarSeg\utah_seg_with_4classesInOtsu\!l%%x!_marrouche_seg.gipl  !b15!
	threshold _marrouche_aw_masked.gipl C:\PhD\experiments\kcl_data\chronic_scans\utah_datasets_redoScarSeg\utah_seg_with_5classesInOtsu\!l%%x!_marrouche_seg.gipl  !b16!

	del _blood_pool.gipl 
	del _atrial_wall.gipl 
	del _dilated_seg.gipl 
	del _gad_seg.gipl 
	del _le.gipl 
	del _prob_map.gipl 
	del _spatial_prior.gipl 
	del output.txt  
	
)

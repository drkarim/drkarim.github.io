Usage

Using the output of image obtained from magic2 with -concomp switch, this code outputs the bounding box coordinates of connected component along with their sizes 

Usage: concompbox -i1 <output of connected component image from magic2> -o <textfile to list co-ordinates of bounding box>

Typical output:

Code 

#include <irtkImage.h>
#include <stdio.h>    
#include <stdlib.h>   
#include <fstream>
#include <iostream>
#include <vector>


void FindComponentBounds(vector<vector<int> >& component_bounds, char* concomp_img)
{
	int N; 
	int x,y,z;
	irtkGreyImage img1; 
	irtkGreyPixel* min, *max; 
	irtkGreyPixel* img1_p;
	min = new irtkGreyPixel(); 
	max = new irtkGreyPixel();

	img1.Read(concomp_img); 
	
	img1.GetMinMax(min, max); 
	N = *max; 

	for (int i=0;i<=N;i++)
	{
		vector<int> minmaxcount; 
		minmaxcount.push_back(1e9);
		minmaxcount.push_back(1e9);
		minmaxcount.push_back(1e9); 
		minmaxcount.push_back(-1);
		minmaxcount.push_back(-1);
		minmaxcount.push_back(-1);
		minmaxcount.push_back(0);
		component_bounds.push_back(minmaxcount);
	}


	int maxX, maxY, maxZ,val;
	maxX = img1.GetX(); 
	maxY = img1.GetY(); 
	maxZ = img1.GetZ(); 
	
	for (int i=0;i<maxX;i++) { for( int j=0;j<maxY;j++) { for (int k=0;k<maxZ;k++) { 

		val = img1.Get(i,j,k); 
		
		if (val > 0 && val <= component_bounds.size())
		{	
			// find the min coordinate of connected component
			if (component_bounds[val][0] > i) component_bounds[val][0] = i;			
			if (component_bounds[val][1] > j) component_bounds[val][1] = j; 
			if (component_bounds[val][2] > k) component_bounds[val][2] = k; 
			
			// find the max coordinate of connected component
			if (component_bounds[val][3] < i) component_bounds[val][3] = i; 
			if (component_bounds[val][4] < j) component_bounds[val][4] = j; 
			if (component_bounds[val][5] < k) component_bounds[val][5] = k; 

			component_bounds[val][6]++; 
		}
		
	}}}
	
}

void WriteMinMaxCountToFile(vector<vector<int> > minmaxcount, char* f_name) 
{
	ofstream fout;
	fout.open(f_name); 

	fout << "Component ID\t" << "MinX\t" << "MinY\t" << "MinZ\t" << "MaxX\t" << "MaxY\t" << "MaxZ\t" << "Size" << endl;
	for (int i=1;i<minmaxcount.size();i++)
	{
		fout << i << "\t" << minmaxcount[i][0] << "\t" << minmaxcount[i][1] << "\t" << minmaxcount[i][2] << "\t" 
			"" << minmaxcount[i][3] << "\t" << minmaxcount[i][4] << "\t" << minmaxcount[i][5] << "\t" << minmaxcount[i][6] << "\n";
			
	}

	fout.close();
}

int main(int argc, char **argv)
{
	int option;
	bool foundArgs1=false,foundArgs2=false;
	irtkRealImage img1, img2; 
	irtkGreyImage img3; 
	char* input_f1="", *output_f="", *input_f2=""; 
	int t1, t2, t3; 

	if (argc >= 1)
	{
			for (int i = 1; i < argc; i++) {
				if (i + 1 != argc) {
					if (string(argv[i]) == "-i1") {
						input_f1 = argv[i+1]; 
						foundArgs1 = true;
					
					}  else if (string(argv[i]) == "-o") {
						output_f = argv[i+1]; 
						foundArgs2 = true; 
					
					}
					
				}
			}
	}

	if (!(foundArgs1 && foundArgs2 ))
	{
		cout << "\n\nUsage: concompbox \n\n\tArguments:"
			"\n\t -i1 \t Connected component image from magic2"
			"\n\t -o \t - Output text file"
			"\n\n";
		exit(0);
	}

	vector<vector<int >> minmaxcount; 
	FindComponentBounds(minmaxcount, input_f1); 
	WriteMinMaxCountToFile(minmaxcount, output_f); 
}

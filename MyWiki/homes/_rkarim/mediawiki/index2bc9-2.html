Usage

Finds intersection of two meshes that have undergone binary thresholding and thus 1 and 0 point scalars. Binary thresholding can be achieved using magic3 --binaryscalar switch. 

meshintersect.exe -m1 <mesh1> -m2 <mesh2> -o <output text file>

Note: mesh1 and mesh2 are geometrically identical.

In the output text file, there are three numbers output:

  • The total points that have a scalar of 1 and are overlappng in both meshes.
  • The total points that have a scalar of 1 from mesh1
  • The total points that have a scalar of 1 from mesh2

#include "rk_test3.h"

void ReadMesh(char* fn1, vtkPolyData* poly) 
{
	vtkSmartPointer<vtkPolyDataReader> reader = vtkSmartPointer<vtkPolyDataReader>::New(); 
	reader->SetFileName(fn1); 
	reader->Update();
	vtkSmartPointer<vtkTriangleFilter> triangleFilter = vtkSmartPointer<vtkTriangleFilter>::New();
	triangleFilter->SetInput(reader->GetOutput());
	triangleFilter->Update();

	poly->DeepCopy(triangleFilter->GetOutput());

}

void MeshIntersect(vtkPolyData* m1, vtkPolyData* m2, double& total_intersect, double& m1_total, double& m2_total)
{
	double val_m1, val_m2; 
	vtkFloatArray* scalars_m1 = vtkFloatArray::New();
	vtkFloatArray* scalars_m2 = vtkFloatArray::New();
	
	total_intersect = 0; 
	m1_total = 0; 
	m2_total = 0; 
	scalars_m1 = vtkFloatArray::SafeDownCast(m1->GetPointData()->GetScalars()); 
	scalars_m2 = vtkFloatArray::SafeDownCast(m2->GetPointData()->GetScalars()); 
	
	
	for (int i=0;i<m1->GetNumberOfPoints();i++)
	{
		 val_m1 = scalars_m1->GetTuple1(i); 
		 val_m2 = scalars_m2->GetTuple1(i); 

		 total_intersect += (i,val_m1*val_m2); 
		 
		 if (val_m1 > 0) 
			 m1_total++; 
		 if (val_m2 > 0) 
			 m2_total++; 
	}


	
	//m2->GetPointData()->SetScalars(new_scalars);
}



int main(int argc, char **argv)
{
	char* input_f1="", *input_f2="", *output_f; 
	bool foundArgs = false; 
	double total_intersect, m1_total, m2_total;
	vtkSmartPointer<vtkPolyData> poly1 = vtkSmartPointer<vtkPolyData>::New();
	vtkSmartPointer<vtkPolyData> poly2 = vtkSmartPointer<vtkPolyData>::New();
	ofstream out; 
	
	if (argc >= 1)
	{
			for (int i = 1; i < argc; i++) {
				if (i + 1 != argc) {
					if (string(argv[i]) == "-m1") {
						input_f1 = argv[i+1]; 
						foundArgs = true;
					} else if (string(argv[i]) == "-m2") {
						input_f2 = argv[i+1]; 
						foundArgs = true;
					} else if (string(argv[i]) == "-o") {
						output_f = argv[i+1]; 
						foundArgs = true;
					}
					
				}
			}
	}
	if (!(foundArgs))
	{
		cerr << "Usage: \nMeshIntersect -m1 <mesh.vtk> -m2 <mesh2.vtk> -o <total_intersection.txt>\n" << endl;
		exit(0);
	}


	ReadMesh(input_f1, poly1);
	ReadMesh(input_f2, poly2);

	 MeshIntersect(poly1, poly2, total_intersect, m1_total, m2_total); 
	out.open(output_f, std::ios_base::app); 
	out << total_intersect << "\t" << m1_total << "\t" << m2_total << endl; 
	cout << "Total intersect = \n" << total_intersect << " points " << endl; 
	out.close(); 
	
	
}

Introduction

Warning: Only randomly selects n data points as opposed to all data read from file

Same as the R-square but see warning above.

Code 

function [R2,m] = my_rsq_for_unfold_areas(file1, file2, n, filename, threshold)

A= load(file1);
%sA = sort(A);
B=load(file2); 

% remove all non-zero entries
nz = (A~=0); 
A=A(nz); 
B=B(nz);
B=B*100000;
if nargin >= 5
    if (threshold ~= -1)
        A0=A>threshold;
        A(A0)=[];
        B(A0)=[];
    end
end

% remove outliers
[B_, B_idx] = deleteoutliers(B);
A(B_idx)=[];
B(B_idx)=[];

if nargin < 3
    randomSelect = 0;
else
    randomSelect = 1;
end
    
%select random entries 
if randomSelect == 1
    sizeA = size(A)
    rn = randperm(sizeA(1,1));
    rn = rn(1:n); 
    A=A(rn); 
    B=B(rn);
end


m=pinv(A)*B;
Ahat = m*A;
R2 = rsquare(B, Ahat);
%R2=0.0
figure1 = figure('Color',[1 1 1]);
hold on 
box on

plot(A,B,'o','Color',[0 0 0],'LineWidth',3,'MarkerSize',2);


sA = [0, max(A)];
plot(sA, m*sA, 'LineWidth',3, 'Color',[1 0 0]);
set(gca,'XLim',[0 max(A)]);
set(gca, 'FontSize', 20); 

% now plotting straight line 
R=sqrt(R2);
str = num2str(R2); 
str_R = num2str(R); 
str2 = ['R = ',str_R]
text(40,500, str2, 'FontSize',20)

xlabel('Area in 3D','FontSize',22);
ylabel('Area in 2D unfold','FontSize',24);

if filename ~= -1
        print(figure1,'-djpeg', filename)
end
  
end

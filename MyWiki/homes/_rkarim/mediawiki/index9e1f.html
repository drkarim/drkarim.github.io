Introduction

Computes a kind of curvature the surface at every point. For each polygon (and its normal), computes the distribution of angle-deviation of normals of its neighbouring polygons. It uses this VTK example.

The standard deviation of the angle is represented at each point (by default). There are also options to represent the mean. This is good for analysing segmentations


Usage 

normalysis.exe -i <shell vtk> -o <shell vtk with normalysis> --mean --std

void ComputeAngle(float x1, float y1, float z1, float x2, float y2, float z2, double& angle) 
{
	double dot = x1*x2 + y1*y2 + z1*z2;
    double crossX = y1*z2-z1*y2;
    double crossY = z1*x2-x1*z2;
    double crossZ = x1*y2-y1*x2;
    double norm = sqrt(crossX*crossX+crossY*crossY+crossZ*crossZ);
    angle= (atan2(norm,dot)/3.1415926*180);
}


void GetConnectedCells(vtkSmartPointer<vtkPolyData> mesh, int seed_cell, std::list<vtkIdType>& neighbors) 
{ 
	
	
	vtkSmartPointer<vtkIdList> cellPointIds  = vtkSmartPointer<vtkIdList>::New(); 
	mesh->GetCellPoints(seed_cell, cellPointIds ); 

	
	for(vtkIdType i = 0; i < cellPointIds->GetNumberOfIds(); i++) 
	{ 
		
		vtkSmartPointer<vtkIdList> idList = vtkSmartPointer<vtkIdList>::New();
		idList->InsertNextId(cellPointIds->GetId(i));

		vtkSmartPointer<vtkIdList> neighborCellIds = vtkSmartPointer<vtkIdList>::New();
		mesh->GetCellNeighbors(seed_cell, idList, neighborCellIds);

		for(vtkIdType j = 0; j < neighborCellIds->GetNumberOfIds(); j++)
		  {
		  neighbors.push_back(neighborCellIds->GetId(j));
		  }
	}
		
	neighbors.unique(); 

   // cout << "There are " << connectedVertices->GetNumberOfIds() << " points connected to point " << seed << endl;
}

void getMeanVariance(vector<double> val, double& mean, double& variance, double& total_errors)
{
	double sum=0.0, N, sum_var=0.0; 

	N = val.size(); 

	for (int i=0;i<N;i++)
	{
		sum += val[i]; 
		//cout << val[i] << ","; 
	}
	
	if (N > 0) 
		mean = sum/N;
	else 
		mean = 0; 

	for (int i=0;i<N;i++)
	{
		sum_var += ((val[i] - mean)*(val[i] - mean));
	}
	
	if (N > 1) 
		variance = sum_var/(N-1);
	else 
		variance = 0; 

	total_errors = sum;
}


void AnalyseNormal(vtkSmartPointer<vtkPolyData> mesh, int std_or_mean) 
{
	vtkSmartPointer<vtkPolyData> polys = vtkSmartPointer<vtkPolyData>::New(); 
	double cN[3];		// cell normal 
	double cNN[3];		// cell neighbours' normal 
	double angle=0, mean=0, variance=0, errors=0; 
	vtkSmartPointer<vtkPolyDataNormals> normals = vtkSmartPointer<vtkPolyDataNormals>::New();
	//normals->FlipNormalsOn();
	normals->AutoOrientNormalsOn();
	normals->ComputeCellNormalsOn();
	//normals->ComputePointNormalsOn();
	normals->SetInput(mesh); 
	normals->Update(); 
	polys = normals->GetOutput(); 
	vector<double> angles; 

	std::list<vtkIdType> cell_neighbours; 
	vtkSmartPointer<vtkFloatArray> cellNormals= vtkFloatArray::SafeDownCast(polys->GetCellData()->GetNormals());

	// Cel scalars to be populated with the computed angle at each cell 
	vtkSmartPointer<vtkFloatArray> scalars = vtkSmartPointer<vtkFloatArray>::New();

	for (vtkIdType cellId=0;cellId<polys->GetNumberOfCells();cellId++)			// running through each cell 
	{
		
		GetConnectedCells(polys, cellId, cell_neighbours); 
		cellNormals->GetTuple(cellId, cN);		// get this cell's normal 
		
		// iterate through each neighbour of the cell 
		for(std::list<vtkIdType>::iterator it1 = cell_neighbours.begin(); it1 != cell_neighbours.end(); it1++)
		{
			cellNormals->GetTuple(*it1, cNN);	// get cell's neighbours normal 
			ComputeAngle(cN[0], cN[1], cN[2], cNN[0], cNN[1], cNN[2], angle); 

			if (angle >= 0 && angle < 180)
				angles.push_back(angle); 
		}
		
		getMeanVariance(angles, mean, variance, errors); 
		

		if (mean > 0 || variance > 0) {
			if (std_or_mean == 1) 
				scalars->InsertTuple1(cellId, mean); 
			else 
				scalars->InsertTuple1(cellId, sqrt(variance)); 
		}
		else {
			scalars->InsertTuple1(cellId, 0); 
		}
		
		mean = 0; 
		variance = 0;
		errors = 0; 
		angle=0;
		angles.clear(); 
		if (!cell_neighbours.empty()) 
			cell_neighbours.clear(); 
	}

	mesh->GetCellData()->SetScalars(scalars);
}



int main(int argc, char *argv[])
{
	vtkPolyData* _surf1, *_surf2; 
	char *filename1, *filename2, *output_f; 
	double error=0; 
	int user_args, std_or_mean=0;		
	bool foundArgs=false; 

	
	
	_surf1 = vtkPolyData::New(); 
	
	
	if (argc >= 1) 
	{
		int optind=1;
		
		while ((optind < argc) && (argv[optind][0]=='-')) {
			
			string sw = argv[optind];
			
			if (sw == "-i") {
				optind++;
				filename1 = argv[optind]; 
				foundArgs = true;	
			}
			else if (sw == "-o") {
				optind++;
				output_f = argv[optind]; 
				foundArgs = true;	
			}
			
			
			else if (sw == "--mean") 
			{
				
				std_or_mean = 2;
				cout << "\n\nNote: Outputting mean"; 
			}

			else if (sw == "--std") 
			{
				
				std_or_mean = 1;
				cout << "\n\nNote: Outputting standard deviation"; 
			}
			
			optind++; 
		}
	}
	
	if (foundArgs == false)
	{
		cout << "Usage: normalisys.exe -i <input vtk> -o <output vtk> --mean --std" << endl; 
		exit(0); 
	}
	

	// Command line arguments found, now Ready to compute 
	vtkSmartPointer<vtkPolyDataReader> reader1 = vtkSmartPointer<vtkPolyDataReader>::New();
	reader1->SetFileName(filename1); 
	reader1->Update();
	_surf1 = reader1->GetOutput();
	
	AnalyseNormal(_surf1, std_or_mean); 
	
	vtkSmartPointer<vtkPolyDataWriter> writer = vtkSmartPointer<vtkPolyDataWriter>::New(); 
	writer->SetInput(_surf1); 
	writer->SetFileName(output_f); 
	writer->Update(); 
	

}

void XylusMainFrameHelper::sliceImageExtensionOnSelectionChange_cb( wxCommandEvent& event )
{
	int selection = m_choice_filenameExt->GetCurrentSelection(); 
	switch (selection) 
	{
		case '0':
			p3D->SetSliceFilenameExt("jpg"); 
		break; 
		case '1':
			p3D->SetSliceFilenameExt("png"); 
		break;
		case '2':
			p3D->SetSliceFilenameExt("gif"); 
		break;
		case '3':
			p3D->SetSliceFilenameExt("bmp"); 
		break;
		case '4':
			p3D->SetSliceFilenameExt("jpeg"); 
		break;
	}
}

void XylusMainFrameHelper::zMintextCtrl_ontextchange_cb( wxCommandEvent& event ){
	double value; 
	wxString s =  m_textCtrl_zMin->GetValue(); 
	string s_str = s.mb_str(); 

	if (s_str == "") 
	{
		value = -1e9;
		p3D->SetZSlicingMin(value); 
	}
	else
	{
		if(s.ToDouble(&value)){
			p3D->SetZSlicingMin(value); 
		}

	}
}


void XylusMainFrameHelper::zMaxtextCtrl_ontextchange_cb( wxCommandEvent& event ){

}


bool MyApp::OnInit()
{
	wxInitAllImageHandlers();
	XylusImgDisplayFrameHelper *displayApp = new XylusImgDisplayFrameHelper(NULL); 
    XylusMainFrameHelper *btnapp = new XylusMainFrameHelper(NULL);
    btnapp->Show(true);

	displayApp->Show(false);
	
	btnapp->SetImageDisplayer(displayApp);

    return true;
}

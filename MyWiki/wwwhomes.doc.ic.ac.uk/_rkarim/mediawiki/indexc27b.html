#include "vtkSmartPointer.h"
#include "vtkSphereSource.h"
#include "vtkPolyDataMapper.h"
#include "vtkActor.h"
#include "vtkRenderer.h"
#include "vtkRenderWindow.h"
#include "vtkRenderWindowInteractor.h"
#include "vtkProperty.h"
#include "vtkCamera.h"
#include "vtkLight.h"
#include "vtkLookupTable.h"
#include "vtkFloatArray.h"
#include "vtkPolyDataReader.h"
#include "vtkPointData.h"
#include "vtkInteractorStyleTrackballCamera.h"
#include "vtkScalarBarActor.h"

int main(int argc, char **argv)
{
	double max_scalar=-1,min_scalar = 1e10,v; 
	vtkSmartPointer<vtkFloatArray> scalars;
	if (argc < 2)
	{
		cout << "displayVTK.exe <shellvtk>"<< endl; 
		exit(1);
	}

	vtkSmartPointer<vtkPolyData> polyData = vtkSmartPointer<vtkPolyData>::New();
	vtkSmartPointer<vtkPolyDataReader> reader = vtkSmartPointer<vtkPolyDataReader>::New(); 
	reader->SetFileName(argv[1]); 
	 reader->Update(); 
	 polyData = reader->GetOutput();
	vtkSmartPointer<vtkPolyDataMapper> shellMapper = vtkSmartPointer<vtkPolyDataMapper>::New();
	shellMapper->SetInput(reader->GetOutput());
	shellMapper->SetScalarModeToUsePointData();
	
//	shellMapper->SetColorModeToMapScalars();
  
    vtkSmartPointer<vtkActor> shellActor = vtkSmartPointer<vtkActor>::New();
	shellActor->SetMapper(shellMapper); 

	scalars =  vtkFloatArray::SafeDownCast(polyData->GetPointData()->GetScalars()); 
	for (int i=0;i<polyData->GetNumberOfPoints();i++)			// running through each polygon point 
	{
		v = scalars->GetTuple1(i);
		
		if (max_scalar<v) 
			max_scalar = v; 
		else if (min_scalar > v && v > 0) 
			min_scalar = v;
	}

	cout << "Min scalar = " << min_scalar << "\nMax scalar = " << max_scalar <<endl;
	
	shellActor->GetProperty()->SetAmbient(0.2); 
	shellActor->GetProperty()->SetDiffuse(0.5); 
	shellActor->GetProperty()->SetSpecular(1.0); 
	shellActor->GetProperty()->SetSpecularPower(90.0); 
	shellActor->GetProperty()->SetInterpolationToPhong();
	shellActor->GetProperty()->SetSpecularColor(1,1,1);
	//shellActor->GetProperty()->EdgeVisibilityOn();
	
	//this line is very important and specifies the range of scalars that will be mapped 
	shellMapper->SetScalarRange(min_scalar,max_scalar);


	vtkSmartPointer<vtkLookupTable> lut =vtkSmartPointer<vtkLookupTable>::New();
	lut->SetNumberOfColors(256);
	//lut->SetTableRange(min_scalar, max_scalar);
	lut->SetHueRange (0.16, 0.0);
	lut->SetSaturationRange (1.0, 1.0);
	lut->SetValueRange (1.0, 1.0);
	lut->Build();
	shellMapper->SetLookupTable(lut);

	vtkSmartPointer<vtkScalarBarActor> scalarBar = vtkSmartPointer<vtkScalarBarActor>::New();
	scalarBar->SetLookupTable(lut);
	scalarBar->SetTitle("scalars");
	scalarBar->SetWidth(0.08); 
	scalarBar->SetHeight(1.0); 
	scalarBar->SetDisplayPosition(8,8); 
	


	
  vtkSmartPointer<vtkRenderer> ren1 = vtkSmartPointer<vtkRenderer>::New();
  vtkSmartPointer<vtkRenderWindow> renWin = vtkSmartPointer<vtkRenderWindow>::New();
  renWin->AddRenderer(ren1);
  vtkSmartPointer<vtkRenderWindowInteractor> iren = vtkSmartPointer<vtkRenderWindowInteractor>::New();
  iren->SetRenderWindow(renWin);
  ren1->AddActor2D(scalarBar);

  // Add the actors to the renderer, set the background and size.
  //
  ren1->AddActor(shellActor);
 
  ren1->SetBackground(0,0,0);
  renWin->SetSize(400, 200);

  vtkSmartPointer<vtkInteractorStyleTrackballCamera> interactorStyle1 = vtkSmartPointer<vtkInteractorStyleTrackballCamera>::New();

  // We want to eliminate perspective effects on the apparent lighting.
  // Parallel camera projection will be used. To zoom in parallel projection
  // mode, the ParallelScale is set.
  //
  ren1->GetActiveCamera()->SetFocalPoint(0,0,0);
  ren1->GetActiveCamera()->SetPosition(0,0,1);
  ren1->GetActiveCamera()->SetViewUp(0,1,0);
  ren1->GetActiveCamera()->ParallelProjectionOn();
  ren1->ResetCamera();
  ren1->GetActiveCamera()->SetParallelScale(1.5);
  
  
  // This starts the event loop and invokes an initial render.
  //
  iren->SetInteractorStyle(interactorStyle1);
  iren->Initialize();
  iren->Start();

  return EXIT_SUCCESS;
}

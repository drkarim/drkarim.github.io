void VizBoxCore::VBC_DisplayPoly(vtkFlRenderWindowInteractor *&flrwi, vtkPolyData* poly, bool isCellData, vtkActor* polyActor)
{
	double min_scalar, max_scalar; 
	vtkSmartPointer<vtkRenderer> renderer;
	vtkSmartPointer<vtkRenderWindow> renderWin;
	renderWin = flrwi->GetRenderWindow();
	vtkSmartPointer<vtkRendererCollection> rendererCollection = renderWin->GetRenderers();
	vtkSmartPointer<vtkLookupTable> lut = vtkSmartPointer<vtkLookupTable>::New();
	vtkSmartPointer<vtkScalarBarActor> scalarBar = vtkSmartPointer<vtkScalarBarActor>::New();
	renderer = rendererCollection->GetFirstRenderer();

	vtkSmartPointer<vtkPolyDataMapper> mapper = vtkSmartPointer<vtkPolyDataMapper>::New();	
	cout << "Adding polydata to mapper, polydata now has " << poly->GetNumberOfCells() << " polygons .. " << endl;
	mapper->SetInput(poly); 
	
	mapper->ScalarVisibilityOn();

	if (isCellData) 
		mapper->SetScalarModeToUseCellData();
	else 
		mapper->SetScalarModeToUsePointData();
	
	VizBoxCore::VBC_getMinMaxScalars(poly, isCellData, min_scalar, max_scalar); 

cout << "Max scalar in polydata = " << max_scalar << endl;
	mapper->SetScalarRange(0, (max_scalar)); 
	lut->SetTableRange(0, (max_scalar)); 
	
	
	// lookuptable 
	lut->SetHueRange(0.33, 0.0);			// red - 0, yellow - 60, green- 120, cyan - 180, blue - 240, magenta - 300 (on the hue circle, pg. 42 vtk book)
	lut->Build(); 

	mapper->SetLookupTable(lut); 
	
	//vtkSmartPointer<vtkActor> polyActor = vtkSmartPointer<vtkActor>::New(); 

	polyActor->SetMapper(mapper); 
	renderer->AddActor(polyActor);
	
	
	scalarBar->SetLookupTable(lut);
	scalarBar->SetTitle("intensity");
	scalarBar->SetWidth(0.08); 
	scalarBar->SetHeight(1.0); 
	scalarBar->SetDisplayPosition(3,8); 
	 
	
	renderer->AddActor2D(scalarBar);			// displays the scalar bar 
	renderWin->AddRenderer(renderer);
	renderWin->Render();
}

	clip->Update();	
	clippedPoly->DeepCopy(clip->GetOutput());
}

void VizBoxCore::VBC_getStatsOnPoly(vtkPolyData* poly, bool isCellData, double& mean, double& stdev) 
{
	vtkSmartPointer<vtkFloatArray> scalars;
	if (isCellData) 
		scalars =  vtkFloatArray::SafeDownCast(poly->GetCellData()->GetScalars()); 
	else 
		scalars = vtkFloatArray::SafeDownCast(poly->GetPointData()->GetScalars()); 

	std::vector<double> temp; 
	for (int i=0;i<scalars->GetNumberOfTuples();i++)
	{
		if (scalars->GetTuple1(i) < 1e5) 
		temp.push_back(scalars->GetTuple1(i)); 
	}
	
	MathBox::getMeanVariance(temp, mean, stdev); 
	stdev = sqrt(stdev);
	cout << "\nmean = " << mean << "\nstdev = " << stdev << endl;
}

void VizBoxCore::VBC_getSelectedStatsOnPoly(vtkPolyData* data_poly, vtkPolyData* select_poly, double val, bool isCellData, double& mean, double& stdev, 
								bool writeToFile, const char* filename)
{
	vtkSmartPointer<vtkFloatArray> scalars; 
	vtkSmartPointer<vtkFloatArray> scalars2; 
	ofstream sout; 
	double s; 
	if (writeToFile) sout.open(filename); 

	if (isCellData) {
		scalars =  vtkFloatArray::SafeDownCast(data_poly->GetCellData()->GetScalars()); 
		scalars2 =  vtkFloatArray::SafeDownCast(select_poly->GetCellData()->GetScalars()); 
	}
	else {
		scalars = vtkFloatArray::SafeDownCast(data_poly->GetPointData()->GetScalars()); 
		scalars2 = vtkFloatArray::SafeDownCast(select_poly->GetPointData()->GetScalars()); 
	}
	std::vector<double> temp; 
	for (int i=0;i<scalars2->GetNumberOfTuples();i++)
	{
		if (scalars2->GetTuple1(i) > val) {
			s = scalars->GetTuple1(i); 
			temp.push_back(s); 

			if (writeToFile) { 
				sout << s << endl; 
			}
		}
	}

	MathBox::getMeanVariance(temp, mean, stdev); 
	stdev = sqrt(stdev);
	cout << "\nmean = " << mean << "\nstdev = " << stdev << endl;
	
	if (writeToFile) { 
		sout << "\n\n" << "=================" << "\nMean\t\t" << mean << "\nStd-dev\t\t" << stdev;

void VizBoxCore::VBC_GetConnectedVertices(vtkSmartPointer<vtkPolyData>  mesh, int seed, vtkSmartPointer<vtkIdList> connectedVertices) 
{ 

  //get all cells that vertex 'seed' is a part of 
  vtkSmartPointer<vtkIdList> cellIdList = vtkSmartPointer<vtkIdList>::New(); 
  mesh->GetPointCells(seed, cellIdList); 


  //loop through all the cells that use the seed point 
  for(vtkIdType i = 0; i < cellIdList->GetNumberOfIds(); i++) 
    { 

    vtkCell* cell = mesh->GetCell(cellIdList->GetId(i)); 
    //cout << "The cell has " << cell->GetNumberOfEdges() << " edges." << endl; 

    //if the cell doesn't have any edges, it is a line 
    if(cell->GetNumberOfEdges() <= 0) 
      { 
      continue; 
      } 

    for(vtkIdType e = 0; e < cell->GetNumberOfEdges(); e++) 
      { 
      vtkCell* edge = cell->GetEdge(e); 

      vtkIdList* pointIdList = edge->GetPointIds(); 
    
      if(pointIdList->GetId(0) == seed || pointIdList->GetId(1) == seed) 
        { 
        if(pointIdList->GetId(0) == seed) 
          { 
          connectedVertices->InsertNextId(pointIdList->GetId(1)); 
          } 
        else 
          { 
          connectedVertices->InsertNextId(pointIdList->GetId(0)); 
          } 
        } 
      } 


    } 

}

void VizBoxCore::VBC_getMinMaxScalars(vtkPolyData* poly, bool isCellData, double& min, double& max) 
{
	int n; 
	double val; 
	min = 1e20; max = -1e20; 

	vtkSmartPointer<vtkFloatArray> scalars = vtkSmartPointer<vtkFloatArray>::New(); 
	
	if (isCellData) { 
		scalars = vtkFloatArray::SafeDownCast(poly->GetCellData()->GetScalars()); 
		n = poly->GetNumberOfCells(); 

	}
	else { 
		scalars = vtkFloatArray::SafeDownCast(poly->GetPointData()->GetScalars()); 
		n = poly->GetNumberOfPoints();
	}

	for (int i=0;i<n;i++)
	{
		val = scalars->GetTuple1(i); 
		if (val > max) max = val; 
		if (val < min) min = val; 
	}

}

void VizBoxCore::VBC_resetUnwantedScalars(vtkPolyData* poly, bool isCellData, double min, double max, double val)
{
	int n; 
	vtkSmartPointer<vtkFloatArray> scalars = vtkSmartPointer<vtkFloatArray>::New(); 
	if (isCellData) { 
		scalars = vtkFloatArray::SafeDownCast(poly->GetCellData()->GetScalars()); 
		n = poly->GetNumberOfCells(); 
	}
	else { 
		scalars = vtkFloatArray::SafeDownCast(poly->GetPointData()->GetScalars()); 
		n = poly->GetNumberOfPoints(); 
	}
	
	for (int i=0;i<n;i++)
	{
		if (scalars->GetTuple1(i) < min && scalars->GetTuple1(i) > max) 
			scalars->SetTuple1(i, val); 
	}

	if (isCellData) poly->GetCellData()->SetScalars(scalars); 
	else poly->GetPointData()->SetScalars(scalars); 

}

void VizBoxCore::VBC_thresholdScalarsOnShell(vtkPolyData* out_poly, bool isCellData, double t, double insideVal, double outsideVal, bool allowAllThatAreLessThanT)
{
	int n; 
	vtkSmartPointer<vtkFloatArray> scalars = vtkSmartPointer<vtkFloatArray>::New(); 
	if (isCellData) { 
		scalars = vtkFloatArray::SafeDownCast(out_poly->GetCellData()->GetScalars()); 
		n = out_poly->GetNumberOfCells(); 
	}
	else { 
		scalars = vtkFloatArray::SafeDownCast(out_poly->GetPointData()->GetScalars());
		n = out_poly->GetNumberOfPoints(); 
	}

	for (int i=0;i<n;i++)
	{
		if (allowAllThatAreLessThanT) {
			if (scalars->GetTuple1(i) <= t)
				scalars->SetTuple1(i, insideVal); 
			else 
				scalars->SetTuple1(i, outsideVal);
		}
		else {
			if (scalars->GetTuple1(i) >= t)
				scalars->SetTuple1(i, insideVal);
			else 
				scalars->SetTuple1(i, outsideVal);
		}
	}

	if (isCellData) out_poly->GetCellData()->SetScalars(scalars); 
	else out_poly->GetPointData()->SetScalars(scalars); 

	
}

void VizBoxCore::VBC_showImageAsOrthogonalPlanes(vtkFlRenderWindowInteractor *&flrwi, irtkRealImage* img, vtkImageActor** sliceArray)
{
	int maxX, maxY, maxZ; 
	double min_grey=1e10, max_grey=-1; 
	irtkRealPixel* im_ip; 

	vtkSmartPointer<vtkImageMapToColors> sliceZColors, sliceXColors, sliceYColors;
	//vtkSmartPointer<vtkImageActor> ySlice, xSlice, zSlice;

	vtkSmartPointer<vtkStructuredPoints> im = vtkSmartPointer<vtkStructuredPoints>::New();
	//xSlice = vtkSmartPointer<vtkImageActor>::New(); 
	//ySlice =  vtkSmartPointer<vtkImageActor>::New();
	//zSlice =  vtkSmartPointer<vtkImageActor>::New(); 

	img->ImageToVTK(im); 
	im_ip = img->GetPointerToVoxels(); 
	for (int i=0;i<img->GetNumberOfVoxels();i++)
	{
		if (*im_ip < min_grey) min_grey = *im_ip; 
		if (*im_ip > max_grey) max_grey = *im_ip; 
		im_ip++; 
	}
		
	vtkSmartPointer<vtkRenderer> renderer;
	vtkSmartPointer<vtkRenderWindow> renderWin;
	renderWin = flrwi->GetRenderWindow();
	vtkSmartPointer<vtkRendererCollection> rendererCollection = renderWin->GetRenderers();
	renderer = rendererCollection->GetFirstRenderer();

	vtkSmartPointer<vtkLookupTable> bwLut = vtkSmartPointer<vtkLookupTable>::New();
	if (max_grey == 1) max_grey=2;		// for binary images 
	else max_grey+=100;					// for greyscales we set the lookup table to 100 units above the max intensity 
	bwLut->SetRange(min_grey, max_grey); // image intensity range
	bwLut->SetValueRange(0.0, 1.0); // from black to white
	bwLut->SetSaturationRange(0.0, 0.0); // no color saturation
	bwLut->SetRampToLinear();
	bwLut->Build();
	
	sliceXColors = vtkSmartPointer<vtkImageMapToColors>::New(); 
	sliceYColors = vtkSmartPointer<vtkImageMapToColors>::New(); 
	sliceZColors = vtkSmartPointer<vtkImageMapToColors>::New(); 

	maxX = img->GetX(); maxY = img->GetY(); maxZ = img->GetZ(); 

	// Z planes
	sliceZColors->SetInput(im);
	sliceZColors->SetLookupTable(bwLut);
	sliceArray[2]->SetInput(sliceZColors->GetOutput());
	sliceArray[2]->SetDisplayExtent(0, maxX-1, 0, maxY-1, 0,0);

	// X Plane
	sliceXColors->SetInput(im);
	sliceXColors->SetLookupTable(bwLut);
	sliceArray[0]->SetInput(sliceXColors->GetOutput());
	sliceArray[0]->SetDisplayExtent(0,0, 0, maxY-1, 0, maxZ-1);

	// Y Plane
	sliceYColors->SetInput(im);
	sliceYColors->SetLookupTable(bwLut);
	sliceArray[1]->SetInput(sliceYColors->GetOutput());
	sliceArray[1]->SetDisplayExtent(0, maxX-1, 0,0, 0, maxZ-1);
	
	renderer->AddActor(sliceArray[0]);
	renderer->AddActor(sliceArray[1]);
	renderer->AddActor(sliceArray[2]);
	
	vtkSmartPointer<vtkCamera> camera = vtkSmartPointer<vtkCamera>::New(); 
	camera = renderer->GetActiveCamera();
	camera->SetPosition(0,0,500);
    camera->SetFocalPoint(0,0,0);
	renderer->SetActiveCamera(camera);
	renderWin->Render();
	
}

void VizBoxCore::VBC_drawSplineCurve(vtkFlRenderWindowInteractor *&flrwi, vector<irtkPoint*> pts, int num_points, double r, double g, double b)
{
	vtkSmartPointer<vtkCellArray> cells = vtkSmartPointer<vtkCellArray>::New();
	vtkSmartPointer<vtkPoints> points = vtkSmartPointer<vtkPoints>::New(); 


	for (int i = 0; i < num_points-1; i++) {
		int ids[2];
		ids[0] = i;
		ids[1] = i + 1;
		cells->InsertNextCell(2);
		cells->InsertCellPoint(ids[0]);
		cells->InsertCellPoint(ids[1]);
	}

	for (int i=0;i<pts.size();i++) 
		points->InsertNextPoint(pts[i]->_x, pts[i]->_y, pts[i]->_z);


	vtkSmartPointer<vtkPolyData> surface = vtkSmartPointer<vtkPolyData>::New();
	surface->SetPoints(points);
	surface->SetLines(cells);

	vtkSmartPointer<vtkTubeFilter> axesTubes = vtkSmartPointer<vtkTubeFilter>::New();
	axesTubes->SetInput(surface);
	axesTubes->SetRadius(0.2);
	axesTubes->SetNumberOfSides(6);

	vtkSmartPointer<vtkPolyDataMapper> surfaceMapper = vtkSmartPointer<vtkPolyDataMapper>::New();
	surfaceMapper->SetInputConnection(axesTubes->GetOutputPort());

	vtkActor *surfaceActor = vtkActor::New();
	surfaceActor->SetMapper(surfaceMapper);
	
	// give color
	surfaceActor->GetProperty()->SetColor(r,g,b); 

	VizBoxCore::VBC_DisplayActor(flrwi, surfaceActor);

}

void VizBoxCore::VBC_centroidOfROIWithinPolyhedron(vtkBoxWidget* roi, vtkPolyData* poly, double* centroid_d)
{
	double xyz[3];
	int num_points_within_roi=0;
	
	vtkSmartPointer<vtkPlanes> planes = vtkSmartPointer<vtkPlanes>::New(); 
	roi->GetPlanes(planes); 
	centroid_d[0] = 0; centroid_d[1] =0; centroid_d[2]=0;

	for (int i=0;i<poly->GetNumberOfPoints();i++)
	{
		poly->GetPoint(i, xyz); 
		if (planes->FunctionValue(xyz) < 0)
		{
			// within the box 
			centroid_d[0]+=xyz[0];
			centroid_d[1]+=xyz[1];
			centroid_d[2]+=xyz[2];

			num_points_within_roi++;
		}

	}

	centroid_d[0] /= num_points_within_roi;
	centroid_d[1] /= num_points_within_roi;
	centroid_d[2] /= num_points_within_roi;
}

	// vtkPointLocator is a spatial search object to quickly locate points in 3D 
	// Quick search is possible by dividing a specified region of space into a regular array of rectangular buckets
	cout << "In VBC_TransferPointDataToTargetPoly, building point locator ... ";
	vtkSmartPointer<vtkPointLocator> point_locator = vtkSmartPointer<vtkPointLocator>::New();
	point_locator->SetDataSet(source); 
	point_locator->AutomaticOn(); 
	point_locator->BuildLocator();
	cout << "completed ... Preparing to transfer data .. some housekeeping .. ";
	
	source_scalars = vtkDoubleArray::SafeDownCast(source->GetPointData()->GetScalars());
	
	cout << "completed ... iterating point-by-point now .. target size = " <<target->GetNumberOfPoints()<<" pts, source size = " << source->GetNumberOfPoints() << " pts .. " ;
	cout << "source point data container size = " <<source_scalars->GetNumberOfTuples() << " values .."; 

	for (int i=0;i<target->GetNumberOfPoints();i++)
	{
		//cout << "i = " << i << ", "; 
		target->GetPoint(i, xyz_target); 
		closestPointID = -1;
	
		closestPointID = point_locator->FindClosestPoint(xyz_target); 
		//cout << "closest = " << closestPointID << ", "; 
		if (closestPointID >= 0 && closestPointID < target->GetNumberOfPoints()) 
		{
			target_scalars->InsertNextTuple1(source_scalars->GetTuple1(closestPointID));	// data from nearest neighbour 
		}else 
		{
			target_scalars->InsertNextTuple1(-100);
		}
	}
	
	cout << "completed ... housekeeping .. ";
	target->GetPointData()->SetScalars(target_scalars); 

	cout << "completed ... writing temp file target.vtk .. ";
	vtkSmartPointer<vtkPolyDataWriter> writer =  vtkSmartPointer<vtkPolyDataWriter>::New(); 
	writer->SetFileName("target.vtk"); 
	writer->SetInput(target); 
	writer->Update(); 
	cout << "completed\n\n" ;
}


void VizBoxCore::VBC_GetMeanOfTopNScalarsInVTKShell(vtkPolyData* poly, double topNPercent, double& mean, bool isCellData) 
{
	vtkSmartPointer<vtkFloatArray> scalars = vtkSmartPointer<vtkFloatArray>::New();
	if (isCellData) 
		scalars =  vtkFloatArray::SafeDownCast(poly->GetCellData()->GetScalars()); 
	else 
		scalars = vtkFloatArray::SafeDownCast(poly->GetPointData()->GetScalars()); 

	std::vector<double> temp; 
	for (int i=0;i<scalars->GetNumberOfTuples();i++)
	{
		if (scalars->GetTuple1(i) < 1e5) 
		temp.push_back(scalars->GetTuple1(i)); 
	}
	
	MathBox::GetMeanOfTopN(temp, topNPercent, mean); 
}

Get slice

Extract the nth slice in the x,y and z slice.

Usage 

getslice.exe -i <input> -o <output> -x -y -z n <x,y,z direction and nth slice>

void GetSlice(irtkGreyImage* img1, irtkGreyImage& img_crop, int in_x, int in_y, int in_z)
{
	int maxX, maxY, maxZ;
	maxX = img1->GetX(); 
	maxY = img1->GetY(); 
	maxZ = img1->GetZ(); 
	int n=0;

	//irtkGreyImage img_crop; 
	

		if (in_x > 0)
		{
			cout << "Cropping along x" << endl;
			img_crop = img1->GetRegion(in_x, 0,0, in_x+1, maxY, maxZ); 
		}
		else if (in_y > 0)
		{
			cout << "Cropping along y" << endl;
			img_crop = img1->GetRegion(0, in_y,0, maxX, in_y+1, maxZ); 
		}
		else if (in_z > 0)
		{
			cout << "Cropping along z" << endl;
			img_crop = img1->GetRegion(0, 0,in_z, maxX, maxY, in_z+1);
		}
	
	img_crop.Write("temp.gipl");
	
}



int main(int argc, char **argv)
{
	int optind;
	bool foundArgs=false;
	irtkGreyImage img1, img_crop;
	int x_y_z=1, nth_slice; 
	
	char* input_f1="", *input_f2="", *output_f="";
	double t1=1, mean, stdev, total_voxels, percent;

	if (argc >= 1) 
	{
		int optind=1;
		
		while ((optind < argc) && (argv[optind][0]=='-')) {
			
			string sw = argv[optind];
			
			if (sw == "-i") {
				optind++;
				input_f1 = argv[optind]; 
				foundArgs = true;	
			}
			
			
			else if (sw == "-o") {
				optind++;
				output_f = argv[optind]; 
			}
				
			else if (sw == "-x") 
			{
				x_y_z = 1;
				optind++;
				nth_slice = atoi(argv[optind]); 
				cout << "\n\nNote: Along x " << nth_slice << " slice"; 
			}

			else if (sw == "-y") 
			{
				x_y_z = 2;
				optind++;
				nth_slice = atoi(argv[optind]); 
				cout << "\n\nNote: Along y " << nth_slice << " slice" ; 
			}

			else if (sw == "-z") 
			{
				x_y_z = 3;
				optind++;
				nth_slice = atoi(argv[optind]); 
				cout << "\n\nNote: Along z " << nth_slice << " slice"; 
			}
			
			
			optind++; 
		}
	}
	
	if (foundArgs == false)
	{
		cout << "Usage: getslice.exe -i <img> -o <slice img> -x -y -z n <get nth slice along x,y or z direciton>" << endl; 
		exit(0); 
	}

	else
	{
		img1.Read(input_f1); 
		img_crop = img1;

		if (x_y_z == 1) 
			GetSlice(&img1, img_crop, nth_slice, 0, 0); 
		else if (x_y_z == 2) 
			GetSlice(&img1, img_crop, 0, nth_slice, 0); 
		else if (x_y_z == 3) 
			GetSlice(&img1, img_crop, 0, 0, nth_slice); 

		img_crop.Write(output_f);
	}

}

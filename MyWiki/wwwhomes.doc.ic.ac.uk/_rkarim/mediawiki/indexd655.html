void EndoSenseToolbox::_writeAblationPointsToFile()
{
	double prim_xyz[3]; 
	double all_xyz[3]; 
	bool isPrim = false; 
	vtkIdType prim=0, all=0;

	/* write the primary ablation points to text file */ 
	ofstream out; 
	out.open("prim_abPts.txt"); 
	out << "The primary ablation points as identified using geodesic distances, note index is the index of the point in the ablation tag file\n\n" << endl;
	
	vtkSmartPointer<vtkPoints> prim_points = vtkSmartPointer<vtkPoints>::New(); 
	vtkSmartPointer<vtkPoints> all_points = vtkSmartPointer<vtkPoints>::New();

	// it is important to have vertices (a point is just that has xyz co-ordinates, it wont display in vtk unless it is represented as a cell 
	// that can be displayed or viewable in paraview)
	vtkSmartPointer<vtkCellArray> prim_vertices = vtkSmartPointer<vtkCellArray>::New();
	vtkSmartPointer<vtkCellArray> all_vertices = vtkSmartPointer<vtkCellArray>::New();

	// and then the writers
	vtkSmartPointer<vtkPolyDataWriter> writer1 = vtkSmartPointer<vtkPolyDataWriter>::New(); 
	vtkSmartPointer<vtkPolyDataWriter> writer2 = vtkSmartPointer<vtkPolyDataWriter>::New(); 

	for (int j=0;j<_pressure_pt.size();j++)			// running through all the points 
	{
		isPrim = false;			
		// search in primary ablation point list if point with this index is in list 
		for (int k=0;k<_primary_ablation_pt.size() && !isPrim;k++) { 
			if (j == _primary_ablation_pt[k]) {
				isPrim = true; 
			}
		}
		
		if (isPrim) {
			prim_xyz[0] = _pressure_pt[j]->_x;  
			prim_xyz[1] = _pressure_pt[j]->_y;  
			prim_xyz[2] = _pressure_pt[j]->_z;  
			prim_points->InsertPoint(prim, prim_xyz); 
			prim_vertices->InsertNextCell(1, prim); 
			prim++;
		}
		
		all_xyz[0] = _pressure_pt[j]->_x;  
		all_xyz[1] = _pressure_pt[j]->_y;  
		all_xyz[2] = _pressure_pt[j]->_z;
		all_points->InsertPoint(all, all_xyz); 
		all_vertices->InsertNextCell(1, all); 
		all++
		
		out << "index = " << j << "\t\t(x,y,z) = " << "(" << _pressure_pt[j]->_x << "," << _pressure_pt[j]->_y << "," << _pressure_pt[j]->_z << ")" << endl;
	}
	out.close(); 
	vtkSmartPointer<vtkPolyData> primPoly = vtkSmartPointer<vtkPolyData>::New(); 
	vtkSmartPointer<vtkPolyData> allPoly = vtkSmartPointer<vtkPolyData>::New(); 
	primPoly->SetPoints(prim_points); 
	allPoly->SetPoints(all_points); 

	primPoly->SetVerts(prim_vertices); 
	allPoly->SetVerts(all_vertices); 
	
	writer1->SetInput(primPoly); 
	writer1->SetFileName("prim_abPts.vtk"); 
	writer1->Update();
	writer2->SetInput(allPoly); 
	writer2->SetFileName("all_abPts.vtk"); 
	writer2->Update();

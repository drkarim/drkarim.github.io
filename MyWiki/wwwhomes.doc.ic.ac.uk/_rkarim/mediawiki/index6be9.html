Usage

dice_3 Arguments:

  • 1st arg: filename of binary segmentation
  • 2nd arg: filename of staple 3D output
  • 3rd arg: filename of text file to append dice to
  • 3rd arg: foreground in binary
  • 4th arg: threshold for staple (>= t)\n\n

See also

  • STAPLE conversion from NRRD to binary image after thresholding
  • IRTK thresholding only if STAPLE is in GIPL or NIFTII format

Code

#if defined(_MSC_VER)
#pragma warning ( disable : 4786 )
#endif

#include "itkImageFileReader.h"
#include "itkImageFileWriter.h"
#include <itkImageIterator.h>
#include "fstream"

using namespace std; 
int main(int argc, char **argv)
{
	fstream out;

  
	char *filename1, *filename2, *filename3;
	double f_value1, f_value2; 
	double x_and_y=0, x=0, y=0, n=0, x_or_y=0, temp1, temp2, total=0;

	if (argc != 6)
	{
		cout << "\n\nUsage: dice_3 \n\n\tArguments:"
			"\n1st arg: filename of binary segmentation" 
			"\n2nd arg: filename of staple 3D output" 
			"\n3rd arg: filename of text file to append dice to" 
			"\n3rd arg: foreground in binary" 
			"\n4th arg: threshold for staple (>= t)\n\n"; 
		exit(1);	
	}

	filename1 = argv[1];
	argv++;
	argc--;
	filename2 = argv[1];
	argv++;
	argc--;
	filename3 = argv[1]; 
	argv++; 
	argc--; 
	f_value1 = atof(argv[1]); 
	argv++;
	argc--;
	f_value2 = atof(argv[1]);

	out.open (filename3, fstream::in | fstream::out | fstream::app); 
	typedef double     PixelType;
	const   unsigned int        Dimension = 3;
	typedef itk::Image< PixelType, Dimension >    ImageType;
	typedef itk::ImageFileReader< ImageType >  ReaderType;
	typedef itk::ImageFileWriter< ImageType >  WriterType;
	typedef itk::ImageRegionIterator< ImageType>       IteratorType; 

	ReaderType::Pointer reader1 = ReaderType::New();
	ReaderType::Pointer reader2 = ReaderType::New();

	reader1->SetFileName( filename1  );
	reader1->Update();
	reader2->SetFileName( filename2  );
	reader2->Update();
	
	ImageType::Pointer im1 = reader1->GetOutput();
	ImageType::Pointer im2 = reader2->GetOutput();

	IteratorType  it1( im1, im1->GetRequestedRegion() );
	IteratorType  it2( im2, im2->GetRequestedRegion() );
	  
	it1.GoToBegin();
	it2.GoToBegin();

	while( !it1.IsAtEnd() && !it2.IsAtEnd() )
	{
		temp1 = it1.Get(); 
		temp2 = it2.Get(); 
		if (temp1 == f_value1 && temp2 > f_value2)
		   x_and_y++; 
		if (temp1 == f_value1)
		   x++; 
		if (temp2 > f_value2)
		   y++; 
		if (temp1 == f_value1 || temp2 > f_value2)
		{
		   x_or_y++; 
		}
		++it1; ++it2; 
		total++; 
	}
	
	double d_c; 
	d_c = (2*(x_and_y))/(x+y);
	d_c = d_c; 
	out << d_c << std::endl; 
	out.close(); 

	std::cout << d_c << std::endl;


}

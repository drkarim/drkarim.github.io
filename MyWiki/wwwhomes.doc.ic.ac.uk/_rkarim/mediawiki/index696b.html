Version

This version uses some old binaries of VTK and ITK, but the release version of this code must run without any external binaries (other than the VTK dlls) on Win7. There is also another Magic2 that has separate features and functions

Usage

  • For generating pixel intensities to file: 
magic -i file_name.gipl -o output.txt --nz x

void outputNonZeroIntensityToFile(irtkRealImage* img, ofstream& out, double padding)
{
	irtkRealPixel* i_p; 
	i_p = img->GetPointerToVoxels(); 

	for (int i=0;i<img->GetNumberOfVoxels();i++)
	{
		if (*i_p > padding) 
			out << *i_p << endl; 
		i_p++;
	}
}

void combineMarroucheWallWithGraphCuts(irtkRealImage* wall, irtkRealImage* gcuts, irtkRealImage* out)
{
	irtkRealPixel* aw_p; 
	irtkRealPixel* gc_p; 
	irtkRealPixel* out_p; 

	aw_p = wall->GetPointerToVoxels(); 
	gc_p = gcuts->GetPointerToVoxels();
	out_p = out->GetPointerToVoxels(); 


	for (int i=0;i<wall->GetNumberOfVoxels();i++) 
	{
		if (*aw_p == 0) {
			
			*out_p = 0; 
		}
		out_p++; gc_p++; aw_p++;
	}

}

void trimImage(irtkRealImage* img, irtkRealImage* out, char* output_filename)
{
	irtkRealImage temp, out_temp; 
	temp =* img; 
	out_temp = temp; 
	double min_x=1e9, min_y=1e9, min_z=1e9; 
	double max_x=-1, max_y=-1, max_z=-1; 

	for (int i=0;i<img->GetX();i++)
	{
		for (int j=0;j<img->GetY();j++)
		{
			for (int k=0;k<img->GetZ();k++) 
			{
				if (img->Get(i,j,k) > 0)
				{
					if (min_x > i) min_x = i; 
					if (min_y > j) min_y = j; 
					if (min_z > k) min_z = k; 

					if (max_x < i) max_x = i; 
					if (max_y < j) max_y = j; 
					if (max_z < k) max_z = k; 
					
				}
			}
		}
	}
	cout << "(minx, miny, minz) = (" << min_x << "," << min_y << "," << min_z << ")\n"; 
	cout << "(maxx, maxy, maxz) = (" << max_x << "," << max_y << "," << max_z << ")\n"; 
	out_temp = temp.GetRegion(min_x, min_y, min_z, 0, max_x, max_y, max_z, img->GetT()); 
	out_temp.Write(output_filename);
	
}

void getFileNames(vector<string> observer_abb, int num_p, vector<string>& filenames)
{
	stringstream ss; 
	for (int i=0;i<observer_abb.size();i++)
	{
		for (int j=0;j<num_p;j++)
		{
			ss << j+1 << "_" << observer_abb[i] << ".gipl"; 
			filenames.push_back(ss.str()); 
			//cout << ss.str() << endl;
			ss.str(""); 
		}
		
	}
}

void getFileNames(string prefix, int start, int stop, vector<string>& filenames)
{
	stringstream ss; 
	for (int i=start;i<=stop;i++)
	{
		ss << prefix << "_" << i << ".gipl"; 
		filenames.push_back(ss.str()); 
		ss.str(""); 
		
		
	}
}



void probAtlasCompute(vector<string> filenames, char* prob_atlas_filename) 
{
	irtkRealImage prob_atlas; 
	irtkRealImage man_seg; 
	
	
	prob_atlas.Read(filenames[0].c_str()); 
	irtkRealPixel *atlas_p, *man_seg_p; 
	atlas_p = prob_atlas.GetPointerToVoxels(); 
	for (int i=0;i<prob_atlas.GetNumberOfVoxels();i++)
	{
		*atlas_p = 0;		// reset
		atlas_p++;
	}

	for (int i=0;i<filenames.size();i++)
	{
		cout << "\nReading " << filenames[i].c_str() << endl; 
		man_seg.Read(filenames[i].c_str()); 
		
		atlas_p = prob_atlas.GetPointerToVoxels(); 
		man_seg_p = man_seg.GetPointerToVoxels(); 

		for (int j=0;j<prob_atlas.GetNumberOfVoxels();j++)
		{
			if (*man_seg_p > 0) {
				*atlas_p += ((double)1/(double)filenames.size())*100; 
			}
			atlas_p++;
			man_seg_p++; 
		}
	
	}
	cout << "\nExiting ... " << endl;
	
	prob_atlas.Write(prob_atlas_filename); 
	
}

void createScarProbAtlas(string prefix, int start, int stop, char* prob_atlas_filename)
{
	vector<string> filenames; 
	getFileNames(prefix, start, stop, filenames); 
	probAtlasCompute(filenames, prob_atlas_filename); 
}

void createScarProbAtlas(int num_p, vector<string> observer_abb, char* prob_atlas_filename) 
{
	vector<string> filenames; 
	getFileNames(observer_abb, num_p, filenames); 
	probAtlasCompute(filenames, prob_atlas_filename); 

}

void createMask(irtkRealImage *irtk_img1, irtkRealImage *irtk_img2, int t) 
{
	irtkRealPixel* img1_p, *img2_p; 

	img1_p = irtk_img1->GetPointerToVoxels(); 
	img2_p = irtk_img2->GetPointerToVoxels(); 

	for (int i=0;i<irtk_img1->GetNumberOfVoxels();i++)
	{
		if (*img1_p != t)  { 
			
			*img2_p = 0;
		}
		img2_p++; 
		img1_p++; 
	
	}
}

void filterPixels(irtkRealImage *irtk_img1, int t, int w) 
{
	irtkRealPixel* img1_p, *img2_p; 

	img1_p = irtk_img1->GetPointerToVoxels(); 
	
	for (int i=0;i<irtk_img1->GetNumberOfVoxels();i++)
	{
		if (*img1_p == t)  { 
			
			*img1_p = w;
		}
		else 
			*img1_p = 0;
		img1_p++; 
	
	
	}
}

void writeHistogram(irtkRealImage *irtk_img1, irtkRealImage *mask_img2, int t, ofstream& outputfile)
{
	irtkRealPixel* img1_p, *img2_p; 

	img1_p = irtk_img1->GetPointerToVoxels(); 
	if (t != -1) img2_p = mask_img2->GetPointerToVoxels(); 
	
	for (int i=0;i<irtk_img1->GetNumberOfVoxels();i++)
	{
		if (t!= -1 && *img2_p == t )  { 
			
			outputfile << *img1_p << endl;
		} 
		else if (t==-1)
		{
			outputfile << *img1_p << endl;
		}

		if (t != -1) img2_p++; 
		img1_p++; 
	
	}

}

int main(int argc, char *argv[])
{
	char* input_f="", *output_f="", *input_f2="", *observer_abb="", *working_dir="", *prefix=""; 
	int np, start, stop, pixel_value, pixel_value_in_output;
	bool foundArgs = false; 
	int option;
	double padding=0;		// do not change this default value or else the default value in nonzero pixel output is changed  
	irtkRealImage irtk_img1, irtk_img2, out_img; 
	if (argc >= 1)
	{
		for (int i = 1; i < argc; i++) {
			if (i + 1 != argc) { 
				 if (string(argv[i]) == "-i") {
					input_f = argv[i+1]; 
					foundArgs = true;	
				 } else if (string(argv[i]) == "-o") {
					output_f = argv[i+1];	
					foundArgs = true; 
				 }
				 else if (string(argv[i]) == "-p") {
					padding = atof(argv[i+1]);
					foundArgs = true; 
				 }
				 else if (string(argv[i]) == "-i2") {
					 input_f2 = argv[i+1]; 
					 foundArgs = true; 
				 }
				 else if (string(argv[i]) == "-np") {
					 np=atoi(argv[i+1]);
					 foundArgs = true; 
				 }
				  else if (string(argv[i]) == "-prefix") {
					 prefix=(argv[i+1]);
					 foundArgs = true; 
				 }
				   else if (string(argv[i]) == "-start") {
					 start=atoi(argv[i+1]);
					 foundArgs = true; 
				 }
				    else if (string(argv[i]) == "-stop") {
					 stop=atoi(argv[i+1]);
					 foundArgs = true; 
				 }
				else if (string(argv[i]) == "-fp") {
					 observer_abb = argv[i+1];
					 foundArgs = false; 
				}
				 else if (string(argv[i]) == "-t") {
					pixel_value = atoi(argv[i+1]);
					foundArgs = true; 
				 }
				else if (string(argv[i]) == "--h") {
					 cout << "\n\nShowing help .. " << endl;
					foundArgs = false; 
				 }	
				 else if (string(argv[i]) == "-w") {
					pixel_value_in_output = atoi(argv[i+1]);
					foundArgs = true; 
				 }
				


				 // all method switches here 
				else if (string(argv[i]) == "--nz") {
					option = 1; 
					foundArgs = true; 
				 }
				else if (string(argv[i]) == "--sm") {
					option = 2; 
					foundArgs = true; 
				}
				else if (string(argv[i]) == "--tm") {
					option = 3; 
					foundArgs = true; 
				}
				else if (string(argv[i]) == "--pb") {
					option = 4; 
					foundArgs = true; 
				}
				else if (string(argv[i]) == "--av") {
					option = 5; 
					foundArgs = true; 
				}
				else if (string(argv[i]) == "--ms") {
					option = 6; 
					foundArgs = true; 
				}
				else if (string(argv[i]) == "--fp") {
					option = 7; 
					foundArgs = true; 
				}
				else if (string(argv[i]) == "--hs") {
					option = 8; 
					foundArgs = true; 
				}
				else if (string(argv[i]) == "--diff") {
					option = 9;
					foundArgs = true; 
				}
				
			}
		}
	}

	if (foundArgs == false)
		{
			cout << "\n\nUsage: magic \n\n\tArguments:"
				"\n\n\tFunction: Get non-zero pixel intensities to file (switch = --nz x)\n\t1) -i : input image file in nii, gipl, etc.\n\t2) -o : output text file\n\t3) -p : padding [optional] (if you want intensities above a certain value and not zero)"
				"\n\n\tFunction: Combine with marrouche wall (switch = --sm x)" 
				"\n\n\tFunction: Trim image (switch = --tm x)\n\t1) -i : input image in nii gipl, etc\n\t2) -o : output image" 
				"\n\n\tFunction: Create scar Prob atlas (switch = --pb x)\n\t1) -np : Number of patients\n\t2) -fp : file prefixes possibly observer initials \n\t\tusing comma-delimited list (for example ab,rk,km)\n\t 3) -o : output atlas name\n" 
				"\n\n\tFunction: Compute average segmentation (switch = --av x)\n\t1) -prefix the segmentation filenames begin with ... \n\t2) -start : the starting number in filename \n\t3) -stop : the ending number in filename\n\t 3) -o : output file name\n" 
				"\n\n\tFunction: Compute mask (switch = --ms x)\n\t1) -i : Mask \n\t2) -i2 : Greyscale image \n\t3) -o : output of the masked file \n\t4) -t : pixel value in mask\n" 
				"\n\n\tFunction: Filter pixels (switch = --fp x)\n\t1) -i : Greyscale image \n\t2) -o : output of the masked file \n\t4) -t : pixel value to filter\n\t5) -w : value of filtered pixels in output image" 
				"\n\n\tFunction: Outputs pixel intensities to file (switch = --hs x)\n\t1) -i : Greyscale image \n\t2) -i2 : The mask file \n\t3) -t : pixel value in mask \n\t4) -o : output text file" 
				"\n\n\tFunction: Outputs Ensite DIFF file foramt (switch = --diff x)\n\t1) -i : VTK shell \n\t2) -o : output diff file name" 
				"\n\n\n\t--h for help" << endl;
			exit(0);
		}

	if (option == 1) 
	{
		// nonzero pixel intensity output
		irtk_img1.Read(input_f); 		
		ofstream out; 
		out.open(output_f); 
		outputNonZeroIntensityToFile(&irtk_img1, out, padding); 
		out.close(); 
	}
	else if (option == 2)
	{
		irtk_img1.Read(input_f); 
		irtk_img2.Read(input_f2); 
		out_img = irtk_img2; 
		
		combineMarroucheWallWithGraphCuts(&irtk_img1, &irtk_img2, &out_img); 
		out_img.Write(output_f); 

	}
	else if (option == 3) 
	{
		irtk_img1.Read(input_f); 
		//out_img = irtk_img1; 

		trimImage(&irtk_img1, &out_img, output_f); 
		
	}
	else if (option == 4) 
	{
		vector<string> observer_list; 
		string observers_str;
		stringstream ss; 
		char *pch; 
		cout << "You specified observers as: " << observer_abb << endl;
		pch = strtok(observer_abb, ","); 
		while (pch != NULL) 
		{
			ss << pch;
			observer_list.push_back(ss.str()); 
			cout << ss.str() << endl;
			pch = strtok(NULL, ","); 
			ss.str(""); 
		}
		createScarProbAtlas(np, observer_list, output_f);
	}
	else if (option == 5) 
	{
		createScarProbAtlas(prefix, start, stop, output_f); 
	}
	else if (option == 6) 
	{
		irtk_img1.Read(input_f); 
		irtk_img2.Read(input_f2); 
		//out_img = irtk_img2; 
		
		cout << "pixel value = " << pixel_value << endl;
		createMask(&irtk_img1, &irtk_img2, pixel_value); 
		irtk_img2.Write(output_f); 
	}
	else if (option == 7)
	{
		irtk_img1.Read(input_f); 
		//irtk_img2.Read(input_f2); 
		//out_img = irtk_img2; 
		
		cout << "pixel value to fitler = " << pixel_value << endl;
		cout << "pixel value in output = " << pixel_value_in_output << endl;
		filterPixels(&irtk_img1, pixel_value, pixel_value_in_output); 
		irtk_img1.Write(output_f); 
	}
	else if (option == 8)
	{
		irtk_img1.Read(input_f); 
		if (pixel_value != -1) irtk_img2.Read(input_f2); 
		ofstream out; 
		out.open(output_f); 

		cout << "pixel value queried in mask = " << pixel_value << endl; 
		writeHistogram(&irtk_img1, &irtk_img2, pixel_value, out); 
		out.close();
		
	}
	else if (option ==9) 
	{
		vtkSmartPointer<vtkPolyDataReader> reader = vtkSmartPointer<vtkPolyDataReader>::New(); 
		reader->SetFileName(input_f); 
		reader->Modified(); 
		vtkCleanPolyData *cleaner = vtkCleanPolyData::New();
		cleaner->SetInput(reader->GetOutput());
		cleaner->Modified();
		cleaner->Update();
		vtkPolyData *target = vtkPolyData::New();
		target = cleaner->GetOutput();
		target->Update();

		EnsiteAdapter* ensiteTool = new EnsiteAdapter(); 
		ensiteTool->DifWriter(output_f, target);
	}

}

Introduction

Computes the boundaries of a binary mask. Also see boundary error metric

Usage 

boundary.exe -i1 <bin. mask gipl> -o <output bondary gipl> -t <n-pixel thick>


Code


#include <irtkImage.h>
#include <irtkEuclideanDistanceTransform.h>

#include <stdio.h>    // to get "printf" function
#include <stdlib.h>   // to get "free" function

void computeEDT(irtkRealImage* img1, irtkRealImage* edt_img) 
{

	irtkRealImage img3; 
	img3 =* img1; 
	irtkRealPixel *p; 
	p = img3.GetPointerToVoxels(); 
	
	// reverse the segmentation mask for EDT
	for (int i=0;i<img3.GetNumberOfVoxels();i++) 
	{
		if (*p>0)
			*p = 0; 
		else
			*p = 1; 
		p++; 
	}

	irtkEuclideanDistanceTransform<irtkRealPixel> *edt = NULL;
	edt = new irtkEuclideanDistanceTransform<irtkRealPixel>(irtkEuclideanDistanceTransform<irtkRealPixel>::irtkDistanceTransform3D);
	edt->SetInput(&img3);
	edt->SetOutput(edt_img);
	edt->Run();
	//edt_img->Write("edt.gipl");
}


// la_seg is needed to label anything outside as no wall 
void GetBoundary(irtkRealImage* edt_img, irtkRealImage* la_seg, irtkRealImage* at_wall, double atwall_thickness)
{
	irtkRealPixel *edt_p, *atwall_p, *laseg_p; 
	edt_p = edt_img->GetPointerToVoxels(); 
	atwall_p = at_wall->GetPointerToVoxels();
	laseg_p = la_seg->GetPointerToVoxels();
	double edt; 

	for (int i=0;i<edt_img->GetNumberOfVoxels();i++)
	{
		//if (*edt_p > 0)
			edt = sqrt(*edt_p);
			//if (edt > 0) cout << *edt_p << "," << edt << "\n";
		//else 
		//	edt = -1;

		if (edt > atwall_thickness){
			//cout << edt << "\t";
			*atwall_p = 0;
		}

		if (*laseg_p == 0)
			*atwall_p = 0;

		edt_p++; 
		atwall_p++; 
		laseg_p++;
	}

	//at_wall->Write("at_wall.gipl"); 
}

void Binarize(irtkRealImage* at_wall)
{
	irtkRealPixel *p; 
	p = at_wall->GetPointerToVoxels(); 
	for (int i=0;i<at_wall->GetNumberOfVoxels();i++)
	{
		if (*p>0) 
		{ 
			*p = 1;
		}
		p++; 
	}
	//at_wall->Write("at_wall.gipl");
}


int main(int argc, char **argv)
{
	int optind;
	bool foundArgs=false;
	irtkRealImage img1;
	irtkRealImage edt_img1;
	irtkRealImage boundary_img1;

	char* input_f1="", *output_f="", *input_f2=""; 
	double t1=1, mean, stdev, total_voxels, percent;

	if (argc >= 1) 
	{
		int optind=1;
		
		while ((optind < argc) && (argv[optind][0]=='-')) {
			
			string sw = argv[optind];
			
			if (sw == "-i") {
				optind++;
				input_f1 = argv[optind]; 
				foundArgs = true;	
			}
			
			else if (sw == "-t") {
				optind++;
				t1 = atoi(argv[optind]); 
				foundArgs = true;	
			}


			else if (sw == "-o") {
				optind++;
				output_f = argv[optind]; 
			}
			
			
			
			optind++; 
		}
	}
	
	if (foundArgs == false)
	{
		cout << "Usage: boundaryerror.exe -i1 <bin. mask gipl> -o <output bondary gipl> -t <n-pixel thick>" << endl; 
		exit(0); 
	}

	else
	{
		img1.Read(input_f1); 
		//img2.Read(input_f2);

		boundary_img1 = img1; 
		//boundary_img2 = img2;

		computeEDT(&img1, &edt_img1);
		//computeEDT(&img2, &edt_img2);
		
		GetBoundary(&edt_img1, &img1, &boundary_img1, t1+1);		// for 1 pixel thickness offset by 1 
		//GetBoundary(&edt_img2, &img2, &boundary_img2, t1);

		Binarize(&boundary_img1);
		//Binarize(&boundary_img2);
		
		boundary_img1.Write(output_f); 

	}

}

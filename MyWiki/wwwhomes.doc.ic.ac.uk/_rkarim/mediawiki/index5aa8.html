Code

Program 1

NOTE: Program 1 does not allow a to define a constraint on the background (or negative) pixels. Program 2 allows this. 

#include <irtkImage.h>


int main(int argc, char **argv)
{
		double x_and_y=0, x=0, y=0, n=0, x_or_y=0, _f_image1, _f_image2; 

		// variables for sensitivity and specificty 
		double false_pos=0, false_neg=0, true_pos=0, true_neg=0,spec=0, sens=0, pos_pred=0;
       char *filename1, *filename2, *filename3, *output;
       irtkGreyPixel *ptr1; 
	   irtkRealPixel *ptr2;
	   irtkRealPixel *ptr3;
	   bool isWriteToFile = false;
	   if (argc < 3) { cerr << "\nComputes specificity and sensitivity and Dice\n\nUsage: dice <test_image> <ground_truth> <forground_in_test>"
		   "<foreground_in_groundtruth> <output-file-to-write-dice>\n\nOutputs dice-efficient. \n\nNote: This is different from the other dice program"
		   "as you have to specify what intensity value denotes foreground in your images.\n\nAlso, you can specify (optional) an output file to write the dice values to"  << endl; exit(1); }
       // get files to subtract
	   if (argc > 5) { 
		   output = argv[4];		// write dice to file 
		   isWriteToFile = true; 
	   }
       filename1 = argv[1];
       argv++;
       argc--;
       filename2 = argv[1];
       argv++;
       argc--;
	   _f_image1 = atoi(argv[1]); 
	   argv++;
       argc--;
	   _f_image2 = atoi(argv[1]); 
	   argv++;
       argc--;
	   //filename3 = argv[1]; 
       output = argv[1];

       // read imaged
       irtkGreyImage input1;
       irtkRealImage input2;
	   irtkRealImage input3;

       double d;
	   double d_c, o_p1, o_p2; 

       input1.Read(filename1);
       input2.Read(filename2);
	   //input3.Read(filename3); 

       ptr1 = input1.GetPointerToVoxels();
       ptr2 = input2.GetPointerToVoxels();
	   //ptr3 = input3.GetPointerToVoxels();

       cout << "number of voxels " << input1.GetNumberOfVoxels() << endl;
       cout << "number of voxels " << input2.GetNumberOfVoxels() << endl;
	   //cout << "number of voxels " << input3.GetNumberOfVoxels() << endl;

	   if (input1.GetNumberOfVoxels()-input2.GetNumberOfVoxels() == 0)

	   {
			for (int i=0;i<input1.GetNumberOfVoxels();i++)
		   {
			   if (*ptr1 == _f_image1 && *ptr2 == _f_image2)
				   x_and_y++; 
			   if (*ptr1 == _f_image1)
				   x++; 
			   if (*ptr2 == _f_image2)
				   y++; 
			   if (*ptr1 ==  _f_image1 || *ptr2 == _f_image2)
			   {
				   x_or_y++; 
			   }
			   if (*ptr1 == _f_image1 && *ptr2 == 0) 
			   {
				   false_pos++; 
			   }
			   if (*ptr1 == 0 && *ptr2 == _f_image2)
			   {
				   false_neg++; 
			   }
			   if (*ptr1 == 0 && *ptr2 == 0)
			   {
					true_neg++; 
			   }
			   ptr1++; ptr2++;  n++;
		   }
			
			
			d_c = (2*((double)x_and_y))/((double)x+(double)y); 
			o_p1 = ((double)x_and_y/(double)x)*100;
			o_p2= ((double)x_and_y/(double)y)*100;
			sens = (x_and_y/(x_and_y+false_neg))*100; 
			spec = (true_neg/(true_neg+false_pos))*100; 
			pos_pred = (x_and_y/(x_and_y + false_pos))*100; 
over_seg_meas = ((x_and_y+false_pos)/(x_and_y+false_neg))*100;

			if (!isWriteToFile) { 
				cout << "\nDice co-efficients:\n=====================\n" << "Total voxels = " << n << "\nImage1 and Image2 overlap voxels = " << x_and_y << "\nImage1 foreground voxels = " << x << "\nImage2 foreground voxels = " << y << endl; 
				cout << "Dice co-efficient = " << d_c << endl; 
				cout << "Overlap as a percentage of image1 voxels = " << o_p1 << endl; 
				cout << "Overlap as a percentage of image2 voxels = " << o_p2 << endl;
				cout << "\n\n==================================\n"; 
				double o_m = (double)x_and_y/(double)x_or_y; 
				cout << "Volumetric overlap metric (i.e. a_and_b / a_or_b)\nImage1 AND Image2 = " << x_and_y << "\nImage1 OR Image2 = "<< x_or_y << "\nOverlap metric = " << o_m << endl;
				cout << "Sensitivity and Specificity tests:\n" << "1. Sensitivity = " << sens << "\n2. Specificity = " << spec << "\n3. Positive"  
					" predicative value = " << pos_pred << endl;
cout << "4. Over-segmentation measure = " << over_seg_meas << endl;
			}
			else
			{
				ofstream fileIO; 
				fileIO.open(output, ios::app); 
				fileIO << d_c << "\t" << sens << "\t" << spec << "\t" << pos_pred << endl;
				fileIO.close(); 
				cout << "Dice, sensitivity and specifity written to file as\nDice\tSpecitifty\tSensitivity\tPositive predictive" << endl; 
			}
			

	   }
	   else
	   {
		   cout << "images with unequal dimensions, cannot compute the dice, must be equal dimensions" << endl;
	   }
}

#include <irtkImage.h>


int main(int argc, char **argv)
{
		double x_and_y=0, x=0, y=0, n=0, x_or_y=0, _f_image1, _f_image2; 

		// variables for sensitivity and specificty 
		double false_pos=0, false_neg=0, true_pos=0, true_neg=0,spec=0, sens=0, pos_pred=0;
       char *filename1, *filename2, *filename3, *output;
       irtkGreyPixel *ptr1; 
	   irtkRealPixel *ptr2;
	   irtkRealPixel *ptr3;
	   bool isWriteToFile = false;
	   if (argc < 3) { cerr << "\nComputes specificity and sensitivity and Dice\n\nUsage: dice <test_image> <ground_truth> <forground_in_test>"
		   "<foreground_in_groundtruth> <search_space_if_any_for_specificity> <output-file-to-write-dice>\n\nOutputs dice-efficient. \n\nNote: This is different from the other dice program"
		   "as you have to specify what intensity value denotes foreground in your images.\n\nAlso, you can specify (optional) an output file to write the dice values to"  << endl; exit(1); }
       // get files to subtract
	   if (argc > 6) { 
		   output = argv[5];		// write dice to file 
		   isWriteToFile = true; 
	   }
       filename1 = argv[1];
       argv++;
       argc--;
       filename2 = argv[1];
       argv++;
       argc--;
	   _f_image1 = atoi(argv[1]); 
	   argv++;
       argc--;
	   _f_image2 = atoi(argv[1]); 
	   argv++;
       argc--;
	   filename3 = argv[1]; 
	   argv++; 
	   argc--;
       output = argv[1];

       // read imaged
       irtkGreyImage input1;
       irtkRealImage input2;
	   irtkRealImage input3;

       double d;
	   double d_c, o_p1, o_p2; 

       input1.Read(filename1);
       input2.Read(filename2);
	   input3.Read(filename3); 

       ptr1 = input1.GetPointerToVoxels();
       ptr2 = input2.GetPointerToVoxels();
	   ptr3 = input3.GetPointerToVoxels();

       cout << "number of voxels " << input1.GetNumberOfVoxels() << endl;
       cout << "number of voxels " << input2.GetNumberOfVoxels() << endl;
	   cout << "number of voxels " << input3.GetNumberOfVoxels() << endl;

	   if (input1.GetNumberOfVoxels()-input2.GetNumberOfVoxels() == 0 && input1.GetNumberOfVoxels()-input3.GetNumberOfVoxels() == 0)

	   {
			for (int i=0;i<input1.GetNumberOfVoxels();i++)
		   {
			   if (*ptr1 == _f_image1 && *ptr2 == _f_image2)
				   x_and_y++; 
			   if (*ptr1 == _f_image1)
				   x++; 
			   if (*ptr2 == _f_image2)
				   y++; 
			   if (*ptr1 ==  _f_image1 || *ptr2 == _f_image2)
			   {
				   x_or_y++; 
			   }
			   if (*ptr1 == _f_image1 && *ptr2 == 0) 
			   {
				   false_pos++; 
			   }
			   if (*ptr1 == 0 && *ptr2 == _f_image2 && *ptr3 > 0) 
			   {
				   false_neg++; 
			   }
			   if (*ptr1 == 0 && *ptr2 == 0 && *ptr3 > 0) 
			   {
					true_neg++; 
			   }
			   ptr1++; ptr2++;  ptr3++; n++;
		   }
			
			
			d_c = (2*((double)x_and_y))/((double)x+(double)y); 
			o_p1 = ((double)x_and_y/(double)x)*100;
			o_p2= ((double)x_and_y/(double)y)*100;
			sens = (x_and_y/(x_and_y+false_neg))*100; 
			spec = (true_neg/(true_neg+false_pos))*100; 
			pos_pred = (x_and_y/(x_and_y + false_pos))*100; 

			if (!isWriteToFile) { 
				cout << "\nDice co-efficients:\n=====================\n" << "Total voxels = " << n << "\nImage1 and Image2 overlap voxels = " << x_and_y << "\nImage1 foreground voxels = " << x << "\nImage2 foreground voxels = " << y << endl; 
				cout << "Dice co-efficient = " << d_c << endl; 
				cout << "Overlap as a percentage of image1 voxels = " << o_p1 << endl; 
				cout << "Overlap as a percentage of image2 voxels = " << o_p2 << endl;
				cout << "\n\n==================================\n"; 
				double o_m = (double)x_and_y/(double)x_or_y; 
				cout << "Volumetric overlap metric (i.e. a_and_b / a_or_b)\nImage1 AND Image2 = " << x_and_y << "\nImage1 OR Image2 = "<< x_or_y << "\nOverlap metric = " << o_m << endl;
				cout << "Sensitivity and Specificity tests:\n" << "1. Sensitivity = " << sens << "\n2. Specificity = " << spec << "\n3. Positive"  
					" predicative value = " << pos_pred << endl;
			}
			else
			{
				ofstream fileIO;

				fileIO.close();

	   }

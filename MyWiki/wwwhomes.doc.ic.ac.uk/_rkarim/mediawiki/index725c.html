If you are looking for how to register CARTO data to MRI and vice versa see here

Usage

Within matlab this is the command: 

TriRepToVTK(<matlab trirep file>, <output vtk file>)

Note:This will only on surface meshes with polygons that are strictly triangles. It is unclear what the cell_list_size parameter expects in VTK file formats. Here is an answer to this question on the VTK users forum. In this code, the cell_list_size is simply (4*num_of_polygons). From the forum it also becomes clear that this should be num_cells * ( num_points_in_each_cell + 1 )

Code 

function TriRepToVTK(i_f, o_file)
    load(i_f)
    V = userdata.surface.triRep.X
    P = userdata.surface.triRep.Triangulation 
    D = userdata.surface.act_bip
    Vs = size(V)
    Ps = size(P)
    Ds = size(D)
    fid = fopen(o_file, 'w'); 
    WriteHeader(fid, Vs(1,1));
    
    % writing vertex information 
    for i=1:Vs(1,1)
        for j=1:Vs(1,2)
            fprintf(fid, '%f ', V(i,j));
        end
        if mod(i, 3) == 0 && i > 0  
            fprintf(fid, '\n'); 
        end
    end
    
    % writing triangulation information 
    fprintf(fid, '\n\nPOLYGONS %d %d\n', Ps(1,1), Ps(1,1)*4);
    for i=1:Ps(1,1)
        fprintf(fid, '3 '); 
        for j=1:Ps(1,2)
           fprintf(fid, '%d ', P(i,j)-1);           %polygon indices in matlab from 0
        end
        fprintf(fid, '\n');
    end

    %write point data information 
    fprintf(fid, '\n\nPOINT_DATA %d\n', Ds(1,1));
    fprintf(fid, 'SCALARS scalars float\n'); 
    fprintf(fid, 'LOOKUP_TABLE default\n'); 
    
    for i=1:Ds(1,1)
        if isnan(D(i,1)) == 1
            D(i,1) = 0;
        end
        fprintf(fid, '%3.3f ', D(i,1));

        
        if mod(i, 9) == 0 & i > 0
            fprintf(fid, '\n'); 
        end
    end
    
end

function WriteHeader(fid, num_points)
    vtk_header = ['# vtk DataFile Version 3.0\nvtk output\nASCII\nDATASET POLYDATA\nPOINTS ', num2str(num_points), ' float\n']; 
    fprintf(fid, vtk_header);
end

Introduction

Investigated following artefacts:

  • Aortic wall enhancement
  • Navigator artefact in RPV

Tasks

  • Manually labelled artefacts as 1 or 2 labels in each image
  • Computed volume for each segmentation in artefacted region

Analyzing artefacts

  • Usage: 
[N,A] = analyze_artefacts;
draw_multiple_bars
  • Code for analyzing artefact data generated by script above: 
function [Nav, Aor] = analyze_artefacts()
  
    subfolder = 'analysis_data';
    which_data_nav = {'_nav_pre', '_nav_post'}; 
    which_data_aortic = {'_aortic_pre', '_aortic_post'}; 
    extension = '.txt'; 
    inst = {'imperial', 'mevis', 'SUNNY', 'HARVARD', 'yale', 'kcl', 'uta', 'UTB'}; 
    %inst_post = {'imperial', 'mevis', 'SUNNY', 'HARVARD', 'yale', 'kcl', 'uta', 'UTB'}; 
    
    Nav = []; %Navigator artefact mean and std 
    Aor = []; %aortic artefact mean and std
    
    %first collect the navigator data of both pre and post 
    for i=1:size(inst,2)
        B = []; 
        for j=1:size(which_data_nav, 2)
            fn = [subfolder '\' inst{i} which_data_nav{j} extension]
            A = load(fn);
            B = [B; 100*(A(:,3)./A(:,4))];
        end
        T = isnan(B); 
        means = mean(B(~T));
        stdev = std(B(~T));
        Nav = [Nav [means; stdev]];
    end
    
    %Next collect aortic data of both pre and post
    for i=1:size(inst,2)
        B = []; 
        for j=1:size(which_data_aortic, 2)
            fn = [subfolder '\' inst{i} which_data_aortic{j} extension]
            A = load(fn);
            B = [B; 100*(A(:,3)./A(:,4))];
        end
        T = isnan(B); 
        means = mean(B(~T));
        stdev = std(B(~T));
        Aor = [Aor [means; stdev]];
    end
    
    
    
    
end

Plotting stacked bars

  • Function draw_multiple_bars 
meanDat = []; 
stdvDat = [];
meanDat(1,:) = N(1,:)
meanDat(2,:) = A(1,:)
stdvDat(1,:) =N(2,:); 
stdvDat(2,:) =A(2,:);

x = 1:size(meanDat,2);

% Plot labels
labels = cell(1,length(x));
for k=1:length(x)
    labels{k} = num2str(x(k));
end

% Plot parameters
fs = 16;
lw = 1;
ms = 5;
edgegap = 1.0;
phw = 5-edgegap; % Half width of bar group
px = 10*((0:length(x)-1)+0.5); % Group centres
bars = [1,2];
chl = phw/(3*length(bars)); % Line cap half length - 1/6 bar width

% Barchart
figure1 = figure('Color',[1 1 1]);
hold on
for k=1:length(x)  % Bars
    x1 = px(k)-phw;  x2 = x1 + 2*phw/length(bars);
    for j=1:length(bars)
        if size(meanDat,1) >= j
            y = meanDat(bars(j),k);
            patch([x1 x1 x2 x2], [0 y y 0], (j/(length(bars)+1))*ones(1,3))
        end
        x1 = x2;  x2 = x1 + 2*phw/length(bars);
    end
end
for k=1:length(x)  % StDv lines
    x1 = px(k)-phw;  x2 = x1 + 2*phw/length(bars);
    for j=1:length(bars)
        if size(meanDat,1) >= j
            if size(stdvDat,1) >= j
                y = meanDat(bars(j),k);
                ys = stdvDat(bars(j),k);
                line(0.5*(x1+x2)*[1 1], y+[-ys ys], 'Color', zeros(1,3), 'LineWidth', lw)
                line(0.5*(x1+x2)+[-chl chl], (y+ys)*[1 1], 'Color', zeros(1,3), 'LineWidth', lw)
                line(0.5*(x1+x2)+[-chl chl], (y-ys)*[1 1], 'Color', zeros(1,3), 'LineWidth', lw)
            end

        end
        x1 = x2;  x2 = x1 + 2*phw/length(bars);
    end
end

set(gca, 'XLim', [px(1)-phw-edgegap px(end)+phw+edgegap])
xlabel('Challenger', 'FontSize', fs);
ylabel('Artefact included in segmentation (%)', 'FontSize', fs);
set(gca, 'FontSize', fs);
set(gca, 'XTick', px);
labels = {'IC','MV', 'SY*', 'HB*', 'YL*','KCL', 'UTA', 'UTB'};
set(gca, 'XTickLabels', labels);
set(gca, 'YLim', [0 100]);
pheight = range(get(gca, 'YLim'));
%ht = text(px, -0.01*pheight*ones(size(px)), labels, 'FontSize', fs, 'HorizontalAlignment', 'center', 'VerticalAlignment', 'cap', 'Interpreter', 'tex');
legend('Navigator', 'Aorta wall');
box on;
print(figure1,'-djpeg', 'artefac_analysis.png');

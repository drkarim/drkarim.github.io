Batch convert image 2 world

Co-ordinate transformation

Converts a list of xyz in a text file from world to image co-ordinate system 

world2image.exe <input text file> <gipl image> <output txt file>

Code 

#if defined(_MSC_VER)
#pragma warning ( disable : 4786 )
#endif

#include <irtkImage.h>
#include <vector>
#include <fstream>

int main(int argc, char **argv)
{
	irtkRealImage img; 
	double t;
	double x_size, y_size, z_size, vol; 
	int n=0;
	
	char *filename1, *filename2, *filename3; 
	
	if (argc < 4) 
	{
		cout << "Usage: world2image <input filename> <gipl image> <output filename>" 
			"above label" << endl;
		exit(0);
	}
	filename1 = argv[1];
	argv++;
	argc--;
	filename2 = argv[1];
	argv++;
	argc--;
	filename3= argv[1];
	argv++;
	argc--;
	
	ifstream infile(filename1); 
	img.Read(filename2); 

	ofstream out; 
	out.open(filename3); 

	double x, y, z; 

	while (infile >> x >> y >> z) 
	{
		//cout << x << ", " << y << ", " << z << endl;
		img.WorldToImage(x, y, z); 
		out << round(x) << "\t" << round(y) << "\t" << round(z) << endl;
	}

	out.close();
	
}

Computing atrial wall and percentage automated scripts

Introduction

These two scripts run percentatria.exe on images in turn and expects the folder to contain: 

1. LGE_14010101.gipl 
2. Segmentation_14010101.gipl

There are two posts on the internet which helped in writing this script:

  • http://www.jamesewelch.com/2008/05/06/how-to-set-variables-in-a-dos-for-loop/
  • Question I asked on stack

Script1 

SETLOCAL ENABLEDELAYEDEXPANSION
REM Extract LGE 
   set LGE_file= 
   for /f %%a IN ('dir /b LGE*') do set LGE_file=!LGE_file! %%a 
   
REM Extract segmentation 
   set Seg_file= 
   for /f %%a IN ('dir /b Seg*') do set Seg_file=!Seg_file! %%a 

REM Extract patient number
set patient_number=
for /f "tokens=2 delims=_." %%a in ("%Seg_file%") do set patient_number=%%a

percentatria -i1 %LGE_file% -i2 %Seg_file% -t 3 -o %patient_number%_outputaw.txt
ren at_wall.gipl %patient_number%_at_wall.gipl 
del edt.gipl

Script2 

SETLOCAL ENABLEDELAYEDEXPANSION
REM Extract LGE 
   set LGE_file= 
   for /f %%a IN ('dir /b LGE*') do set LGE_file=!LGE_file! %%a 
   
REM Extract segmentation 
   set Seg_file= 
   for /f %%a IN ('dir /b Seg*') do set Seg_file=!Seg_file! %%a 

REM Extract patient number
set patient_number=
for /f "tokens=2 delims=_." %%a in ("%Seg_file%") do set patient_number=%%a

percentatria -i1 %LGE_file% -i2 %Seg_file% -t 3 -o temp.txt -aw %patient_number%_ref_at_wall.gipl -out %patient_number%_pixels_in_wall.txt
del temp.txt 
del edt.gipl

Clinical Datasets

SD experiment

To determine what is the best SD in the patient scans. Each patient folder will have this script that obtains the different SDs: 

for /l %%x in (1,1,7) do (
	vin -i _le.nii -m _atrial_wall.nii -r _blood_pool.nii -w . -sd1 -1 -sd2 %%x --sd x -o scar_seg_sd.nii
	magic2 -i1 scar_seg_sd.nii -o scar_seg_sd.nii -t1 2 -t2 1 --chbin x
	rk_test5 -g le_handseg_jas.nii  -t scar_seg_sd.nii -fp LocalRegions.txt -sp _atrial_wall.nii -o dice_grand_sd_%%x.txt
)
del scar_seg_sd.nii

and finally the matlab script that will extract dice_grand_sd_NN.txt from every patient folder and plots the boxplot.

The location of this script is phani-pc/software/matlab folder

Phantom experiments

For varying SC-BP ratio test, these are mostly in folder: 

\chronic_scans\phantom_experiment\journal_paper_phantom_manual_seg\cb_new\12_to_15\after_pmb_review\variable_noise_amplitude_with_blurring

The script used to generate the data: 

setlocal enableDelayedExpansion

for %%y in (140 168 196 224 252 280 308 336 364 392 420) do (
	REM echo "Scar intensity %%y" >> dice.txt
	set n=%%y
	createphantom 2 phan_template.gipl phan_scar.gipl phantom.gipl 140 117 %%y 30
	scaleSpace -o phantom_blur.gipl -sc 1.0 -Gxy phantom.gipl 
	scaleSpace -o _le_t.gipl -sc 2.0 -Gz phantom_blur.gipl 
	set /a k=%%y/25
	customdemrinoise -i _le_t.gipl -o _le.gipl -t %%y
	preScar3d_onlygipl _le.gipl _gad_seg.gipl 3.0 0.5 
	newscar3d -m 3.0 -s 0.5 -auto x 
	
	for /l %%x in (1,1,15) do (
		rescale _gc_seg_out_%%x.gipl _gc_seg_out_%%x.gipl 0 1 
		rk_test5 -g phan_scar.gipl -t _gc_seg_out_%%x.gipl -fp LocalRegions.txt -sp _atrial_wall.gipl -o dice_%%y.txt 
	)
)

rk_test5 here is localdice

  • MATLAB script: 
%   The file prepares the matrix that can be boxplotted
%   It searches for files of type: dice_SNR_SCBP.txt 
%   where SNR and SCBP are the noise SNR and SC-BP ratios respectively
%   However, SCBP is measured here as raw intensity with atrial blood pool
%   mean to be 140. 
%   Parameters 
%   numSlices - total number of slices explored for Dice
%   SNR - the SNR being explored
%   numBestSlices - the number of best slices out of numSlices that you
%   want to be regarded. Best meaning best in terms of the median. By
%   default all the slices are considered 

function D = SNR_explore(numSlices, SNR, numBestSlices)

    if (nargin < 3) 
        numBestSlices = numSlices; 
    end
        
    SC_BP = [168 196 224 252 280];
    D = zeros(numBestSlices,length(SC_BP));    
    
    for j = 1:length(SC_BP)
        fn = strcat('dice_',int2str(SNR),'_',int2str(SC_BP(j)),'.txt');
        Dc = load(fn); 
        median_max = -1; 
        median_max_idx = -1; 

        for k=0:-1+length(Dc)/numSlices; 
            T = Dc((numSlices*k)+1: (numSlices*k)+numSlices);
            T = sort(T, 'descend'); 
            T =  T(1:numBestSlices,1); 
            
            if (median_max < median(T)) 
                median_max_idx = k; 
            end
        end
        
        %debuging
        %t1 =  (median_max_idx*numSlices)+1
        %t2 = (numSlices*median_max_idx)+numSlices
        %T
        T = Dc((median_max_idx*numSlices)+1: (numSlices*median_max_idx)+numSlices,1);
         T = sort(T, 'descend'); 
         T =  T(1:numBestSlices,1);
        D(:,j) = 100*T;
    end

end

function B = rk_boxplot(B)

   % B = load(datafile); 
    
    % PLEASE COMMENT LINES THAT FOLLOW 
    % BEGINNING HERE
    % THIS WIL CONSIDER ONLY THE N BEST SAMPLE 
    %B = abs(B);         % extracts the absolute value, all -100 becomes 100
    %B = sort(B,'descend');  % sort in descending order
    %B = B(1:60,:);            % extract the best n (n=80) samples
    % ENDING HERE
    
    N = size(B,2); 		% all the data is held in matrix B 
    p = [9, 25, 50 75 91];
    %p = [0, 25, 50, 75, 100];
    bp = cell(1,N);
    for k=1:N
        bp{k} = prctile(B(:,k), p);
    end

    fs = 20;
    lw = 2;

    %figure
    hw =  2/16;			% size of the boxes relative to the spacing between them. To make bars smaller, make this number smaller
    hold on
    for k=1:N
        x = k-0.4;
        Ys(k,1)=bp{k}(3)
    end

   
    plot([1:N]-0.4, Ys, '-Or', 'LineWidth', 2, 'MarkerSize', 8, 'LineStyle', '-', 'Color', zeros(1,3));
    
    % This draws the error bars
    for k=1:N
        x = k-0.4;
        line(x*[1 1], [bp{k}(1) bp{k}(5)], 'Color', zeros(1,3), 'LineWidth', lw);
        line([x-hw/2 x+hw/2], bp{k}(1)*[1 1], 'Color', zeros(1,3), 'LineWidth', lw);
        line([x-hw/2 x+hw/2], bp{k}(5)*[1 1], 'Color', zeros(1,3), 'LineWidth', lw);
    end

  
    set(gca, 'XTick', 0.5:N-0.5);
    set(gca, 'XTickLabels', {'1.2','','1.8','','2.2'}); 
   
    set(gca, 'FontSize', fs);
  
   set(gca,'YTick', [0; 25; 50; 75; 100]);
    ylabel('Dice', 'FontSize', fs);
    xlabel('SC-BP ratio', 'FontSize', fs);
    %xlabel('SD method', 'FontSize', fs);


    axis([0 length(bp) 0 110]);
    
    box on;

end

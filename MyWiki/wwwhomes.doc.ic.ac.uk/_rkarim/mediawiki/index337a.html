Version 2.0

Usage is the same as Version 1.0 but it outputs:

  • Point is inside or outside the shell (Convex hull). Uses the vtkEnclosedPoint library API
  • Euclidean distance of point between projected and original

Usage 

ra_tool.exe <ablation_points.vtk> <scar3d_shell_point_scalars.vtk> <distance_to_look_around_point> <output.csv>

double getEuclideanDistanceIn3D(double* p1, double* p2) 
{
	double s;
	s = ((p1[0] - p2[0])*(p1[0] - p2[0])) + ((p1[1] - p2[1])*(p1[1] - p2[1])) + ((p1[2] - p2[2])*(p1[2] - p2[2])); 
	return sqrt(s);
}

double ShortestDistanceFromDijkstra(vtkIdList* list, vtkPolyData* surf) 
{
	vtkIdType prev_id, curr_id; 
	double len=0; 
	double xyz1[3], xyz2[3]; 
	for(vtkIdType e = 0; e < list->GetNumberOfIds(); e++) 
	{
		curr_id = list->GetId(e);

		if (e>0) {
			surf->GetPoint(curr_id, xyz1);
			surf->GetPoint(prev_id, xyz2);

			len += getEuclideanDistanceIn3D(xyz1, xyz2); 
		}
		//_shortestPathVertices.push_back(list->GetId(e)); 
		prev_id = list->GetId(e); 
		
	}

	return len; 
}

bool IsDistanceBetweenTheseVerticesLessThan(vtkDijkstraGraphGeodesicPath* dijkstra, vtkPolyData* surf, double dist_t)
{
	double len = -1; 
	len = ShortestDistanceFromDijkstra(dijkstra->GetIdList(), surf); 
	if (len < dist_t && len > -1) {
		return true; 
	}
	else 
	{
		//cout << "Some distances between points are coming awkward"; 
		return false; 
	}
}

void CollectScarData(double** scar_container, int i, int j, double scar_data)
{
	scar_container[i][j] = scar_data; 
}



int main(int argc, char *argv[])
{
	vtkPolyData* _surf1, *_surf2; 
	char *filename1, *filename2, *output_f; 
	double error=0; 
	double dist_t=5; 			// default distance to look around a point 
	double d; 
	int user_args; 
	vtkIdType closestPointOnSurface2;
	
	vector<double> errors;

	double xyz1[3], xyz2[3];
	vtkIdType pid[1]; 
	
	double total_N=0;
	bool isPointExists;
	
	// for the dice 
	double NEG_INFINITY = -1, MAX_SCAR_VALUES = 1000, num_values = 0; 
	
	double total_errors=0, mean=-1, variance=-1;

	double** scar_container; 
	int scar_container_num_values; 
	int scar_container_num_points; 

	// for storing closest point IDs
	vector<int> closestPoints;
	
	vtkSmartPointer<vtkIdList> pointIdList;
	
	_surf1 = vtkPolyData::New(); 
	_surf2 = vtkPolyData::New(); 
	
	if (argc < 3) { 
		cerr << 
			"Usage: \nra_tool.exe \n<ablation points vtk> \n<RA scar3D> \n<exploring distance>\n<output CSV text file>\n\n"  
			"For further instructions see: http://wwwhomes.doc.ic.ac.uk/~rkarim/mediawiki/index.php?title=Ablation_points_to_surface"
			<< endl; exit(1); 
	}
	else
	{
		user_args = argc; 
	}

	filename1 = argv[1];
	argv++;
	argc--;
	filename2 = argv[1];
	argv++;
	argc--;
	dist_t = atof(argv[1]); 
	argv++; 
	argc--; 
	output_f = argv[1]; 
	argv++; 
	argc--; 


	vtkSmartPointer<vtkPolyDataReader> reader1 = vtkSmartPointer<vtkPolyDataReader>::New();
	reader1->SetFileName(filename1); 
	reader1->Update();
	_surf1 = reader1->GetOutput();
	
	vtkSmartPointer<vtkPolyDataReader> reader2 = vtkSmartPointer<vtkPolyDataReader>::New();
	reader2->SetFileName(filename2); 
	reader2->Update();
	_surf2 = reader2->GetOutput();

	vtkSmartPointer<vtkPolyData> closest_points_poly = vtkSmartPointer<vtkPolyData>::New();  
	vtkSmartPointer<vtkPoints> closest_point_set = vtkSmartPointer<vtkPoints>::New();
	vtkSmartPointer<vtkCellArray> closest_point_set_vertices = vtkSmartPointer<vtkCellArray>::New();

	// build the cell locators, this will help us find the cells more quickly 
	vtkSmartPointer<vtkPointLocator > point_locator1 = vtkSmartPointer<vtkPointLocator >::New();
	//vtkSmartPointer<vtkIdList> cell_points1 = vtkSmartPointer<vtkIdList>::New(); 
	point_locator1->SetDataSet(_surf1); 
	point_locator1->AutomaticOn(); 
	point_locator1->BuildLocator();

	vtkSmartPointer<vtkPointLocator > point_locator2 = vtkSmartPointer<vtkPointLocator >::New();
	//vtkSmartPointer<vtkIdList> cell_points2 = vtkSmartPointer<vtkIdList>::New(); 
	point_locator2->SetDataSet(_surf2); 
	point_locator2->AutomaticOn(); 
	point_locator2->BuildLocator();

	// get scar data 
	vtkSmartPointer<vtkFloatArray> scar_data = vtkSmartPointer<vtkFloatArray>::New(); 
	scar_data = vtkFloatArray::SafeDownCast(_surf2->GetPointData()->GetScalars()); 


	// Dijkstra 
	vtkDijkstraGraphGeodesicPath* dijkstra;
	dijkstra = vtkDijkstraGraphGeodesicPath::New();
	dijkstra->SetInput(_surf2);

	// Begin initialize scar container 
	//MAX_SCAR_VALUES = _surf2->GetNumberOfPoints(); 
	scar_container_num_points = _surf1->GetNumberOfPoints(); 
	scar_container_num_values = MAX_SCAR_VALUES; 
	scar_container = new double*[scar_container_num_points]; 

	for (int i=0;i<scar_container_num_points;i++)
	{
		scar_container[i] = new double[scar_container_num_values]; 
	}

	for (int i=0;i<scar_container_num_points;i++)
	{
		for (int j=0;j<scar_container_num_values;j++)
		{
			scar_container[i][j] = NEG_INFINITY; 
		}
	}
	// End initialize scar container 
	
	// Check whether points are inside or outside 
	// Example taken from http://www.vtk.org/Wiki/VTK/Examples/Cxx/PolyData/PointInsideObject
	vtkSmartPointer<vtkSelectEnclosedPoints> selectEnclosedPoints = vtkSmartPointer<vtkSelectEnclosedPoints>::New();
	selectEnclosedPoints->SetInput(_surf1);
	selectEnclosedPoints->SetSurface(_surf2);	
	selectEnclosedPoints->Update();
	
	for (vtkIdType i=0;i<_surf1->GetNumberOfPoints();i++)			// running through ablation points list 
	{
		num_values = 0; 

		_surf1->GetPoint(i, xyz1);
		closestPointOnSurface2 = point_locator2->FindClosestPoint(xyz1);
		_surf2->GetPoint(closestPointOnSurface2, xyz2);

		pid[0] = closest_point_set->InsertNextPoint(xyz2);			// Saving the points for vtk output 
		// Vertices are important 
		// From http://www.cmake.org/Wiki/VTK/Examples/Cxx/Picking/AreaPicking
		closest_point_set_vertices->InsertNextCell(1, pid);

		closestPoints.push_back(closestPointOnSurface2); 
		cout << "Processing ablation point = " << i << ", closest point found with index " << closestPointOnSurface2 << endl;

		//out << xyz1[0]	<< "," << xyz1[1] << "," << xyz1[2] << "\t" << 

		for (vtkIdType j=0;j<_surf2->GetNumberOfPoints();j++)			
		{
			if (closestPointOnSurface2 > 0) {

				//dijkstra->SetStartVertex(closestPointOnSurface2);
				//dijkstra->SetEndVertex(j);
				//dijkstra->Update(); 
				
				_surf2->GetPoint(j, xyz1); 
				_surf2->GetPoint(closestPointOnSurface2, xyz2);
				
				double dist = getEuclideanDistanceIn3D(xyz1, xyz2); 

				if ( dist < dist_t) 
				//if (IsDistanceBetweenTheseVerticesLessThan(dijkstra, _surf2, dist_t)) 
				{
					if (num_values < scar_container_num_values)
					{
						//cout << "j = " << j << ", " << dist << "\t";
						CollectScarData(scar_container, i, num_values, scar_data->GetTuple1(j)); 
						num_values++; 
					}
					else {
						//cout << "This condition is invalid, check logic" << endl; 
					}
				}

				
			}

		}
		  	
	}

	// ready to write to file 
	// file io 	
	ofstream out;		
	out.open(output_f, std::ios_base::app); 

	for (vtkIdType i=0;i<_surf1->GetNumberOfPoints();i++)			
	{
		_surf1->GetPoint(i, xyz1); 							// writing the original ablation points list 
		out << xyz1[0] << "," << xyz1[1] << "," << xyz1[2] << ",,"; 
	}

	out << "\n"; 

	for (vtkIdType i=0;i<_surf1->GetNumberOfPoints();i++)		
	{
		_surf2->GetPoint(closestPoints[i], xyz1); 			// writing the closest point on surface 
		out << xyz1[0] << "," << xyz1[1] << "," << xyz1[2] << ",,"; 
	} 
	
	out << "\n"; 
	
	// To output whether points inside or outside the shell 
	for (vtkIdType i=0;i<_surf1->GetNumberOfPoints();i++)		
	{
		if (selectEnclosedPoints->IsInside(i) == 1) 
			out << "Inside" << ",,,,"; 
		else 
			out << "Outside" << ",,,,"; 
	}
	out << "\n\n"; 
	
	for (vtkIdType i=0;i<_surf1->GetNumberOfPoints();i++)		
	{
		_surf1->GetPoint(i, xyz1); 
		_surf2->GetPoint(closestPoints[i], xyz2); 
		d = getEuclideanDistanceIn3D(xyz1, xyz2); 
		out << d << ",,,,"; 
	}
	out << "\n\n"; 
	

	for (int j=0;j<scar_container_num_values;j++) {			// writing the scar values 
		for (vtkIdType i=0;i<_surf1->GetNumberOfPoints();i++)
		{
			if (scar_container[i][j] != NEG_INFINITY) {
				out << scar_container[i][j] << ",,,,";
			}
			else 
				out << " " << ",,,,";
		}
		out << "\n"; 
	}
	out.close();


	/*********************************************
	*				DEBUG SECTION 		
	*********************************************/
	closest_points_poly->SetPoints(closest_point_set); 
	closest_points_poly->SetVerts(closest_point_set_vertices); 
	vtkPolyDataWriter * writer = vtkPolyDataWriter::New();
	writer->SetFileTypeToBinary();
	writer->SetFileName("closest_points_debug.vtk");
	writer->SetInput(closest_points_poly);
	writer->Update();

	ofstream out2;
	out2.open("scar_values_debug.csv", std::ios_base::app); 
	for (vtkIdType i=0;i<_surf1->GetNumberOfPoints();i++) { 
		for (vtkIdType j=0;j<scar_container_num_values;j++) { 
			out2 << scar_container[i][j] << ",";
		}
		out2 << endl; 
	}
	out2.close();
	//getMeanVariance(errors, mean, variance, total_errors); 
	//cout << "Error, mean = " << mean << "\nSD = " << sqrt(variance) << "\nVariance = " << variance << "\nTotal errors = " << total_errors; 



}

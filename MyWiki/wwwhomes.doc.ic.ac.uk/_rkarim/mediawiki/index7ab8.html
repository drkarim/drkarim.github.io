Remove overlapping Regions program

From Rashed Karim Wiki

Jump to: navigation, search

Usage

Arguments: <mask1> <mask2> <output image name>

Clips regions of mask2 intersecting with mask1

where mask1 (M1) = 1 and mask2 (M2) = 2 (intensities). Intersect operator = X

if C = M1 X M2, then C = 0. AND M1 = 0, M2 = 1.

Code 

#include <irtkImage.h>
#include <fstream>
int main(int argc, char **argv)
{

	if (argc < 4) 
	{
		cerr << "Arguments: <ventricle myocardium as intensity=1> <blood-pool carelessly drawn over myocardium as intensity=2> <output image name>" << endl;
		exit(1); 
	}

		ofstream output; 
       char *filename1, *filename2, *fileoutput;
       irtkGreyPixel *ptr1, *ptr2;

       // get files to subtract
       filename1 = argv[1];
       argv++;
       argc--;
       filename2 = argv[1];
       argv++;
       argc--;
	fileoutput = argv[1];
		//output.open(fileoutput);
       // read image
       irtkGreyImage input1;
       irtkGreyImage input2;

       double d;
       input1.Read(filename1);
       input2.Read(filename2);

       ptr1 = input1.GetPointerToVoxels();
       ptr2 = input2.GetPointerToVoxels();

       cout << "number of voxels " << input1.GetNumberOfVoxels() << endl;
      // cout << "number of voxels " << input2.GetNumberOfVoxels() << endl;

       for (int i=0;i<input1.GetNumberOfVoxels();i++)
       {
				if (*ptr1 == 1 && *ptr2 == 2)
					*ptr2 = 0;
				else if (*ptr2 == 2)
					*ptr2 = 1; 
				else 
					*ptr2 = 0;
              
               ptr1++;
               ptr2++;
       }

		input2.Write(fileoutput); 
}

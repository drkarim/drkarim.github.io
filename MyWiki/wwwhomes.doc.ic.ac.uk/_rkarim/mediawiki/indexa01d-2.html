void QuickFlatener::DisplayPoly(vtkRenderWindowInteractor *flrwi, vtkPolyData* poly, bool isCellData, vtkActor* polyActor)
{
	double min_scalar, max_scalar; 
	vtkSmartPointer<vtkRenderer> renderer;
	vtkSmartPointer<vtkRenderWindow> renderWin;
	renderWin = flrwi->GetRenderWindow();
	vtkSmartPointer<vtkRendererCollection> rendererCollection = renderWin->GetRenderers();
	vtkSmartPointer<vtkLookupTable> lut = vtkSmartPointer<vtkLookupTable>::New();
	vtkSmartPointer<vtkScalarBarActor> scalarBar = vtkSmartPointer<vtkScalarBarActor>::New();
	renderer = rendererCollection->GetFirstRenderer();

	vtkSmartPointer<vtkPolyDataMapper> mapper = vtkSmartPointer<vtkPolyDataMapper>::New();	
	cout << "Adding polydata to mapper, polydata now has " << poly->GetNumberOfCells() << " polygons .. " << endl;
	mapper->SetInput(poly); 
	
	mapper->ScalarVisibilityOn();

	if (isCellData) 
		mapper->SetScalarModeToUseCellData();
	else 
		mapper->SetScalarModeToUsePointData();
	
	VizBoxCore::VBC_getMinMaxScalars(poly, isCellData, min_scalar, max_scalar); 
	mapper->SetScalarRange(0, (max_scalar)); 
	lut->SetTableRange(0, (max_scalar)); 
	
	
	// lookuptable 
	lut->SetHueRange(0.33, 0.0);			// red - 0, yellow - 60, green- 120, cyan - 180, blue - 240, magenta - 300 (on the hue circle, pg. 42 vtk book)
	lut->Build(); 

	mapper->SetLookupTable(lut); 
	
	//vtkSmartPointer<vtkActor> polyActor = vtkSmartPointer<vtkActor>::New(); 

	polyActor->SetMapper(mapper); 
	renderer->AddActor(polyActor);
	
	
	scalarBar->SetLookupTable(lut);
	scalarBar->SetTitle("intensity");
	scalarBar->SetWidth(0.08); 
	scalarBar->SetHeight(1.0); 
	scalarBar->SetDisplayPosition(3,8); 
	 
	
	renderer->AddActor2D(scalarBar);			// displays the scalar bar 
	renderWin->AddRenderer(renderer);
	renderWin->Render();
}

void QuickFlatener::ReadPolyData(char* polyDataFileName, vtkPolyData* poly)

void QuickFlatener::ReadUVMappingFile(char* uvMapFileName, char* uvPointIDmaxFileName) 
{

	// find maximum id by reading file 
	int max2DPointID=-1, max3DPointID=-1; 
	int line_num=0, index3D, index2D;
	ifstream in1, in2; 
	string l, test_l1, test_l2; 
	char* lineChar, *ch; 
	

	in1.open(uvMapFileName); 
	in2.open(uvPointIDmaxFileName); 

	if (in2.is_open())
	{
		getline(in2, l); 
		max3DPointID = atoi(l.c_str()); 
		l = ""; 
		getline(in2, l); 
		max2DPointID = atoi(l.c_str()); 
		cout << "\nMaximum 2D point ID = " << max2DPointID << "\nMaximum 3d point ID = " << max3DPointID << endl;
	}
	
	// init arrays 
	if (max3DPointID != -1) 
	{
		_3d_2_2d_map_size = max3DPointID+1;
		_2d_2_3d_map_size = max2DPointID+1;
		
		_3d_2_2d_mapping = new int[max3DPointID+1]; 
		_2d_2_3d_mapping = new int[max2DPointID+1]; 
		

		for (int i=0;i<max3DPointID+1;i++) 
			_3d_2_2d_mapping[i] = -1;			// initialize

		for (int i=0;i<max2DPointID+1;i++) 
			_2d_2_3d_mapping[i] = -1;			// initialize
		
		if (in1.is_open())
		{
			while (getline(in1, l)) {
				lineChar = new char[l.size()+1]; 
				strcpy(lineChar, l.c_str()); 

				ch = strtok(lineChar, "\t"); 
				//cout << l << endl;
				test_l1 = ch; 
				//index3D = atoi(ch); 
				ch = strtok(NULL, "\t"); 
				test_l2 = ch; 
				//index2D = atoi(ch); 
			//	cout << test_l1 << ":" << test_l2 << endl;
				if (sscanf(test_l1.c_str(), "%d%", &index2D) == 1 && sscanf(test_l2.c_str(), "%d%", &index3D) == 1) 
				{
					//cout << index3D << ":" << index2D << endl;
					if (index3D <  max3DPointID+1) {
						_3d_2_2d_mapping[index3D] = index2D; 
					}

					if (index2D < max2DPointID + 1) { 
						_2d_2_3d_mapping[index2D] = index3D; 
					}
				}

				delete lineChar; 
				line_num++; 

			}
		}
	}
	else 
	{
		cerr << "Could not read the size of point id array from file " << uvPointIDmaxFileName << endl;
		return;
	}

	in1.close(); 
	in2.close();

}

void CreateSphere(vtkRenderer* renderer, double radius, double* position3D)
{
	vtkSmartPointer<vtkSphereSource> sphere = vtkSmartPointer<vtkSphereSource>::New(); 
	sphere->SetThetaResolution(8);
	sphere->SetPhiResolution(8);
	sphere->SetRadius(radius);

	vtkSmartPointer<vtkPolyDataMapper> sphereMapper =vtkSmartPointer<vtkPolyDataMapper>::New(); 
	sphereMapper->ScalarVisibilityOff();
	sphereMapper->SetInput(sphere->GetOutput());

	vtkSmartPointer<vtkActor> actor = vtkSmartPointer<vtkActor>::New();
	actor->SetMapper(sphereMapper); 
	actor->GetProperty()->SetColor(1,0,1); 
	actor->SetPosition(position3D); 

	renderer->AddActor(actor);
	
	
	
}

void keyPressedFunction_window(vtkObject* obj, unsigned long,void *cb, void*)
{
	float screenX, screenY; 
	vtkIdType cellID, pointID=-1, pointID_mapped=-1; 
	double *pick_position = new double[3]; 
	double *mapped_position = new double[3]; 

	CallBackDataContainer* cb_data = (CallBackDataContainer*)cb; 
	vtkSmartPointer<vtkPointPicker> point_picker = vtkSmartPointer<vtkPointPicker>::New(); 
	point_picker = cb_data->picker; 
	vtkRenderWindowInteractor* iren = vtkRenderWindowInteractor::SafeDownCast(obj);
	vtkSmartPointer<vtkRenderWindow> renderWin = iren->GetRenderWindow();
	vtkSmartPointer<vtkRendererCollection> rendererCollection = renderWin->GetRenderers();
	vtkSmartPointer<vtkRenderer> renderer = vtkSmartPointer<vtkRenderer>::New();
	renderer = rendererCollection->GetFirstRenderer();
	
	// other renderer
	vtkRenderWindowInteractor* other_iren = cb_data->otherRenWinInt; 
	vtkSmartPointer<vtkRenderWindow> OtherRenderWin = other_iren->GetRenderWindow();
	vtkSmartPointer<vtkRendererCollection> OtherRendererCollection = OtherRenderWin->GetRenderers();
	vtkSmartPointer<vtkRenderer> other_renderer = vtkSmartPointer<vtkRenderer>::New();
	other_renderer = OtherRendererCollection->GetFirstRenderer();

	switch (iren->GetKeyCode())
	{
	
		case 'x':			
			
			screenX = iren->GetEventPosition()[0];
			screenY = iren->GetEventPosition()[1];
			
			point_picker->Pick(screenX,screenY,0.0, renderer);
			pick_position = point_picker->GetPickPosition(); 

		
			// 2d point was picked 
			pointID = point_picker->GetPointId(); 
			cout << "point id picked = " << pointID << ", on window = " << cb_data->windowNum << endl;
			if (pointID != -1)
			{
				if (cb_data->windowNum == 1) {
					if (pointID < cb_data->qf->_2d_2_3d_map_size)
						pointID_mapped = cb_data->qf->_2d_2_3d_mapping[pointID];
				}
				else if (cb_data->windowNum == 2) {
					if (pointID < cb_data->qf->_3d_2_2d_map_size)
						pointID_mapped = cb_data->qf->_3d_2_2d_mapping[pointID];
				}
				
				if (pointID_mapped != -1)
				{
					cout << "Corresponding point found on 2D map with id = " << pointID_mapped << endl;
					if (cb_data->windowNum == 1)				// user clicked on 2d map 
					{
						//cout << "3d map has = " << cb_data->Mesh_2D->GetNumberOfPoints() << endl; 
						cb_data->Mesh_3d->GetPoint(pointID_mapped, mapped_position);		// look for point on 3d map 
					}
					if (cb_data->windowNum == 2) {
						//cout << "3d map has = " << cb_data->qf->_2d_map->GetNumberOfPoints() << endl;
						cb_data->Mesh_2d->GetPoint(pointID_mapped, mapped_position); 
					}
					
					// draw point 
					if (cb_data->windowNum == 1) { 
						CreateSphere(renderer, 0.005, pick_position); 
						CreateSphere(other_renderer, 0.5, mapped_position); 
					}
					else {
						CreateSphere(renderer, 0.5, pick_position);
						CreateSphere(other_renderer, 0.005, mapped_position); 
					}

					
					OtherRenderWin->Render(); 
					renderWin->Render();
				}
			}
			
		
		break; 
	}

}


int main(int argc, char **argv)
{
	QuickFlatener* qf = new QuickFlatener(); 
	CallBackDataContainer* cb_data1 = new CallBackDataContainer(); 
	CallBackDataContainer* cb_data2 = new CallBackDataContainer(); 


	vtkPolyData* _2D_mesh, *_3D_mesh; 
	qf->ReadUVMappingFile(argv[3], argv[4]); 
	//qf->DisplayMapping();

	_2D_mesh = vtkPolyData::New(); 
	_3D_mesh = vtkPolyData::New(); 
	
	qf->ReadPolyData(argv[1], _2D_mesh); 
	qf->ReadPolyData(argv[2], _3D_mesh); 

	/**
	* NOTE: 
	* Renderer1 is for 2D mesh 
	* Renderer2 is for 3D mesh 
	*/
	vtkSmartPointer<vtkActor> _actor_2D = vtkSmartPointer<vtkActor>::New(); 
	vtkSmartPointer<vtkActor> _actor_3D = vtkSmartPointer<vtkActor>::New(); 
	vtkSmartPointer<vtkRenderer> renderer1 = vtkSmartPointer<vtkRenderer>::New();
	vtkSmartPointer<vtkRenderWindow> renderWindow1 = vtkSmartPointer<vtkRenderWindow>::New();
	renderWindow1->AddRenderer(renderer1);
	vtkSmartPointer<vtkRenderWindowInteractor> renderWindowInteractor1 = vtkSmartPointer<vtkRenderWindowInteractor>::New();
	renderWindowInteractor1->SetRenderWindow(renderWindow1);
	vtkSmartPointer<vtkInteractorStyleTrackballCamera> interactorStyle1 = vtkSmartPointer<vtkInteractorStyleTrackballCamera>::New();
	renderWindowInteractor1->SetInteractorStyle(interactorStyle1);

	vtkSmartPointer<vtkRenderer> renderer2 = vtkSmartPointer<vtkRenderer>::New();
	vtkSmartPointer<vtkRenderWindow> renderWindow2 = vtkSmartPointer<vtkRenderWindow>::New();
	renderWindow2->AddRenderer(renderer2);
	vtkSmartPointer<vtkRenderWindowInteractor> renderWindowInteractor2 = vtkSmartPointer<vtkRenderWindowInteractor>::New();
	vtkSmartPointer<vtkInteractorStyleTrackballCamera> interactorStyle2 = vtkSmartPointer<vtkInteractorStyleTrackballCamera>::New();
	renderWindowInteractor2->SetRenderWindow(renderWindow2);
	renderWindowInteractor2->SetInteractorStyle(interactorStyle2);

	// point pickers
	vtkPointPicker *point_picker1 = vtkPointPicker::New();
	point_picker1->SetTolerance(0.005); 
	renderWindowInteractor1->SetPicker(point_picker1);			
	
	vtkPointPicker *point_picker2 = vtkPointPicker::New();
	point_picker2->SetTolerance(0.005); 
	renderWindowInteractor2->SetPicker(point_picker2);	

	// callbacks
	vtkCallbackCommand *callback1 = vtkCallbackCommand::New();
	callback1->SetCallback(keyPressedFunction_window);
    renderWindowInteractor1->AddObserver(vtkCommand::KeyPressEvent, callback1);

	vtkCallbackCommand *callback2 = vtkCallbackCommand::New();
	callback2->SetCallback(keyPressedFunction_window);
    renderWindowInteractor2->AddObserver(vtkCommand::KeyPressEvent, callback2);

	cb_data1->picker = point_picker1; 
	cb_data1->qf = qf; 
	cb_data1->windowNum = 1; 
	cb_data1->Mesh_2d = _2D_mesh; 
	cb_data1->Mesh_3d = _3D_mesh;
	cb_data1->otherRenWinInt = renderWindowInteractor2;			// other window interactor

	cb_data2->picker = point_picker2;
	cb_data2->qf = qf; 
	cb_data2->windowNum = 2; 
	cb_data2->Mesh_2d = _2D_mesh; 
	cb_data2->Mesh_3d = _3D_mesh;
	cb_data2->otherRenWinInt = renderWindowInteractor1;	

	callback1->SetClientData((void*)cb_data1); 
	callback2->SetClientData((void*)cb_data2); 

	renderer1->SetBackground(.3, .6, .3); // Background color green
	renderer2->SetBackground(.3, .1, .1); // Background color green



	qf->DisplayPoly(renderWindowInteractor1, _2D_mesh, false, _actor_2D);
	qf->DisplayPoly(renderWindowInteractor2, _3D_mesh, false, _actor_3D);

	renderWindow1->Render();
	renderWindowInteractor1->Start();

	renderWindow2->Render();
	renderWindowInteractor2->Start();
	
}

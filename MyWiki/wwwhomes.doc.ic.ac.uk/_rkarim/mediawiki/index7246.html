VTK Conversion Ver 1

Usage

Converts VTK version 3.0 to an old legacy VTK version 1.0. VTK version 1.0 file format: 

# vtk DataFile Version 1.0
PFH meshIO. World coordinate system.
ASCII

DATASET POLYDATA
POINTS 1433 double
-11.062898 -17.835129 50.443132
...
...
POLYGONS 2839 11356
3 0 1 2
3 3 4 5
3 6 7 8
3 9 10 11
3 12 13 14
3 15 16 17
3 18 19 20
3 21 22 23
3 24 25 26
...
...

The usage of this code is: 

<app_name> <new_version_vtk_file> <version_1_0_vtk_file_output> <new_version_3_0_vtk_file_output_but_triangulated>


NOTE: The shell in the new version of VTK may not be triangulated and thus this code triangulates the mesh. In addition to this, it is unclear to me at the time of this writing what is the size of the cells in the VTK file. See below description in VTK manual about the parameters for POLYGONS: 

Each of these keywords requires two parameters: the number of cells n and the size of the cell list size. 
The cell list size is the total number of integer values required to represent the list (i. e., sum of numPoints and connectivity indices over each cell

Thus the second parameter can be copied from the new version of VTK file generated by the code and pasted into the VTK version 1.0 file generated. This is important for the vtk file to be valid.

#define HAS_VTK 1

#include <vtkSphereSource.h>

#include <vtkPlanes.h>
#include <vtkPointData.h>
#include <vtkPolyDataWriter.h>
#include <vtkDijkstraGraphGeodesicPath.h>

#include <vtkPolyLine.h>

#include <vtkLineSource.h>
#include <vtkTextActor3D.h>
#include <vtkCallbackCommand.h>
#include <vtkCellPicker.h>
#include <vtkInteractorStyleTrackballCamera.h>

using namespace std;

vtkCellPicker *cell_picker;
vtkSmartPointer<vtkPolyData> mesh;
int main(int argc, char **argv){
	 
	char* input_f, *output_f; 

	// Read mesh 
	vtkSmartPointer<vtkPolyDataReader> reader = vtkSmartPointer<vtkPolyDataReader>::New(); 
	reader->SetFileName(argv[1]);		// REMEMBER to pass the mesh as a command-line parameter 
	reader->Update();		
	mesh = vtkSmartPointer<vtkPolyData>::New(); 
	mesh = reader->GetOutput();

	vtkSmartPointer<vtkPolyDataMapper> mapper = vtkSmartPointer<vtkPolyDataMapper>::New(); 
	mapper->SetInput(mesh); 
	
	vtkSmartPointer<vtkActor> actor = vtkSmartPointer<vtkActor>::New(); 
	actor->SetMapper(mapper); 

	vtkSmartPointer<vtkRenderer> renderer = vtkSmartPointer<vtkRenderer>::New(); 
	vtkSmartPointer<vtkRenderWindow> renderWindow = vtkSmartPointer<vtkRenderWindow>::New(); 
	
	renderer->AddActor(actor); 
	renderWindow->AddRenderer(renderer); 

	vtkSmartPointer<vtkRenderWindowInteractor> iren = vtkSmartPointer<vtkRenderWindowInteractor>::New(); 
	iren->SetRenderWindow(renderWindow); 
	vtkSmartPointer<vtkInteractorStyleTrackballCamera> interactorStyle = vtkSmartPointer<vtkInteractorStyleTrackballCamera>::New();
	iren->SetInteractorStyle(interactorStyle);

	vtkCallbackCommand *callback = vtkCallbackCommand::New();			// A new callback - callback is bascially an event handler 
    callback->SetCallback(keyPressEventHandler);		// you are telling what function will do the event handling 
	callback->SetClientData(mesh);			// you are sending the mesh to the event handler (it is the void* last parameter in the function signature) 
    iren->AddObserver(vtkCommand::KeyPressEvent, callback);		// invoke callback when key is pressed

	renderWindow->Render(); 

	// These are the lines to initialize your cell picker - this will allow you to pick points on the mesh 
	
	
	cell_picker = vtkCellPicker::New();
	cell_picker->SetTolerance(0.0005);	// You set a tolerance meaning to what degree of accuracy it is able to select points 
	iren->SetPicker(cell_picker);		// and you tell which interactor this cell picker is part of (you only have one interactor) 


	iren->Start(); 

	return 0;

}

Introduction

  • Analyses the fti and scar data. For more details see here.

Code 


%{
    fname - file containing fti <tab> pt_scar <tab> av_scar_integral   
    is_av_scar - true if you want the average scar to be considered or false if
    not, then av_scar_integral is considered
    pt_scar - the percentage of the 5 or 10 mm radius that is scar 
    av_scar_integral - from the binary scar integral map, the average scar
    integral around within 5 or 10 mm of ablation point 
    Params:
    interval - maximum number of intervals, for example you want 10
    intervals maximum at which ftis will be binned 
    interval_length - how big is each interval, default = 100
    decide size of your interval and interval length based on the max(fti)

    which_cases - you could analyze individual cases or all cases at once: 
    which_case = 1 for case1 and so on. -1 for all cases. 
   
    NOTE VERY IMPORTANT: pt_threshold must be order of 10 between 0 and
    100. 
%}
function [A, C, Ln, bplot] = fti_vs_3m(scar_pt_or_scar_integral, radius_in_fname, ...
    interval, interval_length, pt_threshold, output_f)
    
   
    if scar_pt_or_scar_integral == 1
        idx = 2;            % consider the 2nd column in A vector as this corresponds to av percent scar
    elseif scar_pt_or_scar_integral == 2
        idx = 3;            % 3rd column is scar integral 
    end
    
    path = 'C:\PhD\experiments\kcl_data\endosense\';
    case_folders = { 'case1_MP\2013', ...
                        'case2_SM\2013', ...
                        'case3_JC\2013', ... 
                        'case4_ST\2013', ...
                        'case7_L_walsh\force_analysis_1', ...
                        'case8_MGrainger\force_analysis_1'
                       };
    Ln = [];
    % analyzing either all cases or just one particular case 
    for i=1:length(case_folders);
        L = []; 
        fn = strcat(path,case_folders{i},'\fti_vs_chsh_',int2str(radius_in_fname),'mm.txt');
        L = load(fn);
        Ln = [Ln; L];
    end

    
    A = Ln;
    
  
    % if we are going to consider lesion depth, then scale to a 100 to fit to
    % the percentage scale. Remember they are already normalized to 1 
    if (idx == 3) 
        A(:,3) = A(:,3)*100;
    end
    
    I = A(:,2) > 0;
    A = A(I,:) 
   % A(sum((A==0),2)>0,:) = [];
    
    fti = A(:,1);
    
    max_fti = max(fti) ;         % decide size of your interval and interval length based on the max
    
    
    fti_limit = interval*interval_length; 
    %nbins = histc(fti, [min_fti_round:fti_steps:max_fti]);
    
    %C = [min_fti_round:fti_steps:max_fti_round]
    bplot = NaN*ones(200, 1+interval);                % we dont expect to be binning more than 200 values if so then increase
   
    j=1;
    for i=0:interval_length:fti_limit
        C=[];
        for k=1:length(fti)
            if (i < fti_limit) 
                if fti(k) > i %&& fti(k) < i+interval_length
                   C = [C; A(k,idx)]; 
                end
            else 
                % for last bin, we bin all fti values above the limit
                if fti(k) >= fti_limit 
                    C = [C; A(k,idx)];
                end
            end
        end
        
        bplot(1:length(C),j) = C;
        j=j+1;
    end
    figure1 = figure('Color',[1 1 1]);
   hold on;
    Xs = [0:interval_length:fti_limit];
    Ys = zeros(size(bplot,2),1); 
    Ys2 = zeros(size(bplot,2),1); 
    
    % Extracting how many points have scar. Scar is set as percentage of
    % scarrness starting from 0 to 100 (the outer for loop) and in
    % intervals of 10. 
    
    idx = 0;
    pt_interval  =10; 
    for k=0:10:100
        idx = idx+1;
        for i=1:size(bplot,2)
            count = 0;
            total_count=0;
            for j=1:size(bplot,1)
                if (bplot(j,i) > 0)
                    if bplot(j,i) >= k 
                        count=count+1;
                    end
                    total_count=total_count+1;
                end
            end
            Ys(i,idx) = 100*(count/total_count);
            I = ~isnan(bplot(:,i));
            %Ys(i,idx) = median(bplot(I,i));
        end
        
        % Now draw a line plot 
        %plot(Xs,  Ys, '-Xr', 'LineWidth', 2, 'MarkerSize', 5, 'LineStyle', '-');
    end
    
    
    set(gca, 'FontSize', 12);       % set the size of the X-axis labels here
    
    whichCol = idivide(int32(pt_threshold), 10, 'round'); 
    
    bar(Xs, Ys(:,1+whichCol), 'FaceColor', [0.75, 0.75 , 0.75], 'LineWidth', 2);
    plot(Xs,  Ys(:,1+whichCol), '-or', 'LineWidth', 4, 'MarkerSize', 5, 'LineStyle', '-','Color', [0 0 0]);
    
    set(gca, 'XTick', Xs);
   set(gca, 'XTickLabel', {'>0','>200','>400','>600','>800','>1000','>1200','>1400', '>1600', '>1800'});
   % set(gca, 'YTick', 0:10:100);
    axis([-100 max(Xs)-100 0 100]); 
     
    xlabel('Force Time Integral (FTI)', 'FontSize', 15);
    ylabel('Fraction of lesions resulting in scar (%)', 'FontSize', 15);
    box on; 
    print(figure1,'-djpeg', output_f);
    
end

#include <irtkImage.h>


int main(int argc, char **argv)
{
	int _f_image1=1;
		double x=0, m1=0, m2=0, m1e=0, m2e=0, xm1=0, xm2=0, xm1e=0, xm2e=0, xORm1=0, xORm2=0, xORm1e=0, xORm2e=0, m1m2=0, m1ORm2=0; 
		double _m, _s; 
       char *filename1, *filename2, *output, *filename3, *filename4, *filename5, *filename6;
       irtkGreyPixel *ptr1, *ptr2,  *ptr3, *ptr4, *ptr5;
	   bool isWriteToFile = false;
	   if (argc < 3) { cerr << "\nUsage dice <mean> <sigma> <autoseg> <man1> <man2> <man1-erode> <man2-erode> <output-file-to-write-dice>\n\nOutputs dice-efficient. \n\nNote: This is different from the other dice program as you have to specify what intensity value denotes foreground in your images.\n\nAlso, you can specify (optional) an output file to write the dice values to"  << endl; exit(1); }
       // get files to subtract
	  
	   _m = atof(argv[1]); 
	   argv++;
       argc--;
	   _s = atof(argv[1]); 
	   argv++;
       argc--;
       filename1 = argv[1];
       argv++;
       argc--;
       filename2 = argv[1];
       argv++;
       argc--;
	   filename3 = argv[1];
       argv++;
       argc--;
	   filename4 = argv[1];
       argv++;
       argc--;
	   filename5 = argv[1];
       argv++;
       argc--;
	   filename6 = argv[1];
       argv++;
       argc--;   
       //output = argv[1];

       // read imaged
       irtkGreyImage input1, input2, input3, input4, input5; 


       double d;
	   double d_c, o_p1, o_p2; 

       input1.Read(filename1);
       input2.Read(filename2);
	   input3.Read(filename3); 
	   input4.Read(filename4); 
	   input5.Read(filename5); 

       ptr1 = input1.GetPointerToVoxels();
       ptr2 = input2.GetPointerToVoxels();
	   ptr3 = input3.GetPointerToVoxels();
	   ptr4 = input4.GetPointerToVoxels();
	   ptr5 = input5.GetPointerToVoxels();


       cout << "number of voxels " << input1.GetNumberOfVoxels() << endl;
       cout << "number of voxels " << input2.GetNumberOfVoxels() << endl;

	   if (input1.GetNumberOfVoxels()-input2.GetNumberOfVoxels() == 0)

	   {
			for (int i=0;i<input1.GetNumberOfVoxels();i++)
		   {

			   // intersections 
			   if (*ptr1 == _f_image1 && *ptr2 == _f_image1)
				   xm1++; 
				if (*ptr1 == _f_image1 && *ptr3 == _f_image1)
					xm2++; 
				if (*ptr1 == _f_image1 && *ptr4 == _f_image1)
					xm1e++; 
				if (*ptr1 == _f_image1 && *ptr5 == _f_image1)
					xm2e++; 
				if (*ptr2 == _f_image1 && *ptr3 == _f_image1)
					m1m2++; 


				// just individual images
				if (*ptr1 == _f_image1) {
					x++; }
				if (*ptr2 == _f_image1) {
					m1++; }
				if (*ptr3 == _f_image1) {
					m2++; }
				if (*ptr4 == _f_image1) {
					m1e++; }
				if (*ptr5 == _f_image1){
					m2e++; }
				
			   // and now the conjunctions 
			   if (*ptr1 ==  _f_image1 || *ptr2 == _f_image1)
				   xORm1++; 
			   if (*ptr1 ==  _f_image1 || *ptr3 == _f_image1)
				   xORm2++; 
			   if (*ptr1 ==  _f_image1 || *ptr4 == _f_image1)
				   xORm1e++; 
			   if (*ptr1 ==  _f_image1 || *ptr5 == _f_image1)
				   xORm2e++; 
			   if (*ptr3 ==  _f_image1 || *ptr4 == _f_image1)
				   m1ORm2++; 
			   

			   ptr1++; ptr2++; ptr3++; ptr4++; ptr5++; 
		   }
			
			double dxm1, dxm2, dxm1e, dxm2e, dm1m2; 

			dxm1 = ((double)2*xm1)/xORm1; 
			dxm2 = ((double)2*xm2)/xORm2; 
			dxm1e = ((double)2*xm1e)/xORm1e; 
			dxm2e = ((double)2*xm2e)/xORm2e; 
			dm1m2 = ((double)2*m1m2)/m1ORm2; 
		
			ofstream fileIO; 
			fileIO.open(filename6, ios::app); 
			fileIO << _m << "\t\t" << _s << "\t\t" << dxm1 << "\t\t" <<  dxm2 << "\t\t" << dxm1e << "\t\t" << dxm2e << "\t\t" << dm1m2 << endl; 
			fileIO.close(); 


	   }
	   else
	   {
		   cout << "images with unequal dimensions, cannot compute the dice, must be equal dimensions" << endl;
	   }
}

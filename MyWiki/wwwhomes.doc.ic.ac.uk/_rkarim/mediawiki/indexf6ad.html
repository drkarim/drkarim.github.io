Project thickness to surface polygon

From Rashed Karim Wiki

Jump to: navigation, search

Usage 

cwall.exe -i <input_image gipl or niftii> -s <surface in vtk format> -o <output filename in vtk format> 

Optional parameters: 

-t <specify the cut-off for background intensity, value below which pixel is background. Default is 0. This parameter is optional> 
-smooth <n number of iterations for smoothing

Code 

#define HAS_VTK 1
#include <vtkPolyDataReader.h>
#include <vtkPolyData.h>
#include <vtkSmartPointer.h>
#include <vtkCleanPolyData.h>
#include <irtkImage.h>
#include <vtkFloatArray.h>
#include <vtkPolyDataNormals.h>
#include <vtkPolyDataWriter.h>
#include <vtkCellData.h>
#include <vtkIdList.h>
#include <vtkWindowedSincPolyDataFilter.h>
#include <vtkSmoothPolyDataFilter.h>
#include <vtkDecimatePro.h>

#include <iostream>
#include <fstream>
#include <vector>
#include <cstdlib>


#define STEP_SIZE 1				// THIS MUST BE 1 UNLESS THERE IS GOOD REASON TO CHANGE THIS. 1 PIXEL STEP AT A TIME TO EXPLORE THICKNESS
#define MAX_STEPS 100			// MAXIMUM WALL THICKNESS THIS METHOD CAN MEASURE - PIXEL WIDE


using namespace std; 


void normalizeVector(double &a, double &b, double &c)
{
	double magn; 
	magn = sqrt(pow(a,2)+pow(b,2)+pow(c,2)); 
	a *=(1/magn); 
	b *=(1/magn); 
	c *=(1/magn); 
}

void DemicateSurface(vtkPolyData* poly)
{
	vtkSmartPointer<vtkDecimatePro> deci = vtkSmartPointer<vtkDecimatePro>::New(); 
	deci->SetInput(poly); 
	deci->SetTargetReduction(0.50); 
	deci->PreserveTopologyOff();
	deci->SplittingOn(); 
	deci->SetBoundaryVertexDeletion(1);
	deci->SetMaximumError(VTK_LARGE_FLOAT);
	deci->Update();
	poly->DeepCopy(deci->GetOutput()); 

}

void SmoothSurfaceMesh(vtkPolyData* poly, int iterations)
{
	vtkSmartPointer<vtkWindowedSincPolyDataFilter> smooth = vtkSmartPointer<vtkWindowedSincPolyDataFilter>::New();
	smooth->SetInput(poly);
	smooth->SetPassBand(0.1);
	smooth->SetNumberOfIterations(iterations);
	smooth->BoundarySmoothingOn();
	smooth->SetFeatureAngle(45);
	smooth->SetEdgeAngle(15);
	smooth->Update(); 
	poly->DeepCopy(smooth->GetOutput());
}

void SmoothSurfaceMesh2(vtkPolyData* poly, int iterations)
{
	vtkSmartPointer<vtkSmoothPolyDataFilter> smooth = vtkSmartPointer<vtkSmoothPolyDataFilter>::New();
	smooth->SetInput(poly);
	smooth->SetNumberOfIterations(iterations);
	smooth->BoundarySmoothingOn();
	smooth->SetFeatureAngle(45);
	smooth->SetEdgeAngle(15);
	smooth->Update(); 
	poly->DeepCopy(smooth->GetOutput());
}

void GetThicknessAlongNormal(double n_x, double n_y, double n_z, double centre_x, double centre_y, 
							 double centre_z, irtkRealImage* img, double background, double& wall_thickness)
{
	
	double insty=0,x=0,y=0,z=0, rounded_x, rounded_y, rounded_z; 
	double step_size = STEP_SIZE; 
	double step_max = MAX_STEPS; 
	bool hitBackground = false; 
	int pixel_thickness=0;
	
	// normalize
	normalizeVector(n_x, n_y, n_z);
	

	for (double i=0;i<=step_max && !hitBackground;i+=step_size)
	{
		x = centre_x + (i*n_x); 
		y = centre_y + (i*n_y); 
		z = centre_z + (i*n_z);
		x = floor(x); y=floor(y); z=floor(z);
		pixel_thickness++; 
		
//		cout << "(x,y,z) = " << x << ", " << y << ", "  << z << endl;
		if (x > 0 && y > 0 && z > 0 && x < img->GetX() && y < img->GetY() && z < img->GetZ()) {
			if (img->Get(x,y,z) < background) 
				hitBackground = true; 
		}
	}

	wall_thickness = pixel_thickness; 

}

void GenerateWallThicknessMap(irtkRealImage* wall_img, vtkPolyData* wall_surf, double background_t)
{
	double pN[3];
	double cP[3]; 
	double cX, cY,cZ, scalar=0;
	int num_points = 0;
	vtkIdType num_cell_points; 
	vtkSmartPointer<vtkIdList> cell_points = vtkSmartPointer<vtkIdList>::New();

	vtkSmartPointer<vtkPolyData> polys = vtkSmartPointer<vtkPolyData>::New(); 
	/*
	*	Compute normals of the surface 
	*/
	vtkSmartPointer<vtkPolyDataNormals> normals = vtkSmartPointer<vtkPolyDataNormals>::New();
	normals->FlipNormalsOn();
	normals->ComputeCellNormalsOn();
	normals->SetInput(wall_surf); 
	normals->Update(); 
	polys = normals->GetOutput(); 

	vtkSmartPointer<vtkFloatArray> scalars = vtkSmartPointer<vtkFloatArray>::New();	
	vtkSmartPointer<vtkFloatArray> cellNormals= vtkFloatArray::SafeDownCast(polys->GetCellData()->GetNormals());

	for (int i=0;i<polys->GetNumberOfCells();i++)			// running through each polygon 
	{
		cellNormals->GetTuple(i, pN);
		cX=0; cY=0; cZ=0; num_points=0;
		vtkIdType neighbor_point; 
		polys->GetCellPoints(i, cell_points);
		num_cell_points = cell_points->GetNumberOfIds(); 
        for ( neighbor_point = 0; neighbor_point < num_cell_points; ++neighbor_point ) 
        { 
			// Get the neighbor point id.           
			vtkIdType neighbor_point_id = cell_points->GetId(neighbor_point);

			// Get the neighbor point position. 
			//double cP[3]; 
			polys->GetPoint( neighbor_point_id, cP ); 
			wall_img->WorldToImage(cP[0], cP[1], cP[2]); 
			cX+=cP[0]; cY+=cP[1]; cZ+=cP[2]; 
			num_points++;

		}
		cX/=num_points; cY/=num_points; cZ/=num_points;
		wall_img->WorldToImage(pN[0], pN[1], pN[2]); 
		GetThicknessAlongNormal(pN[0], pN[1], pN[2], cX, cY, cZ, wall_img, background_t, scalar);
	//	cout << "cell = " << i << ", thickness = " << scalar << endl;
		scalars->InsertTuple1(i, scalar); 
	}

	wall_surf->GetCellData()->SetScalars(scalars);

}

int main(int argc, char * argv[])
{

	char* input_img_f="", *input_vtk_f="", *output_f=""; 
	double background_t=0, smoothing_t=10; 
	irtkRealImage irtk_img1; 
	bool foundArgs= false, isSmoothMesh=false;
	vtkPolyData* wall_surf; 

	if (argc >= 1) 
	{
		for (int i = 1; i < argc; i++) {
			if (i + 1 != argc) {
				if (string(argv[i]) == "-i") {
					input_img_f = argv[i+1]; 
					foundArgs = true;	
				} 
				else if (string(argv[i]) == "-s") {
					input_vtk_f = argv[i+1]; 
					foundArgs = true;	
				} 
				else if (string(argv[i]) == "-t") {
					background_t = atof(argv[i+1]); 
					foundArgs = true;	
				} 
				else if (string(argv[i]) == "-o") {
					output_f = argv[i+1]; 
					foundArgs = true;	
				}
				else if (string(argv[i]) == "-smooth") {
					smoothing_t = atoi(argv[i+1]); 
					isSmoothMesh = true; 
				}
			}
		}
	}

	if (foundArgs == false)
	{
		cout << "\nUsage: cwall.exe -i <input image gipl or niftii> -s <surface in vtk format> -o <output filename in vtk format>"
			"\n\nOptional parameters:\n======================\n"
			"\n-t\t<specify the cut-off for background intensity, value below which a \n\tpixel is background. Default is 0. "
			"This parameter is Optional>\n-smooth\t <n number of iterations for smoothing>\n";
		exit(0); 
	}

	// read image 
	irtk_img1.Read(input_img_f); 
	

	vtkSmartPointer<vtkPolyDataReader> reader = vtkSmartPointer<vtkPolyDataReader>::New();
	reader->SetFileName(input_vtk_f); 
	reader->Update(); 
	wall_surf = reader->GetOutput(); 
	
	if (background_t != 0) cout << "background chosen as = " << background_t << endl;

	if (isSmoothMesh) {
		cout << "Smoothing mesh, performing " << smoothing_t << " iterations \n";
		DemicateSurface(wall_surf);
		SmoothSurfaceMesh(wall_surf, smoothing_t);
	}

	GenerateWallThicknessMap(&irtk_img1, wall_surf, background_t); 
	vtkSmartPointer<vtkPolyDataWriter> vtkwriter = vtkSmartPointer<vtkPolyDataWriter>::New(); 
	vtkwriter->SetInput(wall_surf); 
	vtkwriter->SetFileName(output_f); 
	vtkwriter->SetFileTypeToASCII(); 
	vtkwriter->Write(); 

}

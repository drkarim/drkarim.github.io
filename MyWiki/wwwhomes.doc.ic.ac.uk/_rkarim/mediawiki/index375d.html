#include "cv.h"
#include "cvaux.h"
#include "highgui.h"
#include <iostream>

using namespace std;
using namespace cv; 


/* 
*    order specifies whether to half the resolution (order = 2), half of half (order = 4), and so on. .. 
*/
void ConvertToLowResolution(IplImage* high_res, IplImage* low_res, int order)
{
	int occurence=0;		// order of resolution 
	CvSize size; 
	CvScalar high_res_scalar; 
	CvScalar low_res_scalar; 
	size = cvGetSize(high_res); 
	cvZero(low_res); 
	cout << "Size = " << size.height << "x" << size.width << endl;
	for (int i=0;i<size.height;i+=order)
	{
		for (int j=0;j<size.width;j+=order)
		{
			occurence = 0; 
			// check 2^order blocks 
			// first check in high resolution
			for (int m=0;m<order;m++) { 
				for (int n=0;n<order;n++)
				{
					if (i+m < size.height && j+n<size.width) { 
						high_res_scalar = cvGet2D(high_res, i+m, j+n); 
						if (high_res_scalar.val > 0) 
							occurence++; 
					}
				}
			}

			if (occurence == order*order) 
			{
				// now color in low resolution
				for (int m=0;m<order;m++) { 
					for (int n=0;n<order;n++)
					{
						if (i+m < size.height && j+n<size.width) { 
							cvSet2D(low_res, i+m,j+n, high_res_scalar); 
						}
					}
				}
			}
		}
	}

}
int main(int argc, char* argv[])
{

	IplImage* low_res, *high_res;
	if (argc != 3)
	{
		cerr << "Usage (.exe) [path to image file high res] [path to write to file low res]" << endl; 
		exit(0); 
	}

	high_res = cvLoadImage(argv[1]);
	low_res = cvCreateImage(cvGetSize(high_res), IPL_DEPTH_8U,3);
	ConvertToLowResolution(high_res, low_res, 2); 
	cout << "Now writing output .. " << endl;
	imwrite(argv[2], low_res);
	


}

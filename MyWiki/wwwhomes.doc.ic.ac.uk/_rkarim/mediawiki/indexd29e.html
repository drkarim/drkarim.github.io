Surface Dice

For description of what a surface dice follow description on this page.

Description

This is a very problem-specific dice for surfaces that will compute the dice by thresholding the second surface based on values that are input through a text file. It will threshold the surface at 10 different levels, starting from min to max, where min = 0 and max is specified as the third line in text file supplied as input

Usage 

<vtk file 1> <vtk file 2> <scalar max file> <outoput dice file>

void getDice(char* filename1, char* filename2, float _label1, float _label2, ofstream& outToFile) 
{
	vtkPolyData* _surf1, *_surf2; 
	int _surf1_scal, _surf2_scal, n=0;
	double c_x=0, c_y=0, c_z=0;
	vtkIdType cellId, subId, neighbor_point, num_cell_points; 
	double dist; 
	double centroid[3], xyz[3], closestPoint[3];
	
	
	// for the dice 
	double x=0,y=0,x_and_y=0, x_or_y=0; 

	vtkGenericCell *temp_cell;
	temp_cell = vtkGenericCell::New(); 
	vtkIdType closestPointID=-1; 
	vtkIdType cell_point_id;
	 
	vtkCell* closest_cell;
	vtkIdList* pointIdList;
	
	_surf1 = vtkPolyData::New(); 
	_surf2 = vtkPolyData::New(); 
	
	vtkSmartPointer<vtkPolyDataReader> reader1 = vtkSmartPointer<vtkPolyDataReader>::New();
	reader1->SetFileName(filename1); 
	reader1->Update();
	_surf1 = reader1->GetOutput();
	
	vtkSmartPointer<vtkPolyDataReader> reader2 = vtkSmartPointer<vtkPolyDataReader>::New();
	reader2->SetFileName(filename2); 
	reader2->Update();
	_surf2 = reader2->GetOutput();

	// build the cell locators, this will help us find the cells more quickly 
	vtkSmartPointer<vtkCellLocator> cell_locator1 = vtkSmartPointer<vtkCellLocator>::New();
	vtkSmartPointer<vtkIdList> cell_points1 = vtkSmartPointer<vtkIdList>::New(); 
	cell_locator1->SetDataSet(_surf1); 
	cell_locator1->AutomaticOn(); 
	cell_locator1->BuildLocator();

	vtkSmartPointer<vtkCellLocator> cell_locator2 = vtkSmartPointer<vtkCellLocator>::New();
	vtkSmartPointer<vtkIdList> cell_points2 = vtkSmartPointer<vtkIdList>::New(); 
	cell_locator2->SetDataSet(_surf2); 
	cell_locator2->AutomaticOn(); 
	cell_locator2->BuildLocator();


	// now get the scalar arrays for the two surfaces  
	vtkSmartPointer<vtkFloatArray> _surf1_scal_arr;
	_surf1_scal_arr =  vtkFloatArray::SafeDownCast(_surf1->GetCellData()->GetScalars());

	vtkSmartPointer<vtkFloatArray> _surf2_scal_arr;
	_surf2_scal_arr =  vtkFloatArray::SafeDownCast(_surf2->GetCellData()->GetScalars());

	for (vtkIdType i=0;i<_surf1->GetNumberOfCells();i++)			// running through each cell in surface 1 
	{
		_surf1->GetCellPoints(i, cell_points1); 
		_surf1_scal = _surf1_scal_arr->GetTuple1(i); 
		
		if (_IS_EXTREME_DEBUG == 1) cout << "got here 2 " << endl;
		if (_surf1_scal >= 0)			
		{
			num_cell_points = cell_points1->GetNumberOfIds(); 
			c_x=0; c_y=0; c_z=0;
			
			if (_IS_EXTREME_DEBUG == 1) cout << "got here 3 " << endl;
			for ( neighbor_point = 0; neighbor_point < num_cell_points; ++neighbor_point ) 
			{
				vtkIdType neighbour_point_id = cell_points1->GetId(neighbor_point); 
				
				if (_IS_EXTREME_DEBUG == 1) cout << "got here 4 " << endl;
				_surf1->GetPoint(neighbour_point_id, xyz); 
				c_x+=xyz[0]; c_y+=xyz[1]; c_z+=xyz[2]; 
			}
			c_x /= 3; c_y /= 3; c_z /= 3; 
			centroid[0] = c_x; centroid[1] = c_y; centroid[2] = c_z; 
			if (_IS_EXTREME_DEBUG == 1) cout << "centroid (x,y,z) = (" << c_x << ", " << c_y << ", " << c_z << ")" << endl;
			// if the cell is within the region of interest selected by the user 

			cell_locator2->FindClosestPoint(centroid, closestPoint, temp_cell, cellId, subId, dist); 	
			if (cellId > 0)
			{
				_surf2_scal = _surf2_scal_arr->GetTuple1(cellId);

				// so now we can compute the dice since we have two scalar values from cells 
				if (_surf1_scal >= _label1 && _surf2_scal >= _label2)
					x_and_y++;
				if (_surf1_scal >= _label1)
					x++; 
				if (_surf2_scal >= _label2)
					y++;
				if (_surf1_scal >= _label1 || _surf2_scal >= _label2)
					x_or_y++;

			}
			else
			{
				cout << "\n\noouch!! Aren't these surfaces geometrically equal?? I can only now give you an estimate for the dice\n\n" << endl;
			}

		}
		n++;
	}
	
	double d_c, o_p1, o_p2; 
	d_c = (2*((double)x_and_y))/((double)x+(double)y); 
	o_p1 = ((double)x_and_y/(double)x)*100;
	o_p2= ((double)x_and_y/(double)y)*100;

	//cout << "\nDice co-efficients:\n=====================\n" << "Total cells = " << n << "\nImage1 and Image2 overlap cells = " << x_and_y << "\nImage1 foreground cells = " << x << "\nImage2 foreground cells = " << y << endl; 
	outToFile << d_c << endl; 
	//cout << "Overlap as a percentage of image1 cells = " << o_p1 << endl; 
	//cout << "Overlap as a percentage of image2 cells = " << o_p2 << endl;
	//cout << "\n\n==================================\n"; 
	//double o_m = (double)x_and_y/(double)x_or_y; 
	//cout << "Volumetric overlap metric (i.e. a_and_b / a_or_b)\nImage1 AND Image2 = " << x_and_y << "\nImage1 OR Image2 = "<< x_or_y << "\nOverlap metric = " << o_m << endl;
	


	}


int main(int argc, char *argv[])
{
	
	char *filename1, *filename2, *filename3, *filename4; 
	ifstream fout;
	ofstream fout2;
	float _label1, _label2; 

	if (argc < 5) { 
		cerr << "\n(c) RashedSoft 2011\n\nUsage dice_vtk_batch <vtk_file1> <vtk_file2> <scalar_max_file> <output_dice_text_file>"  
			"\n\n\tvtk_file2 must be the output from scar3d (ben knowles auto pilot)\n\tand, scalar_max_file is again the output from the same program and says what is the maximum sdev in the surface (fourth line in file)"<< endl; exit(1); 
	}

	filename1 = argv[1];
	argv++;
	argc--;
	filename2 = argv[1];
	argv++;
	argc--;
	
	//_label2 = atof(argv[1]);
	filename3 = argv[1]; 
	argv++; 
	argc--; 
	filename4 = argv[1]; 
	argv++; 
	argc--; 
	
	fout2.open(filename4, ios::out | ios::app); 
	fout.open(filename3); 
	string line; 
	int num_lines=0;
	double max_sdev; 

	// read the max scalar from file 
	if (fout.is_open())
	{
		while (!fout.eof())
		{
			getline(fout,line); 
			num_lines++;
			if (num_lines == 4)
			{
				from_string<double>(max_sdev, line, std::dec);
			}
		}
	}
	cout << "max sdev from file = " << max_sdev;
	double sep = max_sdev/10; 

	fout2 << endl; 
	for (double i=0;i<=max_sdev;i+=sep)
	{		
		_label1 = 1.1; 
		fout2 << i << "\t\t";
		getDice(filename1, filename2, _label1, i, fout2);
	}
	fout2.close();

}

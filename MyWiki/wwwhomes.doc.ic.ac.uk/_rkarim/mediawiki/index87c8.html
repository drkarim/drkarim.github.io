int main (int argc, char *argv[] )
{
  char* _sc_fn, *_v_fn, *_navx; 
	EnsiteAdapter ensite; 
	EnsiteGraphicsAdapter ensiteGraphics; 
	int num_files, _interpolationtype; 
	vector<vtkIdType> voltageNodeList; 

  vtkPolyData* _sc_poly; 

  if (argc < 3)
	{
		std::cerr << "Usage: " << argv[0]
				<< " voltage <scar_shell_in_vtk> <output_voltage> <navx_voltage_first_file> <interpolation_type, 1=simple, 2=gaussian>" << std::endl;
		return EXIT_FAILURE;
	}

	_sc_fn  = argv[1];
	argc--;
	argv++;
	_v_fn = argv[1];
	argc--;
	argv++;
	_navx = argv[1]; 
	argc--; 
	argv++; 
	_interpolationtype = atoi(argv[1]);
	argc--; 
	argv++; 
	
	
  vtkSmartPointer<vtkPolyDataReader> reader1 = vtkSmartPointer<vtkPolyDataReader>::New();

  reader1->SetFileName(_sc_fn);
  reader1->Update(); 
		
  _sc_poly = reader1->GetOutput(); 
	

  cout << "Need to know how many files there are to import >"; 
	cin >> num_files; 
	
	ensite.VoltageFileReadDriver(_navx, num_files); 
	ensite.TransformXYZVoltagePointsToOurOrientation();
cout << "\nout of TransformXYZVoltagePointsToOurOrientation" << endl;	
	ensiteGraphics.SetVoltageXYZVector(ensite._voltage_xyz); 
	ensiteGraphics.SetVoltageTraceVector(ensite._voltage_trace); 
cout << "\nout of SetVoltageTraceVector" << endl;	
	ensiteGraphics.SetVoltagePeak2PeakVector(ensite._voltage_peak2peak); 
cout << "\nout of SetVoltagePeak2PeakVector" << endl;	
	ensiteGraphics.SetVoltageTempVector(ensite._voltage_peak2peak);				// important as interpolaters now use _voltage_temp to look for voltage
	cout << "\nout of SetVoltageTempVector" << endl;	
	ensite.getClosestPointsOnGeomtryMesh(_sc_poly, voltageNodeList); 

	ensiteGraphics.ComputeShortestPathsUsingDijkstra(_sc_poly, voltageNodeList);
	ensiteGraphics.RelaxNodesWithInfiniteDis(_sc_poly); 
	//ensiteGraphics.computeVoltagePeakToPeakFromWaveform();		
	if (_interpolationtype == 1)
		ensiteGraphics.InterpolateScalars_SimpleInterpolationAtCells(_sc_poly);			// simple interpolation 
	else
		ensiteGraphics.InterpolateScalars(_sc_poly, voltageNodeList, 0, 2);			// COmplex gaussian k-nearest neighbour inteprolation 

	vtkSmartPointer<vtkPolyDataWriter> writer = vtkSmartPointer<vtkPolyDataWriter>::New();
	writer->SetFileName(_v_fn); 
	writer->SetInput(_sc_poly); 
	writer->Update(); 
}

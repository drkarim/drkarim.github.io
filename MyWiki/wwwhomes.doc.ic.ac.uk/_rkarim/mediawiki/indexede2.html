Introduction

You have a list of file from a set of experiments 

Dice_140.txt 
Dice_168.txt 
....

Each of the files has 20 x 20 dice numbers. So it is 20 outputs from graphcuts and the Dice is calculated locally for 20 selected regions. For example, _gc_seg_out_1 generates the first 20 Dice numbers, _gc_seg_out_2 generates the next 20 and so on...

Code 

% Opt = 1 - you want the Dice of the experiment with best median Dice
% Opt = 2 - you want the maximum Dice in each experiment 
% Opt = 3 - you want the Dice of a specific experiment, specify using
% WhichExp or else defaults to first experiment 
function [D, Median_max, Median_max_idx] = SelectBestDice(Opt, DiceOutOf, WhichExp)    

    if nargin < 3
        WhichExp = 1        % defaults
    end
    
    A = [140 168 196 224 252 280 308 336 364 392 420]
    n = size(A,2)
    D=zeros(20,n)
    if DiceOutOf == 100 
        factor=1
    else
        factor=100
    end
     Median_max = zeros(n,1);
     Median_max_idx = ones(n,1);
    for i=1:n
           
            filename = strcat('dice_',int2str(A(1,i)));
            filename = strcat(filename,'.txt');
           % disp(filename)
            B = load(filename)
            B = B*factor;
       if Opt == 1
            for j=1:size(B,1)
                k = mod(j,20) 

                if k~=0 
                    C(k, 1)=B(j,1)
                end
                if k==0 
                    C(20,1)=B(j,1)
                    if Median_max(i,1) < median(C)
                        Median_max(i,1) = median(C)
                        Median_max_idx(i,1) = j
                    end
                    C=[]

                end
            end 
            if Median_max_idx(i,1)-20 >= 0
                D(:,i)=B(Median_max_idx(i,1)-19:Median_max_idx(i,1),1)
                E(:,1)=0
            else
                E(i,1)=A(1,i)
            end
        
       elseif Opt == 2
        
            for j=1:size(B,1)
                k=mod(j,20)
                if k~=0
                    if D(k, i)<B(j,1)
                        D(k,i)=B(j,1)
                    end
                end

                if k==0
                    if D(20,i)<B(j,1)
                        D(20,i)=B(j,1)
                    end
                end

            end
            
       elseif Opt == 3      
           
           for j=1:20
               D(j, i) = B(j*WhichExp,1)
           end 
           
       end
        
       
       
    end


end

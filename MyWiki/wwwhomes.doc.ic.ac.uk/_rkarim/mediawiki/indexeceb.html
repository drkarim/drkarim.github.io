#include "rk_test3.h"

void someFunc() 
{
}

int main(int argc, char **argv)
{
	int option;
	bool foundArgs1=false,foundArgs2=false,foundArgs3=false, foundMethod=false; 
	irtkRealImage img1, img2; 
	irtkGreyImage img3; 
	char* input_f1="", *output_f="", *input_f2=""; 
	int t1, t2, t3; 

	if (argc >= 1)
	{
			for (int i = 1; i < argc; i++) {
				if (i + 1 != argc) {
					if (string(argv[i]) == "-i1") {
						input_f1 = argv[i+1]; 
						foundArgs1 = true;
					} else if (string(argv[i]) == "-i2") {
						input_f2 = argv[i+1]; 
						foundArgs2 = true;
					} else if (string(argv[i]) == "-o") {
						output_f = argv[i+1]; 
						foundArgs3 = true;
					}

					 // all method switches here 
					else if (string(argv[i]) == "--method1") {
						option = 1; 
						foundMethod = true;
					}
				}
			}
	}

	if (!(foundArgs1 && foundArgs2 & foundArgs3) && !foundMethod)
	{
		cout << "\n\nUsage: app \n\n\tArguments:"
			"\n\t -i1 \t - vtk surface"
			"\n\t -i2 \t - Points selected on shell or unfold" 
			"\n\t -o \t - output text file with area of each selected shell segment"
			"\n\n";
		exit(0);
	}

	if (option == 1) 
	{
		someFunc();
	}

}

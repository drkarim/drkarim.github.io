// If the mesh is a topological disk, extract its longest border,
// else compute a very simple cut to make it homeomorphic to a disk.
// Return the border of this region (empty on error)
//
// CAUTION: this cutting algorithm is very naive. Write your own!
static Seam cut_mesh(Parameterization_polyhedron_adaptor& mesh_adaptor)
{
    // Helper class to compute genus or extract borders
    typedef CGAL::Parameterization_mesh_feature_extractor<Parameterization_polyhedron_adaptor>
                                            Mesh_feature_extractor;

    Seam seam;              // returned list

    // Get reference to Polyhedron_3 mesh
    Polyhedron& mesh = mesh_adaptor.get_adapted_mesh();

    // Extract mesh borders and compute genus
    Mesh_feature_extractor feature_extractor(mesh_adaptor);
    int nb_borders = feature_extractor.get_nb_borders();
    int genus = feature_extractor.get_genus();

    // If mesh is a topological disk
    if (genus == 0 && nb_borders > 0)
    {
        // Pick the longest border
        seam = feature_extractor.get_longest_border();
		std::cout << "The mesh was found to be a topological disk" << endl;
    }
    else // if mesh is *not* a topological disk, create a virtual cut
    {
        const int CUT_LENGTH = 6;

        // Build consecutive halfedges array
        Polyhedron::Halfedge_handle seam_halfedges[CUT_LENGTH];
        seam_halfedges[0] = mesh.halfedges_begin();
        if (seam_halfedges[0] == NULL)
            return seam;                    // return empty list
        int i;
        for (i=1; i<CUT_LENGTH; i++)
        {
            seam_halfedges[i] = seam_halfedges[i-1]->next()->opposite()->next();
            if (seam_halfedges[i] == NULL)
                return seam;                // return empty list
        }

        // Convert halfedges array to two-ways vertices list
        for (i=0; i<CUT_LENGTH; i++)
            seam.push_back(seam_halfedges[i]->vertex());
        for (i=CUT_LENGTH-1; i>=0; i--)
            seam.push_back(seam_halfedges[i]->opposite()->vertex());
    }

    return seam;
}


static bool write_vtk_output(Parameterization_polyhedron_adaptor& mesh_adaptor, char* vtk_filename, char* xyz_uv_map_filename, 
							 char* xyz_uv_max_id_filename, vtkPolyData* polyWithColors)
{
	Polyhedron& P = mesh_adaptor.get_adapted_mesh();
	vtkPolyData *poly = vtkPolyData::New();
	vtkPoints *points = vtkPoints::New();
	vtkCellArray *cells = vtkCellArray::New();
	vtkFloatArray *scalars = vtkFloatArray::New();
	int p_counter=-1, faceCounter=0, numPointsFound; 
	float* xyz2D = new float[3]; 
	double* xyz3D = new double[3];
	vtkIdType* pointIDs = new vtkIdType[3]; 
	vtkIdType pointIDIndex, maxPointIDIndexIn3D=-1, maxPointIDIndexIn2D=-1; 
	bool pointFound;
	vtkIdType closestPointID=-1;
	
	ofstream out, outMax; 
	out.open(xyz_uv_map_filename);
	outMax.open(xyz_uv_max_id_filename);

	
	// to search for closest point on polyWithColors for transfering the colors from that shell 
	vtkSmartPointer<vtkPointLocator> point_locator = vtkSmartPointer<vtkPointLocator>::New();
	point_locator->SetDataSet(polyWithColors); 
	point_locator->AutomaticOn(); 
	point_locator->BuildLocator();
	vtkFloatArray* colors = vtkFloatArray::New();
	colors = vtkFloatArray::SafeDownCast(polyWithColors->GetPointData()->GetScalars());
	
	
	std::vector<Point_2> point_list; 
	// first iterate through points and insert points 
	Polyhedron::Halfedge_const_iterator pHalfedge;
    for (pHalfedge = P.halfedges_begin();
         pHalfedge != P.halfedges_end();
         pHalfedge++)
    {
		double x1 = mesh_adaptor.info(pHalfedge->prev())->uv().x() ;
        double y1 =  mesh_adaptor.info(pHalfedge->prev())->uv().y();
        double x2 =  mesh_adaptor.info(pHalfedge)->uv().x();
        double y2 =  mesh_adaptor.info(pHalfedge)->uv().y();
		point_list.push_back(mesh_adaptor.info(pHalfedge->prev())->uv()); 
		point_list.push_back(mesh_adaptor.info(pHalfedge)->uv()); 
		xyz2D[0] = x1; 
		xyz2D[1] = y1; 
		xyz2D[2] = 0;		// no z coordinate 
		points->InsertPoint(++p_counter, xyz2D); 

		if (maxPointIDIndexIn2D < p_counter) 
			maxPointIDIndexIn2D = p_counter;				// keeping track of the maximum point id in 2D mesh 

		xyz3D[0] = (pHalfedge->prev()->vertex()->point()).x(); 
		xyz3D[1] = (pHalfedge->prev()->vertex()->point()).y(); 
		xyz3D[2] = (pHalfedge->prev()->vertex()->point()).z(); 
		
		// now transfer the colors from the vtk shell 
		closestPointID = point_locator->FindClosestPoint(xyz3D); 
		if (closestPointID != -1) { 
			scalars->InsertNextTuple1(colors->GetTuple1(closestPointID)); 
			out << p_counter << "\t" << closestPointID << endl;

			if (maxPointIDIndexIn3D < closestPointID)
				maxPointIDIndexIn3D = closestPointID;			// keeping track of the maximum point id in 3D mesh 
		}
		
		// move to the other vertex of the half-edge 
		xyz2D[0] = x2; 
		xyz2D[1] = y2; 
		xyz2D[2] = 0; 
		points->InsertPoint(++p_counter, xyz2D); 

		if (maxPointIDIndexIn2D < p_counter) 
			maxPointIDIndexIn2D = p_counter;					// keeping track of the maximum point id in 2D mesh 
		
		xyz3D[0] = (pHalfedge->vertex()->point()).x(); 
		xyz3D[1] = (pHalfedge->vertex()->point()).y(); 
		xyz3D[2] = (pHalfedge->vertex()->point()).z(); 
		
		// now transfer the colors from the vtk shell 
		closestPointID = point_locator->FindClosestPoint(xyz3D); 
		if (closestPointID != -1) { 
			scalars->InsertNextTuple1(colors->GetTuple1(closestPointID)); 
			out << p_counter << "\t" << closestPointID << endl;

			if (maxPointIDIndexIn3D < closestPointID)
				maxPointIDIndexIn3D = closestPointID;
		}

		

	}
	std::cout << "Created vtk points .. " << endl;

	pointIDIndex = 0; 
	faceCounter = 0;
	 
	// constructing cells of the flattened polygonal data by visiting each cell in 
	// original and finding its flattened uv co-ordinates using adapater
	// Inside do-while: retrieving the ID of flattened point around a flattened cell 
	for (  Facet_iterator i = P.facets_begin(); i != P.facets_end(); ++i) {
		
		pointIDIndex = 0;
		numPointsFound = 0;

        Halfedge_facet_circulator j = i->facet_begin();
        // Facets in polyhedral surfaces are at least triangles.
        CGAL_assertion( CGAL::circulator_size(j) == 3);
        //std::cout << CGAL::circulator_size(j) << ' ';
        do {
			pointFound = false; 
		
			for (int k=0;k<point_list.size() && !pointFound;k++)		// determine point's id 
			{
				if (mesh_adaptor.info(j)->uv() == point_list[k]) { 
					pointIDs[pointIDIndex++] = k;			// point found, retrieve ID and store 
					pointFound = true; 
					numPointsFound++; 
				}
			}
        } while ( ++j != i->facet_begin());
		
		CGAL_assertion( numPointsFound == 3);
		cells->InsertNextCell(3, pointIDs); 
		//scalars->InsertTuple1(faceCounter++, 1);		// all scalars default to 1 at the moment - no scalar information 
		
		faceCounter++; 
    }
	cout << "All cells for (flattened) polydata are now created .. " << endl;
	poly->SetPoints(points); 
	poly->SetPolys(cells); 
	poly->GetPointData()->SetScalars(scalars); 
	vtkPolyDataWriter *writer = vtkPolyDataWriter::New(); 
	writer->SetInput(poly); 
	writer->SetFileName(vtk_filename); 
	writer->Update(); 

	outMax << maxPointIDIndexIn3D << endl; 
	outMax << maxPointIDIndexIn2D << endl; 
	
	out.close();
	outMax.close(); 
}



// Dump parameterized mesh to an eps file
static bool write_file_eps(const Parameterization_polyhedron_adaptor& mesh_adaptor,
                           const char *pFilename,
                           double scale = 500.0)
{
    const Polyhedron& mesh = mesh_adaptor.get_adapted_mesh();

    std::ofstream out(pFilename);
    if(!out)
        return false;
    CGAL::set_ascii_mode(out);

    // compute bounding box
    double xmin,xmax,ymin,ymax;
    xmin = ymin = xmax = ymax = 0;
    Polyhedron::Halfedge_const_iterator pHalfedge;
    for (pHalfedge = mesh.halfedges_begin();
         pHalfedge != mesh.halfedges_end();
         pHalfedge++)
    {
        double x1 = scale * mesh_adaptor.info(pHalfedge->prev())->uv().x();
        double y1 = scale * mesh_adaptor.info(pHalfedge->prev())->uv().y();
        double x2 = scale * mesh_adaptor.info(pHalfedge)->uv().x();
        double y2 = scale * mesh_adaptor.info(pHalfedge)->uv().y();
        xmin = (std::min)(xmin,x1);
        xmin = (std::min)(xmin,x2);
        xmax = (std::max)(xmax,x1);
        xmax = (std::max)(xmax,x2);
        ymax = (std::max)(ymax,y1);
        ymax = (std::max)(ymax,y2);
        ymin = (std::min)(ymin,y1);
        ymin = (std::min)(ymin,y2);
    }

    out << "%!PS-Adobe-2.0 EPSF-2.0" << std::endl;
    out << "%%BoundingBox: " << int(xmin+0.5) << " "
                                << int(ymin+0.5) << " "
                                << int(xmax+0.5) << " "
                                << int(ymax+0.5) << std::endl;
    out << "%%HiResBoundingBox: " << xmin << " "
                                    << ymin << " "
                                    << xmax << " "
                                    << ymax << std::endl;
    out << "%%EndComments" << std::endl;
    out << "gsave" << std::endl;
    out << "0.1 setlinewidth" << std::endl;

    // color macros
    out << std::endl;
    out << "% RGB color command - r g b C" << std::endl;
    out << "/C { setrgbcolor } bind def" << std::endl;
    out << "/white { 1 1 1 C } bind def" << std::endl;
    out << "/black { 0 0 0 C } bind def" << std::endl;

    // edge macro -> E
    out << std::endl;
    out << "% Black stroke - x1 y1 x2 y2 E" << std::endl;
    out << "/E {moveto lineto stroke} bind def" << std::endl;
    out << "black" << std::endl << std::endl;

    // for each halfedge
    for (pHalfedge = mesh.halfedges_begin();
         pHalfedge != mesh.halfedges_end();
         pHalfedge++)
    {
        double x1 = scale * mesh_adaptor.info(pHalfedge->prev())->uv().x();
        double y1 = scale * mesh_adaptor.info(pHalfedge->prev())->uv().y();
        double x2 = scale * mesh_adaptor.info(pHalfedge)->uv().x();
        double y2 = scale * mesh_adaptor.info(pHalfedge)->uv().y();
        out << x1 << " " << y1 << " " << x2 << " " << y2 << " E" << std::endl;
    }

    /* Emit EPS trailer. */
    out << "grestore" << std::endl;
    out << std::endl;
    out << "showpage" << std::endl;

	
    return true;
}


// ----------------------------------------------------------------------------
// main()
// ----------------------------------------------------------------------------

int main(int argc, char * argv[])
{
    std::cerr << "PARAMETERIZATION" << std::endl;
    std::cerr << "  Discrete Authalic Parameterization" << std::endl;
    std::cerr << "  Square border" << std::endl;
    std::cerr << "  TAUCS solver" << std::endl;
    std::cerr << "  Very simple cut if model is not a topological disk" << std::endl;
    std::cerr << "  Output: EPS" << std::endl;

    //***************************************
    // decode parameters
    //***************************************

    if (argc-1 != 3)
    {
        std::cerr << "Usage: " << argv[0] << " input_file.off output_file.eps" << std::endl;
        return(EXIT_FAILURE);
    }

    // File names are:
    const char* input_filename  = argv[1];
    const char* output_filename = argv[2];

    //***************************************
    // Read the mesh
    //***************************************

    // Read the mesh
    std::ifstream stream(input_filename);
    Polyhedron mesh;
    stream >> mesh;
    if(!stream || !mesh.is_valid() || mesh.empty())
    {
		std::cerr << "Error: cannot read OFF file due to\nStream: " << !stream << "\nMesh is valid = " << mesh.is_valid() << "\n"
			"Mesh empty =  "<< mesh.empty() << "\n\n" << input_filename << std::endl;
        return EXIT_FAILURE;
    }

    //***************************************
    // Create Polyhedron adaptor
    //***************************************

    Parameterization_polyhedron_adaptor mesh_adaptor(mesh);

    //***************************************
    // Virtually cut mesh
    //***************************************

    // The parameterization methods support only meshes that
    // are topological disks => we need to compute a "cutting" of the mesh
    // that makes it homeomorphic to a disk
    Seam seam = cut_mesh(mesh_adaptor);
    if (seam.empty())
    {
        std::cerr << "Input mesh not supported: the example cutting algorithm is too simple to cut this shape" << std::endl;
        return EXIT_FAILURE;
    }

    // Create a second adaptor that virtually "cuts" the mesh following the 'seam' path
    typedef CGAL::Parameterization_mesh_patch_3<Parameterization_polyhedron_adaptor>
                                            Mesh_patch_polyhedron;
    Mesh_patch_polyhedron   mesh_patch(mesh_adaptor, seam.begin(), seam.end());
    if (!mesh_patch.is_valid())
    {
        std::cerr << "Input mesh not supported: non manifold shape or invalid cutting" << std::endl;
        return EXIT_FAILURE;
    }

    //***************************************
    // Discrete Authalic Parameterization (square border)
    // with TAUCS solver
    //***************************************

    // Border parameterizer
    typedef CGAL::Square_border_arc_length_parameterizer_3<Mesh_patch_polyhedron>
                                                            Border_parameterizer;
    // TAUCS solver
    typedef CGAL::Taucs_solver_traits<double>               Solver;

    // Discrete Authalic Parameterization (square border)
    // with TAUCS solver
    typedef CGAL::Discrete_authalic_parameterizer_3<Mesh_patch_polyhedron,
                                                    Border_parameterizer,
                                                    Solver> Parameterizer;

    Parameterizer::Error_code err = CGAL::parameterize(mesh_patch, Parameterizer());
    switch(err) {
    case Parameterizer::OK: // Success
        break;
    case Parameterizer::ERROR_EMPTY_MESH: // Input mesh not supported
    case Parameterizer::ERROR_NON_TRIANGULAR_MESH:
    case Parameterizer::ERROR_NO_TOPOLOGICAL_DISC:
    case Parameterizer::ERROR_BORDER_TOO_SHORT:
        std::cerr << "Input mesh not supported: " << Parameterizer::get_error_message(err) << std::endl;
        return EXIT_FAILURE;
        break;
    default: // Error
        std::cerr << "Error: " << Parameterizer::get_error_message(err) << std::endl;
        return EXIT_FAILURE;
        break;
    };

    //***************************************
    // Output
    //***************************************

    // Write Postscript file 
	/*
    if ( ! write_file_eps(mesh_adaptor, output_filename) )
    {
        std::cerr << "Error: cannot write file " << output_filename << std::endl;
        return EXIT_FAILURE;
    }*/
	vtkSmartPointer<vtkPolyDataReader> reader = vtkSmartPointer<vtkPolyDataReader>::New(); 
	reader->SetFileName(argv[3]); 
	reader->Update();
	write_vtk_output(mesh_adaptor, "vtkOutput.vtk", "uv_xyz_mapping.txt", "uv_xyz_pointID_max.txt", reader->GetOutput()); 
    return EXIT_SUCCESS;
}

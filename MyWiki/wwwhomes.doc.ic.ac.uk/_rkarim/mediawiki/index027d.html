Introduction

Plots a stacked-histogram from data that is obtained from running this script. Note the Ln output from the script is input into this script. The interval and max values used were 200 and 2000 respectively. Here is an example usage: 

[val] = drawPatchedHistogram(Ln, 200, 2000, 80);


Code 


% for scar_cutoff use 70 or 80, dependending on what is the lowest cut-off
% for scar. 70 meaning, 70% of surrounding was labelled (5 or 10mm radius)
% as scar by expert
function [val] = drawPatchedHistogram(Ln, interval, max, scar_cutoff)
    
   w = (0.3*interval); 
   val = zeros(3,max/interval);
  
   % The following elected ablation lesions were removed, as a high fti was
   % surrounded by 0 scar
   exclude_ftis  = [1916; 1909; 3042; 2075; 2033; 1472; 1469; 1608; 1565; 1540; 1531; 1521];
   exclude_index = []; 
   for i=1:size(exclude_ftis) 
       Lnt = Ln(:,1) 
       exclude_index = [exclude_index; find(Lnt == exclude_ftis(i))] 
   end
   
   Ln([exclude_index], :) = []; 
   
   
   %for i=(0+interval)/2:interval:max
   for i=1:max/interval
       I = Ln(:,1) >= (i-1)*interval & Ln(:,1) < i*interval;
       J1 = Ln(I,2) > scar_cutoff;
       J2 = Ln(I,2) > 0 & Ln(I,2) <= scar_cutoff;
       J3 = Ln(I,2) <= 0;
       val(1,i) = 100*sum(J1)/sum(I);
       val(2,i) = 100*sum(J2)/sum(I);
       val(3,i) = 100*sum(J3)/sum(I);
       
   end
   
   figure
   set(gca, 'FontSize', 20);       % set the size of the X-axis labels here
   xlabel('Force Time Integral (FTI) \pm{100} gs', 'FontSize', 25);
   %ylabel('Frequency', 'FontSize', 25);
   ylabel('Number of Lesions (%)', 'FontSize', 25);
   box on;
   for i=1:max/interval
       x1 = (i-0.5)*interval - w;
       x2 = (i-0.5)*interval + w;
       patch([x1 x2 x2 x1], [0 0 val(3,i) val(3,i)], 0.35*ones(1,3));
       patch([x1 x2 x2 x1], [0 0 val(2,i) val(2,i)]+val(3,i), 0.5*ones(1,3));
       patch([x1 x2 x2 x1], [0 0 val(1,i) val(1,i)]+val(3,i)+val(2,i), 0.75*ones(1,3));
   end
   set(gca, 'XTick', [100:200:2000]);
   %set(gca, 'XTickLabel', {'200','100','400-600','600-800','800-1000','1000-1200','1200-1400','1400-1600', '1600-1800', '1800-2000'});
   set(gca, 'YLim', [0 105]);
   set(gca, 'YTick', [0:10:100]);
end

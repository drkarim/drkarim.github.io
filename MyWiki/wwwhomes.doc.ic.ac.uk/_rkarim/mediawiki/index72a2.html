Usage

The output of staple implemented using ITKs staple library function generates an output image with pixel values 0 to 1. To threshold to binary image, the following program performs simple thresholding. A threshold of 0.5 - 0.7 works well. 

FixStaple, thresholds the output of ITK staple:
Arguments: <staple image> <new image> <threshold>

Note: In phani-pc you will also find this code as called staple_threshold.exe

Code 

#if defined(_MSC_VER)
#pragma warning ( disable : 4786 )
#endif

#include <irtkImage.h>
#include <vector>

int main(int argc, char **argv)
{
	irtkRealImage img; 
	double t;
	
	char *filename1, *filename2; 
	if (arc < 3) 
	{
		cout << "FixStaple, thresholds the output of ITK staple:\nArguments:\t<staple image> <new image> <threshold>\n";
	}
	filename1 = argv[1];
	argv++;
	argc--;
	filename2 = argv[1];
	argv++;
	argc--;
	t = atof(argv[1]);
	argv++;
	argc--;
	
	img.Read(filename1); 

	irtkRealPixel* img_p; 
	img_p = img.GetPointerToVoxels(); 

	for (int i=0;i<img.GetNumberOfVoxels();i++)
	{
		if (*img_p > t)
			*img_p = 1; 
		else
			*img_p = 0; 
		img_p++; 
	}
	
	img.Write(filename2);
}

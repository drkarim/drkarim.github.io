Write mask

Writes intensities within binary mask to file

Usage 

Writemask.exe -i1 <intensity img> -i2 <mask> -o <output text file>

Code 

#include <irtkImage.h>


#include <stdio.h>    // to get "printf" function
#include <stdlib.h>   // to get "free" function
#include <fstream>   

// fn1 is intensity image 
// fn2 is mask image 
// output_f is text output file 
void WriteMask(char* fn1, char* fn2, char* output_f) 
{
	irtkRealImage mask, img; 
	img.Read(fn1);
	mask.Read(fn2); 
	ofstream out; 
	
	irtkRealPixel* img_p, *mask_p; 

	img_p = img.GetPointerToVoxels();
	mask_p = mask.GetPointerToVoxels();
	out.open(output_f); 

	for (int i=0;i<mask.GetNumberOfVoxels();i++)
	{
		
		if (*mask_p > 0) 
		{
			out << *img_p << endl; 
		}

		img_p++; 
		mask_p++;
	}
	
	out.close();
}


int main(int argc, char **argv)
{
	int option;
	bool foundArgs1=false,foundArgs2=false,foundArgs3=false, foundArgs4 = false, foundMethod=false, tofile =false; 
	irtkRealImage img1, img2; 
	irtkGreyImage img3; 
	char* input_f1="", *output_f="", *input_f2=""; 
	int t1, t2, t3; 

	if (argc >= 1)
	{
			for (int i = 1; i < argc; i++) {
				if (i + 1 != argc) {
					if (string(argv[i]) == "-i1") {
						input_f1 = argv[i+1]; 
						foundArgs1 = true;
					} else if (string(argv[i]) == "-i2") {
						input_f2 = argv[i+1]; 
						foundArgs2 = true;
					} else if (string(argv[i]) == "-o") { 
						output_f = argv[i+1];
						tofile = true; 
					}					
					
				}
			}
	}

	if (!(foundArgs1 && foundArgs2))	{
		cout << "\n\nUsage: writemask \n\n\tArguments:"
			"\n\t -i1 \t - intensity image"
			"\n\t -i2 \t - mask (segmentation)" 
			"\n\t -o \t - output text file"
			"\n\n";
		exit(0);
	}
	else
	{
		WriteMask(input_f1, input_f2, output_f);
	}
	

}

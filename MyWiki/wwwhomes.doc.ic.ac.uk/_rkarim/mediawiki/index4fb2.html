#include <iostream>
#include <string>
#include <sstream>


#include <cmath> 
#include <ctime>
#include "printer3DBoxCore.h"

using namespace std;
using namespace boost;

int main(int argc, char* argv[])
{
	
	string readChar;
	string writeChar;
	string writeChars="ddBBBddd1"; 

	TimeoutSerial serial("COM7",9600);
	serial.setTimeout(posix_time::seconds(4));
	for (int i=0;i<writeChars.length();i++) 
	{
		//cout << i << endl;
		
		try {
			//Text test
			writeChar = writeChars.at(i);
			cout << "Sending " << writeChar << ", waiting for reply .. reply = ";
			serial.writeString(writeChar); 
			
			while (readChar != writeChar) {
				try { 
					readChar = serial.readString(1);				
				}
				catch (timeout_exception& e)
				{
					readChar = writeChar; 
				}
			}

			cout << readChar << endl;
			readChar = "";

	 
		} catch(boost::system::system_error& e)
		{
			cout<<"Error: "<<e.what()<<endl;
			return 1;
		}
	}
	serial.close();

}

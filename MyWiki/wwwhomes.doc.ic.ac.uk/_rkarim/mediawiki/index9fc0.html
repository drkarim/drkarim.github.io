Introduction

This should be used in conjucntion with boundary computation which computes the boundaries of a binary mask.

Usage

void getMeanVariance(vector<double> val, double& mean, double& variance, double& total)
{
	double sum=0.0, N, sum_var=0.0; 

	N = val.size(); 

	for (int i=0;i<N;i++)
	{
		sum += val[i]; 
	}

	mean = sum/N;

	for (int i=0;i<N;i++)
	{
		sum_var += (val[i] - mean)*(val[i] - mean);
	}

	variance = sum_var/(N-1);
	total = sum;
}


double getEuclideanDistanceIn3D(int x1, int y1, int z1,int x2,int y2, int z2) 
{
	double* p1 = new double[3]; 
	double* p2 = new double[3];
	p1[0] = x1; p1[1] = y1; p1[2] = z1; 
	p2[0] = x2; p2[1] = y2; p2[2] = z2; 

	double s;
	s = ((p1[0] - p2[0])*(p1[0] - p2[0])) + ((p1[1] - p2[1])*(p1[1] - p2[1])) + ((p1[2] - p2[2])*(p1[2] - p2[2])); 
	return sqrt(s);
}

void GetSlice(irtkGreyImage* img1, irtkGreyImage& img_crop, int in_x, int in_y, int in_z)
{
	int maxX, maxY, maxZ;
	maxX = img1->GetX(); 
	maxY = img1->GetY(); 
	maxZ = img1->GetZ(); 
	int n=0;

	//irtkGreyImage img_crop; 
	

	if (in_x > 0 || in_y > 0 || in_z > 0) 
	{
		if (in_x > 0)
		{
			img_crop = img1->GetRegion(in_x, 0,0, in_x+1, maxY, maxZ); 
		}
		else if (in_y > 0)
		{
			img_crop = img1->GetRegion(0, in_y,0, maxX, in_y+1, maxZ); 
		}
		else if (in_z > 0)
		{
			img_crop = img1->GetRegion(0, 0,in_z, maxX, maxY, in_z+1);
		}
	}

	
}

void BoundaryDistance(irtkGreyImage* img1_crop, irtkGreyImage* img2_crop, double& total, double& mean, double& std) 
{
	int maxX, maxY, maxZ; 
	int curr_minX=1e9, curr_minY=-1, curr_minZ=-1; 
	double curr_dist=-1; 
	double curr_min_dist = 1e9; 
	vector<double> min_dists; 
	double variance; 
	
	maxX = img1_crop->GetX(); 
	maxY = img1_crop->GetY(); 
	maxZ = img1_crop->GetZ(); 
	
	for (int x=0;x<maxX;x++) 
	{
		for (int y=0;y<maxY;y++)
		{
			for (int z=0;z<maxZ;z++) 
			{
				if (img1_crop->Get(x,y,z) > 0) 
				{
					curr_min_dist=1e9;  
					
					// find its nearest pixel that is foreground 
					for (int x1=0;x1<maxX;x1++)
					{
						for (int y1=0;y1<maxY;y1++)
						{
							for (int z1=0;z1<maxZ;z1++) 
							{
								if (img2_crop->Get(x1,y1,z1) > 0) 
								{	
									curr_dist = getEuclideanDistanceIn3D(x,y,z,x1,y1,z1); 
									//cout << curr_dist << endl;
									if (curr_min_dist > curr_dist) 
									{
										curr_min_dist = curr_dist; 
									}
								}
							}
						}
					}
					
					if (curr_min_dist < 1e9) 
						min_dists.push_back(curr_min_dist); 
				}
			}
		}
	}

	 getMeanVariance(min_dists, mean, variance, total);
	 std = sqrt(variance); 
	
}


void GetBoundaryError(irtkGreyImage* img1, irtkGreyImage* img2, ofstream& fileIO, int x_y_z) 
{
	int maxX, maxY, maxZ; 
	
	double total_error, mean, std; 
	maxX = img1->GetX(); 
	maxY = img1->GetY(); 
	maxZ = img1->GetZ(); 
	
	//vector<double> total_errors; 

	irtkGreyImage img1_crop, img2_crop; 
	fileIO << "Slice No.\t" << "Total error\t" << "Mean error\t" << "Std. error\t" << endl; 
	fileIO << "=========\t" << "===========\t" << "==========\t" << "==========\t" << endl; 

	if (x_y_z == 1) 
	{
		// along x-direction 
		for (int x=0;x<maxX;x++)
		{
			GetSlice(img1, img1_crop, x, 0, 0);
			GetSlice(img2, img2_crop, x, 0, 0);
			
			BoundaryDistance(&img1_crop, &img2_crop, total_error, mean, std); 
			fileIO << x << "\t" << total_error << "\t" << mean << "\t" << std << endl;  
			//total_errors.push_back(total_error); 
		}

	}
	
	else if (x_y_z == 2) 
	{
		// along y-direction 
		for (int y=0;y<maxY;y++)
		{
			GetSlice(img1, img1_crop, 0, y, 0);
			GetSlice(img2, img2_crop, 0, y, 0);

			BoundaryDistance(&img1_crop, &img2_crop, total_error, mean, std); 
			fileIO << y << "\t" << total_error << "\t" << mean << "\t" << std << endl;  
			//total_errors.push_back(total_error); 
			
		}
	}

	else if (x_y_z == 3) 
	{

		// along z-direction 
		for (int z=0;z<maxZ;z++)
		{
			GetSlice(img1, img1_crop, 0, 0, z);
			GetSlice(img2, img2_crop, 0, 0, z);

			BoundaryDistance(&img1_crop, &img2_crop, total_error, mean, std); 
			fileIO << z << "\t" << total_error << "\t" << mean << "\t" << std << endl;  
			//total_errors.push_back(total_error); 
			
		}
	}
	

}


void Binarize(irtkRealImage* at_wall)
{
	irtkRealPixel *p; 
	p = at_wall->GetPointerToVoxels(); 
	for (int i=0;i<at_wall->GetNumberOfVoxels();i++)
	{
		if (*p>0) 
		{ 
			*p = 1;
		}
		p++; 
	}
	//at_wall->Write("at_wall.gipl");
}


int main(int argc, char **argv)
{
	int optind;
	bool foundArgs=false;
	irtkGreyImage img1, img2;
	int x_y_z=1;			
	
	char* input_f1="", *input_f2="", *output_f="";
	double t1=1, mean, stdev, total_voxels, percent;

	if (argc >= 1) 
	{
		int optind=1;
		
		while ((optind < argc) && (argv[optind][0]=='-')) {
			
			string sw = argv[optind];
			
			if (sw == "-i1") {
				optind++;
				input_f1 = argv[optind]; 
				foundArgs = true;	
			}
			
			if (sw == "-i2") {
				optind++;
				input_f2 = argv[optind]; 
				foundArgs = true;	
			}
			

			else if (sw == "-o") {
				optind++;
				output_f = argv[optind]; 
			}
				
			else if (sw == "--x") 
			{
				x_y_z = 1;
				cout << "\n\nNote: Along x"; 
			}

			else if (sw == "--y") 
			{
				x_y_z = 2;
				cout << "\n\nNote: Along x"; 
			}

			else if (sw == "--z") 
			{
				x_y_z = 3;
				cout << "\n\nNote: Along x"; 
			}
			
			
			optind++; 
		}
	}
	
	if (foundArgs == false)
	{
		cout << "Usage: boundaryerror.exe -i1 <boundary1> -i2 <boundary2> -o <output txt> --x --y --z <along which direction>" << endl; 
		exit(0); 
	}

	else
	{
		img1.Read(input_f1); 
		img2.Read(input_f2);

		// check images are same size 
		if (!(img1.GetX() == img2.GetY() && img1.GetY() == img2.GetY() && img1.GetZ() == img2.GetZ()))
		{
			cout << "Image size mismatch, images need to be of equal dimensions! " << endl; 
			exit(0); 
		}
		
		ofstream fileIO; 
		fileIO.open(output_f); 
		
		GetBoundaryError(&img1, &img2, fileIO, x_y_z);
		
	}

}

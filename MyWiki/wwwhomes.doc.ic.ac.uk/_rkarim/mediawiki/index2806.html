/*
*	measure = 1 , will return the mean 
*	measure = 2 , will return the std. deviation (but must provide the mean) 
*/
double getStats(int measure, double mean)
{
	double sum=0, max=-1, t; 
	double size = i_ctr.size(); 

	if (measure == 1)			// reutnr mean 
	{
		for (int i=0;i<size;i++)
		{
			sum+=i_ctr[i]; 
		}

		if (size > 0)
			return sum/size;
		else 
			return 0; 
	}
	if (measure == 2)			// return stdev
	{
		for (int i=0;i<size;i++)
		{
			t = i_ctr[i]; 
			sum += ((t-mean)*(t-mean));
		}
		if (size > 0)
			return sqrt(sum / size);
		else 
			return 0; 
	}
}

// taken from http://mathbits.com/mathbits/compsci/Arrays/Bubble.htm
void bubble_sort(double *array, int arrayLength)
{
      int i, j, flag = 1;    // set flag to 1 to begin initial pass
      double temp;             // holding variable
     
     for(i = 1; (i <= arrayLength) && flag; i++)
     {
         flag = 0;
         for (j=0; j < (arrayLength -1); j++)
         {
              if (array[j+1] > array[j])      // ascending order simply changes to <
              { 
                    temp = array[j];             // swap elements
                    array[j] = array[j+1];
                    array[j+1] = temp;
                    flag = 1;               // indicates that a swap occurred.
               }
          }
     }
     return;   //arrays are passed to functions by address; nothing is returned
}

double GetMeanOfTopIntensities(irtkRealImage* img, int top_n)
{
	double* i_arr, *topN;  
	vector<double> i_ctr; 

	// sort image intensities to get peak infarct 
	irtkRealPixel* i_p; 
	i_p = img->GetPointerToVoxels(); 
	
	int count = 0; 
	topN = new double[top_n]; 
	
	for (int i=0;i<img->GetNumberOfVoxels();i++) { 
		if (*i_p > 0) count++; 	
		i_p++;
	}

	i_arr = new double[count]; 
	i_p = img->GetPointerToVoxels(); 
	
	int k=0;
	for (int i=0;i<img->GetNumberOfVoxels();i++) {
		if (*i_p > 0) 
			i_arr[k++] = *i_p; 
	
		i_p++; 
	}

	bubble_sort(i_arr, count); 
	
	// gets the top unique n elements 
	// for peak infarct this is the N maximal intensity values 
	int found_n = 0; 
	bool f = false; 
	for (int i=0;i<count && found_n < top_n ;i++)
	{
		f = false; 
		if (found_n > 0) { 
			for (int j=0;j<found_n && !f ;j++) 
			{
				if (topN[j] != i_arr[i] && j==found_n-1)		// we have reached end of topN list and yet haven't 
																// come across this intensity value, must be unique then 
				{
					// element is unique insert
					topN[found_n] = i_arr[i]; 
					found_n++; 
				}
			}
		}
		else { 
			// only for the first top element - this is unique to begin with 
			topN[found_n] = i_arr[i]; 
			found_n++; 

		}
	}

	// get mean of topN elements
	i_ctr.clear();	// global container for getting mean and stdev
	int kn=0;
	for (int i=0;i<top_n;i++)
	{
		if (topN[i] > 0) {
			i_ctr.push_back(topN[i]); 
			kn++;
		}
	}
	cout << "\nDEBUG: only " << kn << " maximally high intensity pixels found" << endl; 
	return getStats(i_ctr, 1, -1);		// gets the mean 
}

void computeEDT(irtkRealImage* img1, irtkRealImage* img2, irtkRealImage* edt_img) 
{

	irtkRealImage img3; 
	img3 =* img2; 
	irtkRealPixel *p; 
	p = img3.GetPointerToVoxels(); 
	
	// reverse the segmentation mask for EDT
	for (int i=0;i<img3.GetNumberOfVoxels();i++) 
	{
		if (*p>0)
			*p = 0; 
		else
			*p = 1; 
		p++; 
	}

	irtkEuclideanDistanceTransform<irtkRealPixel> *edt = NULL;
	edt = new irtkEuclideanDistanceTransform<irtkRealPixel>(irtkEuclideanDistanceTransform<irtkRealPixel>::irtkDistanceTransform3D);
	edt->SetInput(&img3);
	edt->SetOutput(edt_img);
	edt->Run();
	edt_img->Write("edt.gipl");
}

void GetATWall(irtkRealImage* edt_img, irtkRealImage* la_seg, irtkRealImage* at_wall, double atwall_thickness)
{
	irtkRealPixel *edt_p, *atwall_p, *laseg_p; 
	edt_p = edt_img->GetPointerToVoxels(); 
	atwall_p = at_wall->GetPointerToVoxels();
	laseg_p = la_seg->GetPointerToVoxels();
	double edt; 

	for (int i=0;i<edt_img->GetNumberOfVoxels();i++)
	{
		//if (*edt_p > 0)
			edt = sqrt(*edt_p);
			//if (edt > 0) cout << *edt_p << "," << edt << "\n";
		//else 
		//	edt = -1;

		if (edt > atwall_thickness){
			//cout << edt << "\t";
			*atwall_p = 0;
		}

		if (*laseg_p == 0)
			*atwall_p = 0;

		edt_p++; 
		atwall_p++; 
		laseg_p++;
	}

	//at_wall->Write("at_wall.gipl"); 
}

void GetBinaryAtWall(irtkRealImage* at_wall)
{
	irtkRealPixel *p; 
	p = at_wall->GetPointerToVoxels(); 
	for (int i=0;i<at_wall->GetNumberOfVoxels();i++)
	{
		if (*p>0) 
		{ 
			*p = 1;
		}
		p++; 
	}
	at_wall->Write("at_wall.gipl");
}

void AtWallMaskToIntensityImage(irtkRealImage* at_wall_mask, irtkRealImage* le_img) 
{
	irtkRealPixel *p, *le; 
	p = at_wall_mask->GetPointerToVoxels(); 
	le = le_img->GetPointerToVoxels(); 

	for (int i=0;i<at_wall_mask->GetNumberOfVoxels();i++)
	{
		if (*p>0) 
		{ 
			*p = *le;
		}

		p++; 
		le++; 
	}
}

void GetMeanSD(irtkRealImage* at_wall, double top98pct, double& mean, double& stdev, double &total_voxels, ofstream& out) 
{
	irtkRealPixel *p; 
	double sum_i=0, n=0, sum_var=0, n2=0;
	double utah_threshold = 0.4;		// utah thresholds at 40% of max intensity 
	double intensity_at_utah_threshold = utah_threshold*top98pct; 
	p = at_wall->GetPointerToVoxels(); 

	for (int i=0;i<at_wall->GetNumberOfVoxels();i++)
	{
		if (*p>0 && *p<intensity_at_utah_threshold) 
		{ 
			sum_i = sum_i + *p; 
			n++; 
		}

		if (*p>0)
		{
			if (out != NULL) 
				out << *p << endl;

			n2++; 
		}

		p++;
	}

	mean = sum_i/n; 
	
	p = at_wall->GetPointerToVoxels(); 
	
	for (int i=0;i<at_wall->GetNumberOfVoxels();i++)
	{
		if (*p>0 && *p<intensity_at_utah_threshold)
		{
			sum_var += (*p - mean)*(*p - mean);
		}
		p++;
	}
	
	stdev = sqrt(sum_var/n);

	// we use n2 as total voxels should be total atrial wall voxels not those between 0 and utah 
	// threshold
	total_voxels = n2;				

}

void GetPercentAboveSD(irtkRealImage* at_wall, double mean, double sd, double nsd, double total_voxels, double &percent) 
{
	irtkRealPixel *p; 
	double n=0, threshold; 
	p = at_wall->GetPointerToVoxels(); 
	threshold = (mean + (nsd*sd));
	for (int i=0;i<at_wall->GetNumberOfVoxels();i++)
	{
		if (*p>0) 
		{ 
			if (*p >= threshold )
			{
				n++;
			}
		}
		p++;
		
	}

	percent = 100*(n/total_voxels);
	//cout << threshold << ", " << n << endl;
}


int main(int argc, char **argv)
{
	int option;
	bool foundArgs1=false,foundArgs2=false,foundArgs3=false, foundMethod=false; 
	irtkRealImage img1, img2; 
	irtkRealImage edt_img2; 
	irtkRealImage at_wall; 
	char* input_f1="", *output_f1="", *output_f2="", *input_f2="", *input_f3=""; 
	double t1, mean, stdev, total_voxels, percent,meanOfTop;
	bool aw_image_available = false, is_aw_intensity_output_to_file = false; 

	if (argc >= 1)
	{
			for (int i = 1; i < argc; i++) {
				if (i + 1 != argc) {
					if (string(argv[i]) == "-i1") {
						input_f1 = argv[i+1]; 
						foundArgs1 = true;
					} else if (string(argv[i]) == "-i2") {
						input_f2 = argv[i+1]; 
						foundArgs2 = true;
					} else if (string(argv[i]) == "-t") {
						t1 = atof(argv[i+1]); 
					} else if (string(argv[i]) == "-aw") {
						input_f3 = argv[i+1]; 
						aw_image_available = true; 
					} else if (string(argv[i]) == "-o") {
						output_f1 = argv[i+1]; 
						foundArgs3 = true;
					} else if (string(argv[i]) == "-out") {
						output_f2 = argv[i+1]; 
						is_aw_intensity_output_to_file = true; 
					}

				}
			}
	}

	if (!(foundArgs1 && foundArgs2 & foundArgs3))
	{
		cout << "\n\nUsage: percentatria \n\n\tArguments:"
			"\n\t -i1 \t - lge image gipl"
			"\n\t -i2 \t - la segmentation gipl" 
			"\n\t -t \t - wall thickness in pixels" 
			"\n\t -o \t - text file for output"
			"\n\t -aw (optional) \t - modified atrial wall segmentation"
			"\n\t -out (optional) \t - output atrial wall intensity to file"
			"\n\n";
		exit(0);
	}

	else
	{
		img1.Read(input_f1); 
		img2.Read(input_f2);
		
		if (!aw_image_available) { 
			at_wall = img1; 
			computeEDT(&img1, &img2, &edt_img2);
			GetATWall(&edt_img2, &img2, &at_wall, t1);
		}
		else
		{
			at_wall.Read(input_f3); 
			AtWallMaskToIntensityImage(&at_wall, &img1); 
		}

		meanOfTop = GetMeanOfTopIntensities(&at_wall, 10);
		cout << "Mean of top intensities = " << meanOfTop << endl;	

		ofstream output_inetensity_2_file; 
		if (is_aw_intensity_output_to_file) { 
			output_inetensity_2_file.open(output_f2);
		}
		GetMeanSD(&at_wall, meanOfTop, mean, stdev, total_voxels, output_inetensity_2_file);
		

		ofstream output_percent_2_file; 
		output_percent_2_file.open(output_f1);
		for (double i=0;i<=10;i+=0.5)
		{
			GetPercentAboveSD(&at_wall, mean, stdev, i, total_voxels, percent);
			output_percent_2_file << i << "\t\t" << percent << endl;
		}
		
		GetBinaryAtWall(&at_wall);
		cout << "mean = " << mean << "\nStdev = " << stdev << "\nTotal voxels = " << total_voxels << endl;

	}

}

Background

Code takes in polygonal mesh with scar information, and Gang's geodesic distance-force maps to compare the force and scar at each ablation location. There is more information on how to use scar3d to view force ablation tags here

void initFTIMeanScarArray(vector< vector< double > >& _v, int items)
{
	for (int i=0;i<items+1;i++)
	{
		vector<double> v;
		v.push_back(0);
		_v.push_back(v); 
	}
}


void initUniqueFTIArray(vector<double>& _v, double _fti)
{
	bool found = false; 
	for (int i=0;i<_v.size() && !found;i++) { 
		if (_v[i] == _fti)
			found = true; 
	}

	if (found == false) { 
		_v.push_back(_fti);

	}
}


bool populateFTIMeanScarArray(vector<double>& _fti_v, vector< vector< double > >& _scarMean, double _fti, double meanScar)
{
	int i; 
	bool found = false; 
	// search for the index of the fti 
	for (i=0;i<_fti_v.size() && !found; i++)
	{
		if (_fti_v[i] == _fti) found = true; 
	}

	if (found == true)
	{
		if (i < _scarMean.size()) { 
			_scarMean[i].push_back(meanScar); 
			
			return true; 
		}
		else { 
			cout << "i = " << i << ", and fti = " << _fti << endl;
			return false; 
		}
	}
	else
		return false; 
}

void writeScarFTIForUniformDistanceForceMapsToFile(ofstream& out, vector<double> _fti, vector< vector< double > > _scarMean)
{
	double _tot, n=0; 
	for (int i=0;i<_fti.size();i++)
	{
		_tot = 0, n=0; 
		if (i < _scarMean.size()) {
			for (int j=0;j<_scarMean[i].size();j++)
			{
				if ( _scarMean[i][j] > 0)	{			// we can have negative std. devs in scar - ignoring these. 
					_tot += _scarMean[i][j]; 
					++n;
				}
			}
		}
		
		if (_tot > 0) { 
			out << _fti[i] << "\t\t" << _tot/n << endl; 
		}
	}
}

int main(int argc, char **argv)
{

	// common 
	int forceMapType=1;
	char* _sc_fn, *_gm_fn, *_txt_fn; 
	double _gm_pt[3], closestPoint[3], dist, _tot, mean, _forceAtPt; 
	vector<vtkIdType> edgeNeighbours; 

	// the following two arrays run correspondingly
	// according to FTI (unique)
	vector<double> _uniqueFTIs; 
	vector< vector< double > > _MeanScarAtEachUniqueFTIInUniformGangMaps; 

	// vtk 
	vtkPolyData* _sc_poly, *_gm_poly; 
	vtkDoubleArray* gangmapScalars;
	vtkFloatArray *scarScalars; 
	vtkIdType cellId, subId, neighbor_point, cell_point_id, closestPointID=-1; 
	vtkCell* closest_cell;
	vtkIdList* pointIdList;
	vtkGenericCell *temp_cell;


	// begin program
	temp_cell = vtkGenericCell::New(); 
	
	ofstream out; 

	if (argc < 3)
	{
		std::cerr << "Usage: " << argv[0]
				<< " scarFTI <scar_shell_in_vtk> <gang_map_vtk> <output_txt_file>" << std::endl;
		return EXIT_FAILURE;
	}

	_sc_fn  = argv[1];
	argc--;
	argv++;
	_gm_fn = argv[1];
	argc--;
	argv++;
	_txt_fn = argv[1]; 
	argc--; 
	argv++; 

	vtkSmartPointer<vtkPolyDataReader> reader1 = vtkSmartPointer<vtkPolyDataReader>::New();
	vtkSmartPointer<vtkPolyDataReader> reader2 = vtkSmartPointer<vtkPolyDataReader>::New();


	cout << "Is the distance-force map distributedn\n1) linearly or,\n2) uniformly ?\n>" << endl;
	cin >> forceMapType; 

	reader1->SetFileName(_sc_fn); 
	reader2->SetFileName(_gm_fn); 
	out.open(_txt_fn); 
	out << "Pressure" << "\t\t" << "Mean scar load" << "\t\t" << "# of triangles" << "\n";
	out << "==========" << "\t\t" << "==============" << "\t\t" << "==============" << "\n";

	if (_IS_DEBUG == 1) cout << "\n\nReading vtk files into memory .. "; 
	reader1->Update(); reader2->Update(); 
	_sc_poly = reader1->GetOutput(); 
	_gm_poly = reader2->GetOutput(); 
	if (_IS_DEBUG == 1) cout << " finished reading .. "; 
	_sc_poly->BuildLinks(); 
	_gm_poly->BuildLinks();

	gangmapScalars = vtkDoubleArray::New(); 
	scarScalars = vtkFloatArray::New(); 

	gangmapScalars =  vtkDoubleArray::SafeDownCast(_gm_poly->GetPointData()->GetScalars()); 
	scarScalars = vtkFloatArray::SafeDownCast(_sc_poly->GetCellData()->GetScalars()); 
	
	if (_IS_DEBUG == 1) cout << "\n\nfinished reading scar and force data from the vtk files .. "; 
	vtkSmartPointer<vtkCellLocator> cell_locator = vtkSmartPointer<vtkCellLocator>::New();
	vtkSmartPointer<vtkIdList> cell_points = vtkSmartPointer<vtkIdList>::New(); 
	cell_locator->SetDataSet(_sc_poly);  
	cell_locator->AutomaticOn(); 
	cell_locator->BuildLocator();
	if (_IS_DEBUG == 1) cout << "\n\nfinished building cell locator to search on scar map .. " << endl; 

	vtkSmartPointer<vtkPoints> _gm_points = vtkSmartPointer<vtkPoints>::New();
	_gm_points->ShallowCopy(_gm_poly->GetPoints());
/*
if (_IS_DEBUG == 1) cout << "\nNumber of points on gangmap = " << _gm_poly->GetNumberOfPoints() << endl;
if (_IS_DEBUG == 1) cout << "and the number of scalars in this map = " << gangmapScalars->GetNumberOfTuples() << endl;

if (_IS_DEBUG == 1) cout << "\nand cells on scarMap = " << _sc_poly->GetNumberOfCells() << endl;
if (_IS_DEBUG == 1) cout << "and the number of scalars in this map = " << scarScalars->GetNumberOfTuples() << endl;
*/

	// if we have a gang map which is uniform (rather than lineraly distributed) 
	// we will run one pass of gang maps to init necessary vectors
	if (forceMapType == 2) { 
		for (vtkIdType _gm_poly_ptId = 0; _gm_poly_ptId<_gm_poly->GetNumberOfPoints(); _gm_poly_ptId++)
		{
			_forceAtPt = gangmapScalars->GetTuple1(_gm_poly_ptId); 
			if (_forceAtPt > 0)
			{
				initUniqueFTIArray(_uniqueFTIs, _forceAtPt); 
			}
		}
		cout << "finished initialization part 1" << endl;

		initFTIMeanScarArray(_MeanScarAtEachUniqueFTIInUniformGangMaps, _uniqueFTIs.size());
		cout << "finished initialization some necessary arrays for storing scar and ftis" << endl;
		//cout << "the size of these arrays should be equal, lets check = " << _MeanScarAtEachUniqueFTIInUniformGangMaps.size() << " and " << _uniqueFTIs.size() << endl;
	}
	for (vtkIdType _gm_poly_ptId = 0; _gm_poly_ptId<_gm_poly->GetNumberOfPoints(); _gm_poly_ptId++)
	{
		_tot = 0; 
		_forceAtPt = gangmapScalars->GetTuple1(_gm_poly_ptId); 
		//if (_IS_DEBUG == 1) cout << " pt. id = " << _gm_poly_ptId << ", force = " << _forceAtPt;

		if (_forceAtPt > 0)
		{
			_gm_points->GetPoint(_gm_poly_ptId, _gm_pt);
			cell_locator->FindClosestPoint(_gm_pt, closestPoint, temp_cell, cellId, subId, dist); 
			//if (_IS_DEBUG == 1) cout << "closestPoint is success ";
			// find neighbours of this cell (cellId) 
			
			// from http://vtk.1045678.n5.nabble.com/getcellneighbours-of-polydata-td1245707.html#a1245713
			// getting first vertices of this cell 
			// and then get the 'EDGE' neighbours of this cells. 
			
			vtkSmartPointer<vtkIdList> vertexIds = vtkSmartPointer<vtkIdList>::New();
			if (cellId > 0) _sc_poly->GetCellPoints(cellId, vertexIds); 
			
			if (vertexIds->GetNumberOfIds() == 3 && cellId < _sc_poly->GetNumberOfCells()) { 
				vtkSmartPointer<vtkIdList> neighbourCellIds1 =vtkSmartPointer<vtkIdList>::New();
				_sc_poly->GetCellEdgeNeighbors(cellId, vertexIds->GetId(0), vertexIds->GetId(1), neighbourCellIds1); 
				if (neighbourCellIds1->GetNumberOfIds() > 0) edgeNeighbours.push_back(neighbourCellIds1->GetId(0)); 

				vtkSmartPointer<vtkIdList> neighbourCellIds2 =vtkSmartPointer<vtkIdList>::New();
				_sc_poly->GetCellEdgeNeighbors(cellId, vertexIds->GetId(1), vertexIds->GetId(2), neighbourCellIds2); 
				if (neighbourCellIds2->GetNumberOfIds() > 0)  edgeNeighbours.push_back(neighbourCellIds2->GetId(0)); 

				
				vtkSmartPointer<vtkIdList> neighbourCellIds3 =vtkSmartPointer<vtkIdList>::New();
				_sc_poly->GetCellEdgeNeighbors(cellId, vertexIds->GetId(2), vertexIds->GetId(0), neighbourCellIds3); 
				if (neighbourCellIds3->GetNumberOfIds() > 0) edgeNeighbours.push_back(neighbourCellIds3->GetId(0)); 

				for (int i=0;i<edgeNeighbours.size();i++)
				{
					_tot += scarScalars->GetTuple1(edgeNeighbours[i]); 
				}
				mean = _tot/edgeNeighbours.size(); 
				
				if (forceMapType == 1) { 
					out << _forceAtPt << "\t\t" << mean << "\t\t" << edgeNeighbours.size() << "\n";
				} 
				else
				{
					// force ditsance map is not linear, it is uniform
					if (!populateFTIMeanScarArray(_uniqueFTIs, _MeanScarAtEachUniqueFTIInUniformGangMaps, _forceAtPt, mean))
						cout << "To my surprise, I have come across an FTI I wasn't expecting .. let me run now, inspect later ... ";
					
				}
				if (edgeNeighbours.size() > 0) edgeNeighbours.clear(); 
			}
		}
	}

	if (forceMapType == 2)
	{
		writeScarFTIForUniformDistanceForceMapsToFile(out, _uniqueFTIs, _MeanScarAtEachUniqueFTIInUniformGangMaps); 
	}
	out.close();

}

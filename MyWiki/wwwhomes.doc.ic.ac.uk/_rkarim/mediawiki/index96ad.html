Volume code

  • Computes the total number of pixels and convert it to mm3 using pixel resolution 
#if defined(_MSC_VER)
#pragma warning ( disable : 4786 )
#endif

#include <irtkImage.h>
#include <vector>

int main(int argc, char **argv)
{
	irtkRealImage img; 
	double t;
	double x_size, y_size, z_size, vol; 
	int n=0;
	
	char *filename1, *filename2; 
	
	if (argc < 3) 
	{
		cout << "Usage: GetVol <image filename> <output textfile> <label>\n\nOutputs the volume in mm cubed by summing pixels" 
			"above label" << endl;
		exit(0);
	}
	filename1 = argv[1];
	argv++;
	argc--;
	filename2 = argv[1];
	argv++;
	argc--;
	t = atof(argv[1]);
	argv++;
	argc--;
	
	img.Read(filename1); 
	img.GetPixelSize(&x_size, &y_size, &z_size);
	cout << "\nX = " << x_size << "\nY = " << y_size << "\nZ = " << z_size << endl;
	irtkRealPixel* img_p; 
	img_p = img.GetPointerToVoxels(); 

	for (int i=0;i<img.GetNumberOfVoxels();i++)
	{
		if (*img_p > t)
			n++;	
		img_p++; 
	}
	
	vol =x_size*y_size*z_size*n;
	ofstream fileIO; 
	fileIO.open(filename2, ios::app); 
	fileIO << vol << endl;
	fileIO.close();
	
}

API

Form submission

  • Form submission API - please refer to Google Docs
  • For testing form submission, POST data is submitted via this php script: 
<?php 
    
    include_once("../api/includes/Cfulcurls.php");
    include_once("../api/includes/Cfultoolbox.php");
    include_once("../api/includes/Datatable.php");
    include_once("../api/includes/Toolbox.php");
    include_once("../api/includes/utility.php");
    include_once("../api/includes/XMLToolbox.php");
    include_once("../api/includes/CSSToolbox.php");
    include_once("../api/includes/ImageContent.php");
   

    $url = "GET-URL-FROM-CONFIDENTIAL-DOCUMENTATION";    
    $send_data = array("app"=>"sail", "field1"=>"field1value", "field2"=>"field2value", "send_to"=>"rashed.karim@gmail.com");
    $content = json_encode($send_data);

    $curl = curl_init($url);
    curl_setopt($curl, CURLOPT_HEADER, false);
    curl_setopt($curl, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($curl, CURLOPT_HTTPHEADER,
            array("Content-type: application/json"));
    curl_setopt($curl, CURLOPT_POST, true);
    curl_setopt($curl, CURLOPT_POSTFIELDS, $content);

    $json_response = curl_exec($curl);

    $status = curl_getinfo($curl, CURLINFO_HTTP_CODE);

    if ( $status != 201 ) {
        die("Error: call to URL $url failed with status $status, response $json_response, curl_error " . curl_error($curl) . ", curl_errno " . curl_errno($curl));
    }


    curl_close($curl);

    $response = json_decode($json_response, true);

   
?>

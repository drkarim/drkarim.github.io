#define _DUMMY_MODE 1

#include <iostream>
#include <string>
#include <sstream>

// boost
#include <boost/filesystem.hpp>

#include <boost/date_time/posix_time/posix_time.hpp>
#include <boost/asio.hpp>

// standard C
#include <string>

// opencv 
#include "cv.h"
#include "cvaux.h"
#include "highgui.h"

using namespace std;
using namespace boost::filesystem;
using namespace boost::lambda;
using namespace cv; 




void FixFirstGreySlice(string path ,string newPathFolder, CvScalar greyColor)
{
	IplImage* img;
	stringstream newPathFileName; 
	string oldname; 
	oldname = path.substr(1+path.find_last_of('/'));
	newPathFileName << newPathFolder << "/" << oldname;
	
	img = cvLoadImage(newPathFileName.str().c_str());
	for (int i = 0; i < img->height; i++)
	{
		for (int j = 0; j < img->width; j++)
		{
			cvSet2D(img, i,j, greyColor);
		}
	}
	imwrite(newPathFileName.str().c_str(), img);
	cvReleaseImage(&img); 

}


int main(int argc, char* argv[])
{
	string oldpath, newpath, pathToFirstGreySlice=""; 
	string oldname, newname; 
	stringstream newpath_folder; 
	stringstream strstr, strstr2;
	
	bool recurse_into_subdirs = false;
	IplImage* img;
	IplImage* img2; 
	int grey; 
	CvSize size;
	int w,h,x,y; 
	double print_loc_in_h, print_loc_in_w; 

	if (_DUMMY_MODE == 0) { 
		if (argc < 5 || argc > 8)
		{
			cerr << "\nUsage: <.exe> <path_to_slices> <desired_width> <desired_height> <slices_save_path> \n\nOptional parameters:\n After <slices_save_path>:\n"
				"Note if using optional params all must be stated\notherwise use grey-value = -1 to ignore\n" 
				"<grey-value> <loc_in_h> <loc_in_w>" << endl;
			exit(0);
		}

		// arguments from command line 
		w = atoi(argv[2]);
		h = atoi(argv[3]);
		
		oldpath = argv[1];
		newpath_folder << argv[4]; 
		
		// optional params 
		if (argc > 5){		
			grey =  atoi(argv[5]);
			print_loc_in_h = atof(argv[6]); 
			print_loc_in_w = atof(argv[7]);
		}
		else 
		{
			print_loc_in_h = 0.75; 
			print_loc_in_w = 0.75;
			grey = -1;

		}
	}
	else
	{
		// DUMMY MODE 
		w = 1920; h = 1080; 
		oldpath = "C:\\slice_test\\print_2"; 
		newpath_folder << "C:\\slice_test\\print"; 
		grey = -1; 
		cout << "Running with default values:\nSD Image path = " << oldpath << "\nHD Image path = " << newpath_folder.str();
		cout << "\n\nPlease centering in y-direction (a number between 0 and 1) For e.g: 0.5 centers image in y: " << endl; 
		cin >> print_loc_in_h;
		cout << "\n\nPlease centering in x-direction (a number between 0 and 1) For For e.g: 0.5 centers image in x: " << endl; 
		cin >> print_loc_in_w;

	}

	const path & directory = oldpath.c_str();
	CvScalar greyValue; 
	bool isInit = false; 
	size.width = w; 
	size.height = h;
	CvScalar currPixVal;


	if( exists( directory ) )
	{
		directory_iterator end ;
		for( directory_iterator iter(directory) ; iter != end ; ++iter ) {
			if ( !is_directory( *iter ) )
			{
				// rename files 	
				img2 = cvCreateImage(size, IPL_DEPTH_8U,3);
				cvZero(img2); 	

				strstr << iter->path(); 
				oldpath = strstr.str();

				// opencv doesnt like backslash in filename paths so replacing them with forward slashes 
				std::replace(oldpath.begin(), oldpath.end(), '\\','/'); 
				
				// remove apostrophes from filename, wierd manifestation from Boost file iterator 
				oldpath = oldpath.substr(1, oldpath.length()-2);					
				oldname = oldpath.substr(1+oldpath.find_last_of('/'));						
				cout << oldname << endl;

				if (oldname.find("init")!=string::npos)
				{
					isInit = true;
				}
				img = cvLoadImage(oldpath.c_str());
				cout << img->height << " , " << img->width << endl;
				
				x = img->height; 
				y = img->width; 

			
				for (int i = 0; i < h; i++)
				{
					for (int j = 0; j < w; j++)
					{
						if (!isInit) { 
							if (i >= print_loc_in_h*(h-x) && j >= print_loc_in_w*(w-y))
							{
								if (i - print_loc_in_h*(h-x) < x && j - print_loc_in_w*(w-y) < y) 
								{
									currPixVal = cvGet2D(img, i - print_loc_in_h*(h-x), j - print_loc_in_w*(w-y));
								
									if ( currPixVal.val[0] > 0 || currPixVal.val[1] > 0 || currPixVal.val[2] > 0)
									{
										//cout << currPixVal.val[0] << "," << currPixVal.val[1] << "," << currPixVal.val[2] << endl;
										if (grey > 0)
											cvSet2D(img2, i,j, cvScalar(grey,grey,grey)); 
										else
											cvSet2D(img2, i,j, currPixVal);
										
									}
									else if (currPixVal.val[0] == 0 && currPixVal.val[1] == 0 && currPixVal.val[2] == 0)
									{
										cvSet2D(img2, i,j, cvScalar(0,0,0));
									}
									
								}
							}
							else
							{
								cvSet2D(img2, i,j, cvScalar(0,0,0));
							}
						}
						else
						{
							cvSet2D(img2, i,j, cvGet2D(img, i,j));
						}
					}
				}
				
				
				
				strstr2 << newpath_folder.str() << "/" << oldname; 
				newname = strstr2.str();
				std::replace(newname.begin(), newname.end(), '\\','/');
				cout << "Writing new resolution image to " << newname <<endl;
				imwrite(newname.c_str(), img2);
				
				
				cvReleaseImage(&img); 
				cvReleaseImage(&img2); 
			
				
				if (isInit) isInit = !isInit;
				strstr2.str("");
				strstr.str("");
			}
		} // end for 

		/*
		if (pathToFirstGreySlice != "")
		{
			FixFirstGreySlice(pathToFirstGreySlice, newpath_folder.str(), greyValue);
		}*/

	}  

	

}

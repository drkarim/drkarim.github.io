	void ListenMessage()
	{
		 array<Byte>^data;
		 System::String^ responseData;

		 // Read the first batch of the TcpServer response bytes.
		 while (true) { 

			 try { 
				data = gcnew array<Byte>(256);
				responseData = System::String::Empty;
				stream = client->GetStream();
				Int32 bytes = stream->Read( data, 0, data->Length );
				responseData = Text::Encoding::ASCII->GetString( data, 0, bytes );

				if (responseData != "") 
					Console::WriteLine( "Received: {0}", responseData );

				SeeMessageSetStatus(responseData);
				client->GetStream()->Flush();
				 
			 }
			 catch ( ArgumentNullException^ e ) 
			{
				Console::WriteLine( "ArgumentNullException: {0}", e );
			}
			catch ( SocketException^ e ) 
			{
				Console::WriteLine( "SocketException: {0}", e );
			}
		}

	}
	
	static void ConnectWorks(  System::String^ ipAddress, System::String^ message )
	{
	   try
	   {
		  // Create a TcpClient.
		  // Note, for this client to work you need to have a TcpServer 
		  // connected to the same address as specified by the server, port
		  // combination.
		IPAddress^ address = IPAddress::Parse(ipAddress);
		Int32 port = 13000;
		TcpClient^ client = gcnew TcpClient();
		client->Connect(address, port);

		  // Translate the passed message into ASCII and store it as a Byte array.
		  array<Byte>^data = Text::Encoding::ASCII->GetBytes( message );

		  // Get a client stream for reading and writing.
		  //  Stream stream = client->GetStream();

		  NetworkStream^ stream = client->GetStream();

		  // Send the message to the connected TcpServer. 
		  stream->Write( data, 0, data->Length );

		  Console::WriteLine( "Sent: {0}", message );

		  // Receive the TcpServer::response.

		  // Buffer to store the response bytes.
		  data = gcnew array<Byte>(256);

		  // String to store the response ASCII representation.
		  System::String^ responseData = System::String::Empty;

		  // Read the first batch of the TcpServer response bytes.
		  Int32 bytes = stream->Read( data, 0, data->Length );
		  responseData = Text::Encoding::ASCII->GetString( data, 0, bytes );
		  Console::WriteLine( "Received: {0}", responseData );

		  // Close everything.
		  client->Close();
	   }
	   catch ( ArgumentNullException^ e ) 
	   {
		  Console::WriteLine( "ArgumentNullException: {0}", e );
	   }
	   catch ( SocketException^ e ) 
	   {
		  Console::WriteLine( "SocketException: {0}", e );
	   }

	   Console::WriteLine( "\n Press Enter to continue..." );
	   Console::Read();
	}

void open_landmarks_cb(Fl_Widget*, void* v)
{
	stringstream ss; 
	std::ifstream infile;
	char* fn; 
	fn = fl_file_chooser("Open landmark text file", "*.txt", dir_path);
	double x,y,z;
	total_landmarks_loaded = 0;

	int ln=0;
	landmarks = new double*[100];			// No more than 100 landmarks are possible
	for (int i=0;i<100;i++)
	{
		landmarks[i] = new double[3];	// each landmark is an xyz co-ordinate, and thus 3 
	}
	ss << "Loading points:\n" << endl;
	if (fn!=NULL) { 
		infile.open(fn); 
		while (infile >> x >> y >> z) { 
			landmarks[ln][0] = x; 
			landmarks[ln][1] = y; 
			landmarks[ln][2] = z; 
			ss << "(" << x << "," << y << "," << z << ")\n";
			ln++;
			total_landmarks_loaded++;
		}
		
		infile.close();
		ss << "Finished loading" << total_landmarks_loaded << " landmarks" << endl;
		OutputMssgs->value(ss.str().c_str()); 
	}

	// display landmarks 
	for (int i=0;i<total_landmarks_loaded;i++)
	{
		vtkSmartPointer<vtkActor> actor = vtkSmartPointer<vtkActor>::New();
		VenusVizBoxCore::VBC_CreateSphere(renderer, 1,0,0, 2.0, landmarks[i], actor);
	}
}

void compute_registration_matrix_cb(Fl_Widget*, void* v)
{

	stringstream ss; 
	// validate everything is ok before computing absolute orientation 

	// check if num of points in physical and robot co-ordinate systems match up 
	if (LandmarksOnPhysical.size() != LandmarksOnModel.size())
	{
		ss << "Landmarks in:\nPhysical=" << LandmarksOnPhysical.size() << "\nModel=" << LandmarksOnPhysical.size() << "\n\nSince they are not equal\nI cannot proceed!";
		OutputMssgs->value(ss.str().c_str()); 
		return; 
	}
	else if (LandmarksOnPhysical.size() < 3)
	{
		ss << "Atleast 3 landmarks are\nnecessary for registration\nCurrently=" << LandmarksOnPhysical.size() << endl;
		OutputMssgs->value(ss.str().c_str()); 
		return;
	}
	
	// all OK so far 
	ss << "Computing physical 2 model .." << endl;
	OutputMssgs->value(ss.str().c_str()); 
	AbsoluteOrientation::compute(LandmarksOnPhysical, LandmarksOnModel, Physical2Model);
	ss << "Done!";
	OutputMssgs->value(ss.str().c_str()); 
	mode = 1;			// now we can start started tracking 
}

void RecordCatheterCoordinateCalibration()
{
	stringstream ss; 
	Point3D t;

	}else if (mode == 11) {

void ComputeCatheterCalibration()
{
	double* Cath_XAxis = new double[3];

	double* Cath_ZAxis = new double[3]; 
	double* origin_on_cath = new double[3];
	double* point_on_z_cath = new double[3]; 
	double* point_on_y_cath = new double[3]; 
	double* point_on_x_cath = new double[3]; 
	double* physical_origin = new double[3]; 
	double* physical_xaxis = new double[3]; 
	double* physical_yaxis = new double[3]; 
	double* physical_zaxis = new double[3]; 
	//	Z axis vector 
	calibrator = new CatheterCalibration(isXPlusCalibrated); 
	calibrator->SetOrigin(Origin[0], Origin[1], Origin[2]);
	calibrator->SetZP(ZPlus[0], ZPlus[1], ZPlus[2]); 

	// record either the X or Y 
	if (!isXPlusCalibrated) { 
		calibrator->SetYP(YPlus[0], YPlus[1], YPlus[2]); 
	}
	else {
		calibrator->SetXP(XPlus[0], XPlus[1], XPlus[2]); 
	}

	// compute catheter axes in physical co-ordinate space 
	calibrator->GetXAxis(Cath_XAxis[0], Cath_XAxis[1], Cath_XAxis[2]);
	calibrator->GetYAxis(Cath_YAxis[0], Cath_YAxis[1], Cath_YAxis[2]);
	calibrator->GetZAxis(Cath_ZAxis[0], Cath_ZAxis[1], Cath_ZAxis[2]);
	
	std::stringstream ss;

	ss << "Done!" << endl;
	OutputMssgs->value(ss.str().c_str()); 

	m00 = calibrator->GetRotMat(0,0);
	m01 = calibrator->GetRotMat(0,1);
	m02 = calibrator->GetRotMat(0,2);

	m21 = calibrator->GetRotMat(2,1); 
	m22 = calibrator->GetRotMat(2,2);
	t1 = calibrator->GetTransMat(0); 
	t2 = calibrator->GetTransMat(1);
	t3 = calibrator->GetTransMat(2);

	//Catheter2Physical.getRotationMatrix(m00, m01, m02, m10, m11, m12, m20, m21, m22);
	//Catheter2Physical.getTranslation(t1, t2, t3);
	ss << m00 << " " << m01 << " " << m02 << " " << t1 << "\n" << m10 << " " << m11 << " " << m12 << " " << t2 << "\n" << m20 << " " << m21 << " " << m22 << " " << t3 << "\n0 0 0 1";

		out << x << " " << y << " " << z << " " << axyz[0] << " " << axyz[1] << " " << axyz[2];
		ss << "Saved successfully to\n" << fn << endl;
		OutputMssgs->value(ss.str().c_str()); 
		out.close();
	}
	
}

/*
*	This will handle the event when a user presses the 'x' on the keyboard 
*/
void keyPressEventHandler(vtkObject* obj, unsigned long,void *sr, void *v)
{
	/* Remember you are marking your pick position with sphere. First step is to pick and then to place a sphere in the cell you have picked
		The code below picks a cell, and you mark one of the cell's vertex with a sphere
	*/
	double screenX, screenY;		// where in the screen you wil be clicking 
	vtkIdType cellID, pointID=-1;		// to store cellID of the picked cell
	double *pick_position = new double[3];
	
	vtkRenderWindowInteractor* iren = vtkRenderWindowInteractor::SafeDownCast(obj); 	// you need the pointer to the interactor, without this you cannot access your 3D scene 
	vtkSmartPointer<vtkRenderWindow> renderWin = iren->GetRenderWindow();			// and from there you get your renderer and your renderwindow 
	vtkSmartPointer<vtkRendererCollection> rendererCollection = renderWin->GetRenderers();		// a render collection is a collection of your renderers but you only have one 
	vtkSmartPointer<vtkRenderer> renderer = vtkSmartPointer<vtkRenderer>::New();
	renderer = rendererCollection->GetFirstRenderer();			// you finally get your renderer from the renderwindow intaractor, this is on what you will put the 
																// the point (mark it using a sphere) and add it as an actor just as you added your mesh 
	//vtkSmartPointer<vtkPolyDataReader> reader = vtkSmartPointer<vtkPolyDataReader>::New();
	
	switch (iren->GetKeyCode())
		{
			case 'x':
			
				screenX = iren->GetEventPosition()[0];			// get the x and y co-ordinates on the screen where you have hit 'x'
				screenY = iren->GetEventPosition()[1];

				cell_picker->Pick(screenX,screenY,0.0, renderer);			// tell the picker where user has pressed 'x' on the screen, a ray is then sent through the scene 

				// 2d point was picked 
				cellID = cell_picker->GetCellId();			// Picking has finished after call to Pick(), now you need to find which polygon the casted ray has intersected (in your line of sight)
				pick_position = cell_picker->GetPickPosition();		// from cellID that was picked, get the (x,y,z) co-ordinates of the mesh polygon to put your sphere 

				pointID = VenusVizBoxCore::VBC_GetFirstCellVertex(Model_A, cellID, pick_position);		// you are seeking the position of the picked cell's vertex and also its ID. 
				stringstream stbf; 
				stbf << "Picked=" << pointID << "\n(X,Y,Z)= " << pick_position[0] << ", " << pick_position[1] << "," << pick_position[2] << ")\n";				
				OutputMssgs->value(stbf.str().c_str());
				
				if (mode != 50) { 
					// Registration mode - registering the physical to the model space
					LandmarksOnModel.push_back(Point3D(pick_position));			// storing landmark points
					Communicator::SendPosReq();
				}
				else if (mode == 50)
				{
					// control mode - send physical co-ordinates to robot 
					// so first convert model to physical space using the inverse transformation
					Point3D p(pick_position);
					Point3D pT;
					Physical2Model.applyInverse(p, pT);
					stringstream sm; 
					sm << "TARGET," << pT[0] << "," << pT[1] << "," << pT[2];
					System::String^ sendSm = gcnew System::String(sm.str().c_str());
					Communicator::SendMessage(sendSm);
				}
				vtkActor* spAct = vtkActor::New(); 
				VenusVizBoxCore::VBC_CreateSphere(renderer, 0,0,1, 1.5, pick_position, spAct );		// now draw the sphere
				SphActorsForReg.push_back(spAct);
				iren->Render();

		
		break;
		
	}
}

	fl_vtk_window->show();

	// cell picker 
	cell_picker = vtkCellPicker::New();
	cell_picker->SetTolerance(0.0005);	// You set a tolerance meaning to what degree of accuracy it is able to select points 
	fl_vtk_window->SetPicker(cell_picker);		// and you tell which interactor this cell picker is part of (you only have one interactor)

	vtkCallbackCommand *callback = vtkCallbackCommand::New();			// A new callback - callback is bascially an event handler 
    callback->SetCallback(keyPressEventHandler);		// you are telling what function will do the event handling 
    fl_vtk_window->AddObserver(vtkCommand::KeyPressEvent, callback);		// invoke callback when key is pressed

	renderer->SetBackground(1,1,1);
	renWin->AddRenderer(renderer);
	fl_vtk_window->SetRenderWindow(renWin);
	fl_vtk_window->Initialize();
	int fl_ret = Fl::run();
	return fl_ret;

}

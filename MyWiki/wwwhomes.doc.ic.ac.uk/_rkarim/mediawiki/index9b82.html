MATLAB scripts

  • This scripts reads all the files that were generated by the above DOS script C:\PhD\experiments\kcl_data\isbi_seg_challenge_upload\enhancement_model_check\<inst>_pre/post and invokes the histogram plotter program

Driver script

  • The script doesn't need to be changed much as it plots the graphs for either the pre or the post for all institutes in one graph.
  • Currently it works for post data. The only that needs changing for pre data is in:
    • fname = strcat(inst_post to inst_pre
    • again in fname = strcat '_b_' to '_a_'
    • and finally, invoke the ReadAllFiles script with parameter filename = 'pre_graph.jpg'
  • The mean and variance are output in p 

function ReadAllFiles(filename)
    
    inst(1,:) = [3,5,7,12,14,17,21,23,26,28];   % yale
    inst(2,:) = [6,8,13,15,16,20,22,24,27,30];  % utah  
    inst(3,:) = [1,2,4,9,10,11,18,19,25,29];    % kcl 
   
    inst_name = {'yale'; 'utah'; 'kcl'};
    colors = [1 0 0; 0 1 0; 0 0 1]; 
    
    inst_pre = {'C:\PhD\experiments\kcl_data\isbi_seg_challenge_upload\enhancement_model_check\yale_pre\'; ... 
        'C:\PhD\experiments\kcl_data\isbi_seg_challenge_upload\enhancement_model_check\utah_pre\'; ... 
        'C:\PhD\experiments\kcl_data\isbi_seg_challenge_upload\enhancement_model_check\kcl_pre\'}; 
    inst_post = {'C:\PhD\experiments\kcl_data\isbi_seg_challenge_upload\enhancement_model_check\yale_post\'; ... 
        'C:\PhD\experiments\kcl_data\isbi_seg_challenge_upload\enhancement_model_check\utah_post\'; ... 
        'C:\PhD\experiments\kcl_data\isbi_seg_challenge_upload\enhancement_model_check\kcl_post\'}; 
    figure1 = figure('Color',[1 1 1]);
    for k=1:3 
        A = []; 
        for i=1:size(inst,2)
            fname = strcat(inst_post{k}, inst_name{k},'_b_',int2str(inst(k,i)),'.txt');
            %A(:,i) = load(fname);
            A = [A; load(fname)];
        end
        size(A)
        GaussianFit(A, 20, 16, colors(k,:));
    end
    legend('Yale', 'Utah', 'KCL', 'Location', 'NorthEast');
    if filename ~= -1
        print(figure1,'-djpeg', filename)
    end

end

Histogram plotter 


function GaussianFit(A, bins, fs, color)
    C=[0];
    if nargin < 3
        fs = 16;
    end
   
   [n, xout] = hist(A,bins);
   
   % normalizing the histograms 
   n = n / sum(n);
    
  % plot(xout, n, 'LineWidth',3, 'Color',color);
   box on;
   grid on;
   hold on;
   %[n2, idx] = sort(n); 
   mA = mean(A);
   p0 = [max(n), mA, std(A)];
   
  % This is for curve fitting (Fitting a gaussian curve) 
   xout =[xout(1)+(-5:-1)*(xout(2)-xout(1)), xout, xout(end)+(1:5)*(xout(end)-xout(end-1))];
   n = [(1:5)*0, n, (1:5)*0];
  p = lsqcurvefit(@Gauss, p0, xout, n);
  p
  
  % Generate more x points so that the Gaussian curve is higher resolution
  % in x 
   x = linspace(xout(1), xout(end), 50);        
   y = Gauss(p, x);
   plot(x, y, 'LineWidth',4, 'Color',color);
  
    set(gca, 'FontSize', fs);
    ylabel('Probability density', 'FontSize', fs);
    xlabel('Normalized intensity (SD)', 'FontSize', fs);
    
end

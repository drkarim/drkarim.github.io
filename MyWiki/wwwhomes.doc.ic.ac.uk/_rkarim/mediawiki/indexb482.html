float VPhantom::unifRand()
{
    return rand() / float(RAND_MAX);
}


float VPhantom::unifRand(float a, float b)
{
    return (b-a)*unifRand() + a;
}

void VPhantom::seed()
{
    srand(time(0));
}

/************************************************************************************
*
*	Box-Muller transform for generating random numbers that follow a normal distribution 
*	From: http://www.cpp-home.com/forum/viewtopic.php?f=7&t=14288
*	and also see polar-form in: http://en.wikipedia.org/wiki/Box-Muller_transform
*	Implementation found in: ftp://ftp.taygeta.com/pub/c/boxmuller.c and http://www.taygeta.com/random/boxmuller.html
*
*************************************************************************************/
float VPhantom::box_muller(float m, float s)	/* normal random variate generator */
{				        /* mean m, standard deviation s */
	float x1, x2, w, y1;
	static float y2;
	static int use_last = 0;

	if (use_last)		        /* use value from previous call */
	{
		y1 = y2;
		use_last = 0;
	}
	else
	{
		do {
			x1 = 2.0 * unifRand() - 1.0;
			x2 = 2.0 * unifRand() - 1.0;
			w = x1 * x1 + x2 * x2;
		} while ( w >= 1.0 );

		w = sqrt( (-2.0 * log( w ) ) / w );
		y1 = x1 * w;
		y2 = x2 * w;
		use_last = 1;
	}

	return( m + y1 * s );
}

void VPhantom::ColorPhantomWithParamFile_Driver(char* img_fn, char* param_fn, char* newPhnatom_fn)
{
	img1.Read(img_fn); 
	cout << "Reading parameter file .. ";
	ReadPhantomParamFile(param_fn); 
	cout << "Finished .. Reading image file and coloring phantom .. ";
	CreatePhantomUsingParamFile(&img1, newPhnatom_fn);
	cout << "almost finished now .. adding noise .. "; 
	AddGaussianNoise(newPhnatom_fn);
	cout << "finished .. now applying blur .. ";
	AddGaussianBlur(newPhnatom_fn); 
	cout << "Phantom creation complete!" << endl;
}


void VPhantom::ReadPhantomParamFile(char* fn)
{
	ifstream param_fl(fn); 
	string line; 
	char* lineStrings, *str; 
	str = new char[30]; 
	int count=0;

	if (param_fl.is_open())
	{
		while (!param_fl.eof())
		{
			getline(param_fl, line);
			strcpy(str, line.c_str()); 
			lineStrings = strtok(str, "\t"); 

			while (lineStrings != NULL)
			{
				if (count == 0) { 
					_labels.push_back(atof(lineStrings)); 
					cout << "\nLabel = " << atof(lineStrings) << endl;
				}
				else if (count == 1) { 
					_mean_ity.push_back(atof(lineStrings)); 
					cout << "Mean of label = " << atof(lineStrings) << endl;
				}
				else if (count == 2) {
					_std_ity.push_back(atof(lineStrings));
					cout << "S.Dev of label = " << atof(lineStrings) << endl;
				}
				else if (count == 3)  { 
					_amplitude_noise.push_back(atof(lineStrings));
					cout << "Noise amplitude = " << atof(lineStrings) << endl;
				}

				lineStrings = strtok (NULL, "\t");
				count++;
					
			}
			count = 0;		// reset 

		}
	}

	if (_labels.size() == _mean_ity.size() && _std_ity.size() == _mean_ity.size())
		_isParamFileOK = true;
	else
		cerr << "\nError:\nAre you sure your parameter file is correct.\nIt must include a mean and std. dev for each label\n"
		"\nFor example:\n<label>... <tab> ... <mean intensity> ... <tab> ... <std. dev.>\n1\t100\t200\n2\t50\t10\n";
}

void VPhantom::CreatePhantomUsingParamFile(irtkRealImage* img, char* phantomNewFileName)
{
	irtkRealPixel *i_p; 
	bool labelFound = false; 
	int j_i;

	if (_isParamFileOK == false) { 
		cerr << "\nError: Seems parameter file is incorrect? SOrry can't proceed further have to terminate. \n" ;
		return; 
	}
	
	if (img->GetZSize() > 1) 
		_imgDim = 3; 
	else 
		_imgDim = 2;
	cout << "Setting image dimension to = " << _imgDim << endl;

	i_p = img->GetPointerToVoxels(); 

	for (int i=0;i<img->GetNumberOfVoxels();i++)
	{
		// loop through the labels 
		labelFound = false; 
		j_i = -1;
		for (int j=0;j<_labels.size() && !labelFound;j++)
		{
			if (*i_p == _labels[j])
			{
				j_i = j;
				labelFound = true; 
			}
		}

		if (labelFound)
		{
			*i_p = abs(box_muller(_mean_ity[j_i], _std_ity[j_i]));
		}
		i_p++; 
	}

	img->Write(phantomNewFileName);
}


void VPhantom::AddGaussianBlur(char* fn) 
{
	// get sigma 
	double sigma = -1; 
	irtkRealImage img, blurred_img; 
	img.Read(fn); 
	
	for (int i=0;i<_labels.size();i++)
	{
		if (_labels[i] == 101)			// label 100 corresponds to noise
		{
			sigma = _mean_ity[i];			
		}
	}
	
	if (sigma == -1) sigma = 1.0;			// use default 

	blurred_img = img; 
	irtkGaussianBlurring<irtkRealPixel> gaussianBlurring(sigma);
	gaussianBlurring.SetInput(&img); 
	gaussianBlurring.SetOutput(&blurred_img); 
	gaussianBlurring.Run();
	blurred_img.Write(fn);
}

void VPhantom::AddGaussianNoise(char* fn)
{

	double mean=-1, var=-1, amplitude=-1; 
	
	const   unsigned int     Dimension = 2;
	typedef short      PixelType;
	typedef itk::Image< PixelType, Dimension >    ImageType;
	typedef itk::ImageFileReader< ImageType >  ReaderType;
	typedef itk::ImageFileWriter< ImageType >  WriterType;
	typedef itk::ImageRegionIterator< ImageType>       IteratorType; 


	// get mean, var and amplitude from param file 
	for (int i=0;i<_labels.size();i++)
	{
		if (_labels[i] == 100)			// label 100 corresponds to noise
		{
			mean = _mean_ity[i];
			var = _std_ity[i]*_std_ity[i];
			amplitude = _amplitude_noise[0];
		}
	}

	if (mean == -1)
	{
		mean = 0; var = 1; amplitude = 5;		// default
	}

	// Gaussian distribution 
	itk::Statistics::GaussianDistribution::Pointer gaussian = itk::Statistics::GaussianDistribution::New();
	gaussian->SetMean(mean);
	gaussian->SetVariance(var);
	
	// Random number generator 
	typedef itk::Statistics::MersenneTwisterRandomVariateGenerator GeneratorType;
    GeneratorType::Pointer generator = GeneratorType::New();
    generator->Initialize();

	ReaderType::Pointer reader = ReaderType::New();
	WriterType::Pointer writer = WriterType::New();

	reader->SetFileName( fn  );
	reader->Update();
	
	ImageType::Pointer im = reader->GetOutput();

	IteratorType  it( im, im->GetRequestedRegion() );
	  
	it.GoToBegin();

	while( !it.IsAtEnd() )
	{
		
		it.Set(it.Get()+(amplitude*gaussian->InverseCDF(generator->GetUniformVariate(0,1))));
		++it;
	}

	
	writer->SetFileName(fn); 
	writer->SetInput(im); 
	writer->Update();
}

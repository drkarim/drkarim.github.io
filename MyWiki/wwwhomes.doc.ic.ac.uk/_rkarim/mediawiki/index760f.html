void normalizeVector(double &a, double &b, double &c)
{
	double magn; 
	magn = sqrt((a*a)+(b*b)+(c*c)); 
	a *=(1/magn); 
	b *=(1/magn); 
	c *=(1/magn); 
}

void DemicateSurface(vtkPolyData* poly)
{
	vtkSmartPointer<vtkDecimatePro> deci = vtkSmartPointer<vtkDecimatePro>::New(); 
	deci->SetInput(poly); 
	deci->SetTargetReduction(0.50); 
	deci->PreserveTopologyOff();
	deci->SplittingOn(); 
	deci->SetBoundaryVertexDeletion(1);
	deci->SetMaximumError(VTK_LARGE_FLOAT);
	deci->Update();
	poly->DeepCopy(deci->GetOutput()); 

}

void SmoothSurfaceMesh(vtkPolyData* poly, int iterations)
{
	vtkSmartPointer<vtkWindowedSincPolyDataFilter> smooth = vtkSmartPointer<vtkWindowedSincPolyDataFilter>::New();
	smooth->SetInput(poly);
	smooth->SetPassBand(0.1);
	smooth->SetNumberOfIterations(iterations);
	smooth->BoundarySmoothingOn();
	smooth->SetFeatureAngle(45);
	smooth->SetEdgeAngle(15);
	smooth->Update(); 
	poly->DeepCopy(smooth->GetOutput());
}

void SmoothSurfaceMesh2(vtkPolyData* poly, int iterations)
{

	smooth->SetInput(poly);
	smooth->SetNumberOfIterations(iterations);
	smooth->BoundarySmoothingOn();
	smooth->SetFeatureAngle(45);
	smooth->SetEdgeAngle(15);
	smooth->Update(); 
	poly->DeepCopy(smooth->GetOutput());
}

void ConvertImageToIstropicPixelResolution(irtkRealImage* img)
{
	double xsize=1, ysize=1, zsize=1; 
// if using irtk version 2010 or lower 
irtkImageFunction<irtkRealPixel> *interpolator = NULL;
	interpolator = new irtkNearestNeighborInterpolateImageFunction<irtkRealPixel>;

//	irtkImageFunction *interpolator = NULL;
// interpolator = new irtkNearestNeighborInterpolateImageFunction;
	irtkResampling<irtkRealPixel> resampling(xsize, ysize, zsize);
    resampling.SetInput(img);
    resampling.SetOutput(img);
    resampling.SetInterpolator(interpolator);
    resampling.Run();
}

bool GetMaxIntensityAlongNormal(double n_x, double n_y, double n_z, double centre_x, double centre_y, 
							 double centre_z, irtkRealImage* img, double user_steps, double& max_i)
{
	
	double insty=0,x=0,y=0,z=0, rounded_x, rounded_y, rounded_z; 
	double prev_x=0, prev_y=0, prev_z=0;
	double step_size = STEP_SIZE; 
	double step_max = user_steps; 
	bool hitBackground = false; 
	int pixel_thickness=0;
	
	max_i = -1;
	// normalize
	normalizeVector(n_x, n_y, n_z);

	for (double i=-step_max;i<=step_max;i+=step_size)
	{
		x = centre_x + (i*n_x); 
		y = centre_y + (i*n_y); 
		z = centre_z + (i*n_z);
		//x = floor(x); y=floor(y); z=floor(z);
		img->WorldToImage(x,y,z);
		
//		cout << "(x,y,z) = " << x << ", " << y << ", "  << z << endl;
		if (x > 0 && y > 0 && z > 0 && x < img->GetX() && y < img->GetY() && z < img->GetZ()) {
			//cout << img->Get(x,y,z) << "\t";
			if (img->Get(x,y,z) > max_i) {
				max_i = img->Get(x,y,z);
			}
		}

		
	}
	
	return (max_i > 0);

}

void MaskLGEImage(irtkRealImage* wall_img, irtkRealImage* lge_img)
{
	irtkRealPixel* wall_ip, *lge_ip; 
	
	wall_ip = wall_img->GetPointerToVoxels(); 
	lge_ip = lge_img->GetPointerToVoxels(); 
	

	for (int i=0;i<wall_img->GetNumberOfVoxels();i++)
	{
		if (*wall_ip <= 0) 
			*lge_ip = 0; 
		wall_ip++; 
		lge_ip++; 
	}

	//lge_img->Write("test.gipl");
}


void GenerateSurfaceMesh(irtkRealImage* wall_img, vtkPolyData* wall_surf, double mcubes_threshold) 
{
	// convert irtk image to a structured point set os unsigned scalars
	vtkSmartPointer<vtkStructuredPoints> vtkImage = vtkSmartPointer<vtkStructuredPoints>::New();
	vtkSmartPointer<vtkImageCast> imageCast = vtkSmartPointer<vtkImageCast>::New();
	wall_img->ImageToVTK(vtkImage);
	imageCast->SetInput(vtkImage);
	imageCast->SetOutputScalarTypeToUnsignedShort();
	vtkSmartPointer<vtkMarchingCubes> cubes = vtkSmartPointer<vtkMarchingCubes>::New();
	cubes->SetInputConnection(imageCast->GetOutputPort());
	cubes->SetValue(0,mcubes_threshold);
	cubes->Update();
	wall_surf->DeepCopy(cubes->GetOutput());
}

void GenerateMaxIntensityMap(irtkRealImage* wall_img, irtkRealImage* edt_img, vtkPolyData* wall_surf, double user_steps, double mean, double std)
{
	double pN[3];
	double cP[3]; 
	double cX, cY,cZ, scalar=0;
	double num_points = 0;
	int success = 0;
	vtkIdType num_cell_points; 
	vtkSmartPointer<vtkIdList> cell_points = vtkSmartPointer<vtkIdList>::New();

	vtkSmartPointer<vtkPolyData> polys = vtkSmartPointer<vtkPolyData>::New(); 
	/*
	*	Compute normals of the surface 
	*/
	vtkSmartPointer<vtkPolyDataNormals> normals = vtkSmartPointer<vtkPolyDataNormals>::New();
	normals->FlipNormalsOn();
	normals->AutoOrientNormalsOn();
	normals->ComputeCellNormalsOff();
	normals->ComputePointNormalsOn();
	normals->SetInput(wall_surf); 
	normals->Update(); 
	polys = normals->GetOutput(); 

	vtkSmartPointer<vtkFloatArray> scalars = vtkSmartPointer<vtkFloatArray>::New();	
	//vtkSmartPointer<vtkFloatArray> cellNormals= vtkFloatArray::SafeDownCast(polys->GetCellData()->GetNormals());
	vtkSmartPointer<vtkFloatArray> pointNormals= vtkFloatArray::SafeDownCast(polys->GetPointData()->GetNormals());

	for (int i=0;i<polys->GetNumberOfPoints();i++)			// running through each polygon 
	{
		pointNormals->GetTuple(i, pN);
		polys->GetPoint(i, cP);
		
		//if (GetThicknessAlongNormal(pN[0], pN[1], pN[2], cP[0], cP[1], cP[2], wall_img, background_t, scalar)) { 
		//	success++;			// a thickness could be computed
		//}

		if (GetMaxIntensityAlongNormal(pN[0], pN[1], pN[2], cP[0], cP[1], cP[2], edt_img, user_steps, scalar)) { 
			success++;			// a thickness could be computed
		}
		
		if (scalar > 0) { 

			scalar = ((scalar-mean)/std);
			scalars->InsertTuple1(i, scalar); 
		}
		else 
			scalars->InsertTuple1(i, 0); 
		scalar=0;
	}
	
	//wall_surf->GetCellData()->SetScalars(scalars);
	wall_surf->GetPointData()->SetScalars(scalars);
	cout << "Number of cells successfully computer thickness for = " << success << " out of a total of " << polys->GetNumberOfPoints() << " cells" <<endl;

}

void ReadDataFile(char* fn, string patient_lge_reference, double &mean, double &stdev) 
{
	std::ifstream infile(fn);
	std::string line;

	while (std::getline(infile, line))
	{
		std::istringstream iss(line);
		double mean_from_file, stdev_from_file;
		string fnf; 
		if (!(iss >> fnf >> mean_from_file >> stdev_from_file)) { 
			cout << "could not read line" << endl; 
			break;
		} // error
			
		// process pair (a,b)
		//cout << fnf << ", " << a << ", " << b << endl;

		if (patient_lge_reference.compare(fnf) == 0) 
		{
			mean = mean_from_file; 
			stdev = stdev_from_file; 
			break; 
		}
	}
	cout << "Found mean  = " << mean << ", and stdev = " << stdev << endl;
}

string ExtractFileName(string full_path)
{
	string filename = full_path; 
	
	// Remove directory if present.
	// Do this before extension removal incase directory has a period character.
	const size_t last_slash_idx = filename.find_last_of("\\/");
	if (std::string::npos != last_slash_idx)
	{
		filename.erase(0, last_slash_idx + 1);
	}

	// Remove extension if present.
	const size_t period_idx = filename.rfind('.');
	if (std::string::npos != period_idx)
	{
		filename.erase(period_idx);
	}
	
	cout << "Extracted patient lge reference as = " << filename << endl;
	return filename;
}


int main(int argc, char * argv[])
{

	char* input_img_f1="", *input_img_f2="", *input_vtk_f="", *output_f="", *data_file=""; 
	double user_steps=MAX_STEPS, smoothing_t=10; 
	irtkRealImage wall_img; 
	irtkRealImage lge_img; 
	bool foundArgs= false, isSmoothMesh=false, isDataFileInput=false;
	vtkPolyData* wall_surf; 
	double mean=0, std=1; 
	string patient_lge_reference; 
	string output_vtk;
	string output_vtk_dir_prefix; 

	if (argc >= 1) 
	{
		for (int i = 1; i < argc; i++) {
			if (i + 1 != argc) {
				if (string(argv[i]) == "-w") {
					input_img_f1 = argv[i+1]; 
					foundArgs = true;	
				}
				else if (string(argv[i]) == "-l") {
					input_img_f2 = argv[i+1]; 
					foundArgs = true;	
				}
				
				else if (string(argv[i]) == "-t") {
					user_steps = atof(argv[i+1]); 
					foundArgs = true;	
				} 
				else if (string(argv[i]) == "-smooth") {
					smoothing_t = atoi(argv[i+1]); 
					isSmoothMesh = true; 
				}
				else if (string(argv[i]) == "-data") {
					data_file = argv[i+1];
					isDataFileInput = true; 
				}
			}
		}
	}

	if (foundArgs == false)
	{
		cout << "\nUsage: scar3d_wall.exe  -w <wall binary gipl> -l <lge gipl>"
			"\n\nOptional parameters:\n======================\n"
			"\n-t\t<specify number of pixels in and out. Default is 4. "
			"This parameter is Optional>\n-smooth\t <n number of iterations for smoothing>\n";
		exit(0); 
	}

	// read image 
	wall_img.Read(input_img_f1); 
	lge_img.Read(input_img_f2);
	
	patient_lge_reference = ExtractFileName(input_img_f2); 

	output_vtk = "C:\\temp\\"+patient_lge_reference+".vtk"; 
	
	if (isDataFileInput)  
	{
		ReadDataFile(data_file, patient_lge_reference, mean, std); 
	}
	 
	ConvertImageToIstropicPixelResolution(&wall_img); 
	ConvertImageToIstropicPixelResolution(&lge_img); 
	MaskLGEImage(&wall_img, &lge_img);
	//GetXYZDiffBetweenWorldAndImage(&wall_img); 

	wall_surf = vtkPolyData::New();
	
	GenerateSurfaceMesh(&wall_img, wall_surf, 0.5);
		

	cout << "Generated surface mesh .. " << endl;
	
	if (isSmoothMesh) {
		cout << "Smoothing mesh, performing " << smoothing_t << " iterations \n";
		DemicateSurface(wall_surf);
		SmoothSurfaceMesh(wall_surf, smoothing_t);
	}

	GenerateMaxIntensityMap(&wall_img, &lge_img, wall_surf, user_steps, mean, std); 

	vtkSmartPointer<vtkPolyDataWriter> vtkwriter = vtkSmartPointer<vtkPolyDataWriter>::New(); 
	vtkwriter->SetInput(wall_surf); 
	vtkwriter->SetFileName(output_vtk.c_str()); 
	vtkwriter->SetFileTypeToASCII(); 
	vtkwriter->Write(); 

}

void QuickFlatener::DisplayPoly(vtkRenderWindowInteractor *flrwi, vtkPolyData* poly, bool isCellData, vtkActor* polyActor)
{
	double min_scalar, max_scalar; 
	vtkSmartPointer<vtkRenderer> renderer;
	vtkSmartPointer<vtkRenderWindow> renderWin;
	renderWin = flrwi->GetRenderWindow();
	vtkSmartPointer<vtkRendererCollection> rendererCollection = renderWin->GetRenderers();
	vtkSmartPointer<vtkLookupTable> lut = vtkSmartPointer<vtkLookupTable>::New();
	vtkSmartPointer<vtkScalarBarActor> scalarBar = vtkSmartPointer<vtkScalarBarActor>::New();
	renderer = rendererCollection->GetFirstRenderer();

	vtkSmartPointer<vtkPolyDataMapper> mapper = vtkSmartPointer<vtkPolyDataMapper>::New();	
	cout << "Adding polydata to mapper, polydata now has " << poly->GetNumberOfCells() << " polygons .. " << endl;
	mapper->SetInput(poly); 
	
	mapper->ScalarVisibilityOn();

	if (isCellData) 
		mapper->SetScalarModeToUseCellData();
	else 
		mapper->SetScalarModeToUsePointData();
	
	VizBoxCore::VBC_getMinMaxScalars(poly, isCellData, min_scalar, max_scalar); 
	mapper->SetScalarRange(min_scalar, (max_scalar)); 
	lut->SetTableRange(min_scalar, (max_scalar)); 
	
	
	// lookuptable 
	lut->SetHueRange(0.33, 0.0);			// red - 0, yellow - 60, green- 120, cyan - 180, blue - 240, magenta - 300 (on the hue circle, pg. 42 vtk book)
	lut->Build(); 

	mapper->SetLookupTable(lut); 
	
	//vtkSmartPointer<vtkActor> polyActor = vtkSmartPointer<vtkActor>::New(); 

	polyActor->SetMapper(mapper); 
	renderer->AddActor(polyActor);
	
	
	scalarBar->SetLookupTable(lut);
	scalarBar->SetTitle("Intensity");
	scalarBar->GetTitleTextProperty()->SetColor(0,0,0);
	scalarBar->SetWidth(0.08); 
	scalarBar->SetHeight(1.0); 
	scalarBar->SetDisplayPosition(3,8); 
	scalarBar->GetLabelTextProperty()->SetColor(0,0,0);
	 
	
	renderer->AddActor2D(scalarBar);			// displays the scalar bar 
	renderWin->AddRenderer(renderer);
	renderWin->Render();
}

void QuickFlatener::ReadUVMappingFile(char* uvMapFileName, char* uvPointIDmaxFileName) 
{

	// find maximum id by reading file 
	int max2DPointID=-1, max3DPointID=-1; 
	int line_num=0, index3D, index2D;
	ifstream in1, in2; 
	string l, test_l1, test_l2; 
	char* lineChar, *ch; 

	in1.open(uvMapFileName); 
	in2.open(uvPointIDmaxFileName); 

	if (in2.is_open())
	{
		getline(in2, l); 
		max3DPointID = atoi(l.c_str()); 
		l = ""; 
		getline(in2, l); 
		max2DPointID = atoi(l.c_str()); 
		cout << "\nMaximum 2D point ID = " << max2DPointID << "\nMaximum 3d point ID = " << max3DPointID << endl;
	}
	
	// init arrays 
	if (max3DPointID != -1) 
	{
		_3d_2_2d_map_size = max3DPointID+1;
		_2d_2_3d_map_size = max2DPointID+1;
		
		_3d_2_2d_mapping = new int[max3DPointID+1]; 
		_2d_2_3d_mapping = new int[max2DPointID+1]; 
		

		for (int i=0;i<max3DPointID+1;i++) 
			_3d_2_2d_mapping[i] = -1;			// initialize

		for (int i=0;i<max2DPointID+1;i++) 
			_2d_2_3d_mapping[i] = -1;			// initialize
		
		if (in1.is_open())
		{
			while (getline(in1, l)) {
				lineChar = new char[l.size()+1]; 
				strcpy(lineChar, l.c_str()); 

				ch = strtok(lineChar, "\t"); 
				//cout << l << endl;
				test_l1 = ch; 
				//index3D = atoi(ch); 
				ch = strtok(NULL, "\t"); 
				test_l2 = ch; 
				//index2D = atoi(ch); 
			//	cout << test_l1 << ":" << test_l2 << endl;
				if (sscanf(test_l1.c_str(), "%d%", &index2D) == 1 && sscanf(test_l2.c_str(), "%d%", &index3D) == 1) 
				{
					//cout << index3D << ":" << index2D << endl;
					if (index3D <  max3DPointID+1) {
						_3d_2_2d_mapping[index3D] = index2D; 
					}

					if (index2D < max2DPointID + 1) { 
						_2d_2_3d_mapping[index2D] = index3D; 
					}
				}

				delete lineChar; 
				line_num++; 

			}
		}
	}
	else 
	{
		cerr << "Could not read the size of point id array from file " << uvPointIDmaxFileName << endl;
		return;
	}

	in1.close(); 
	in2.close();

}

void CreateSphere(vtkRenderer* renderer, double radius, double* position3D)
{
	vtkSmartPointer<vtkSphereSource> sphere = vtkSmartPointer<vtkSphereSource>::New(); 
	sphere->SetThetaResolution(8);
	sphere->SetPhiResolution(8);
	sphere->SetRadius(radius);

	vtkSmartPointer<vtkPolyDataMapper> sphereMapper =vtkSmartPointer<vtkPolyDataMapper>::New(); 
	sphereMapper->ScalarVisibilityOff();
	sphereMapper->SetInput(sphere->GetOutput());

	vtkSmartPointer<vtkActor> actor = vtkSmartPointer<vtkActor>::New();
	actor->SetMapper(sphereMapper); 
	actor->GetProperty()->SetColor(1,0,1); 
	actor->SetPosition(position3D); 

	renderer->AddActor(actor);
	
}

	double *mapped_position = new double[3];

	vtkSmartPointer<vtkRendererCollection> rendererCollection = renderWin->GetRenderers();
	vtkSmartPointer<vtkRenderer> renderer = vtkSmartPointer<vtkRenderer>::New();

	vtkSmartPointer<vtkRenderWindow> OtherRenderWin = other_iren->GetRenderWindow();
	vtkSmartPointer<vtkRendererCollection> OtherRendererCollection = OtherRenderWin->GetRenderers();
	vtkSmartPointer<vtkRenderer> other_renderer = vtkSmartPointer<vtkRenderer>::New();
	other_renderer = OtherRendererCollection->GetFirstRenderer();

	// file writers for writing points selected to file 
	

	if (!cb_data->TextFileWriter3d.is_open() && !cb_data->TextFileWriter2d.is_open())  {
		cb_data->TextFileWriter3d.open("point_id_selected_on_shell.txt"); 
		cb_data->TextFileWriter2d.open("point_id_selected_on_unfold.txt"); 
	}
	cout << "Mouse button pressed" << endl;
	switch (iren->GetKeyCode())
	{
	
		case 'x':			
			
			screenX = iren->GetEventPosition()[0];
			screenY = iren->GetEventPosition()[1];
			
			point_picker->Pick(screenX,screenY,0.0, renderer);
			pick_position = point_picker->GetPickPosition(); 

		
			// 2d point was picked 
			pointID = point_picker->GetPointId(); 
			cout << "point id picked = " << pointID << ", on window = " << cb_data->windowNum << endl;
			if (pointID != -1)
			{
				if (cb_data->windowNum == 1) {
					if (pointID < cb_data->qf->_2d_2_3d_map_size)
						pointID_mapped = cb_data->qf->_2d_2_3d_mapping[pointID];
				}
				else if (cb_data->windowNum == 2) {
					if (pointID < cb_data->qf->_3d_2_2d_map_size)
						pointID_mapped = cb_data->qf->_3d_2_2d_mapping[pointID];
				}
				
				if (pointID_mapped != -1)
				{
					cout << "Corresponding point found on 2D map with id = " << pointID_mapped << endl;
					if (cb_data->windowNum == 1)				// user clicked on 2d map 
					{
						//cout << "3d map has = " << cb_data->Mesh_2D->GetNumberOfPoints() << endl; 
						cb_data->Mesh_3d->GetPoint(pointID_mapped, mapped_position);		// look for point on 3d map 
						cb_data->TextFileWriter2d << pointID << endl; 
						cb_data->TextFileWriter3d << pointID_mapped << endl; 
					}
					if (cb_data->windowNum == 2) {
						//cout << "3d map has = " << cb_data->qf->_2d_map->GetNumberOfPoints() << endl;
						cb_data->Mesh_2d->GetPoint(pointID_mapped, mapped_position); 
						cb_data->TextFileWriter2d << pointID_mapped << endl; 
						cb_data->TextFileWriter3d<< pointID << endl; 
					}
					
					// draw point 
					if (cb_data->windowNum == 1) { 
						CreateSphere(renderer, 0.005, pick_position); 
						CreateSphere(other_renderer, 1.5, mapped_position); 
					}
					else {
						CreateSphere(renderer, 1.5, pick_position);
						CreateSphere(other_renderer, 0.005, mapped_position); 
					}

					
					OtherRenderWin->Render(); 
					renderWin->Render();
				}
			}
			
		
		break; 
	
		case 'o': 
			cb_data->TextFileWriter3d << "X" << endl; 			// X in file denotes that selecting points for this region 	is finished
			cb_data->TextFileWriter2d << "X" << endl; 
			//cb_data->TextFileWriter3d.close(); 
			//cb_data->TextFileWriter2d.close();
		break; 
	}

void GetConnectedVertices_2(vtkSmartPointer<vtkPolyData> mesh, int id, vtkSmartPointer<vtkIdList> connectedVertices)
{
 
  //get all cells that vertex 'id' is a part of
  vtkSmartPointer<vtkIdList> cellIdList = vtkSmartPointer<vtkIdList>::New();
  mesh->GetPointCells(id, cellIdList);
 
  /*
  cout << "Vertex 0 is used in cells ";
  for(vtkIdType i = 0; i < cellIdList->GetNumberOfIds(); i++)
    {
    cout << cellIdList->GetId(i) << ", ";
    }
  cout << endl;
  */
 
  for(vtkIdType i = 0; i < cellIdList->GetNumberOfIds(); i++)
    {
    //cout << "id " << i << " : " << cellIdList->GetId(i) << endl;
 
    vtkSmartPointer<vtkIdList> pointIdList = vtkSmartPointer<vtkIdList>::New();
    mesh->GetCellPoints(cellIdList->GetId(i), pointIdList);
 
    //cout << "End points are " << pointIdList->GetId(0) << " and " << pointIdList->GetId(1) << endl;
 
    if(pointIdList->GetId(0) != id)
      {
      //cout << "Connected to " << pointIdList->GetId(0) << endl;
      connectedVertices->InsertNextId(pointIdList->GetId(0));
      }
    else
      {
      //cout << "Connected to " << pointIdList->GetId(1) << endl;
      connectedVertices->InsertNextId(pointIdList->GetId(1));
      }
    }
 
}


void GetConnectedVertices(vtkSmartPointer<vtkPolyData> mesh, int seed, vtkSmartPointer<vtkIdList> connectedVertices) 
{
	/*
	vtkSmartPointer<vtkExtractEdges> extractEdges = vtkSmartPointer<vtkExtractEdges>::New();
	extractEdges->SetInput(mesh_0);
	extractEdges->Update();

	//get all cells that vertex 'seed' is a part of 
	
	
	vtkSmartPointer<vtkIdList> cellIdList = vtkSmartPointer<vtkIdList>::New(); 
	mesh->GetPointCells(seed, cellIdList); 

	//cout << "There are " << cellIdList->GetNumberOfIds() << " cells that use point " << seed << endl; 

	//loop through all the cells that use the seed point 
	for(vtkIdType i = 0; i < cellIdList->GetNumberOfIds(); i++) 
	{ 

		vtkCell* cell = mesh->GetCell(cellIdList->GetId(i)); 
		//cout << "The cell has " << cell->GetNumberOfEdges() << " edges." << endl; 

		//if the cell doesn't have any edges, it is a line 
		//if(cell->GetNumberOfEdges() <= 0) 
		//{ 
		//	continue; 
		//} 

		// Going through the edges of each cell, remember that an edge is 
		// made up of two vertices 
		for(vtkIdType e = 0; e < cell->GetNumberOfEdges(); e++) 
		{ 
			vtkCell* edge = cell->GetEdge(e); 

			vtkIdList* pointIdList = edge->GetPointIds(); 
			//cout << "This cell uses " << pointIdList->GetNumberOfIds() << " points" << endl; 

			if(pointIdList->GetId(0) == seed || pointIdList->GetId(1) == seed) 
			{ 
				if(pointIdList->GetId(0) == seed) 
				 { 
				  connectedVertices->InsertNextId(pointIdList->GetId(1)); 
				 } 
				else 
				  { 
				  connectedVertices->InsertNextId(pointIdList->GetId(0)); 
				  } 
			} 
		} 


	} 

	 //cout << "There are " << connectedVertices->GetNumberOfIds() << " points connected to point " << seed << endl; 
} 

double GetEuclideanDistanceIn3D(double* p1, double* p2) 
{
	double s;
	s = ((p1[0] - p2[0])*(p1[0] - p2[0])) + ((p1[1] - p2[1])*(p1[1] - p2[1])) + ((p1[2] - p2[2])*(p1[2] - p2[2])); 
	return sqrt(s);
}

void AverageDistanceToNeighbour(vtkPolyData* map2D, vtkPolyData* map3D, QuickFlatener* qf, double* avg_neighbour_distances, double factor) 
{
	vtkSmartPointer<vtkIdList> neighbours = vtkSmartPointer<vtkIdList>::New();
	int index2D;
	double* n_coord_self_2d = new double[3];
	double* n_coord_neighbour_2d = new double[3];
	double* n_coord_self_3d = new double[3];
	double* n_coord_neighbour_3d = new double[3];
	double avg_dist_2d=0, total_dist_2d=0, total_neighbours=0; 
	double avg_dist_3d=0, total_dist_3d=0; 
	double n_percent=0, prev_p_complete_int=0;
	
	vtkSmartPointer<vtkTriangleFilter> triangleFilter = vtkSmartPointer<vtkTriangleFilter>::New();
	triangleFilter->SetInput(map3D); 
	triangleFilter->Update();

	vtkSmartPointer<vtkExtractEdges> extractEdges = vtkSmartPointer<vtkExtractEdges>::New();
	 extractEdges->SetInputConnection(triangleFilter->GetOutputPort());
	extractEdges->Update();
	vtkSmartPointer<vtkPolyData> map3D_2 = extractEdges->GetOutput();
	map3D_2->BuildLinks();
cout << "Using a factor of " << factor << " to adjust scaling between 3D and 2D" << endl;
	// computing average distance to neighbour for each vertex
	for (int i=0;i<map3D_2->GetNumberOfPoints();i++)
	{
		total_dist_2d = 0; 
		total_dist_3d = 0; 
		total_neighbours=0;
		if (i < map3D->GetNumberOfPoints()) { 
				index2D = qf->_3d_2_2d_mapping[i];
				//cout << "Processing point " << index2D << endl;
				map3D_2->GetPoint(i, n_coord_self_3d);
				if (index2D > 0 && index2D < map2D->GetNumberOfPoints()) {
					
					map2D->GetPoint(index2D, n_coord_self_2d); 
					
					//cout << "(x = " << n_coord_self[0] << ", " << n_coord_self[1] << ", " << n_coord_self[2] << ")"; 
					GetConnectedVertices_2(map3D_2, i, neighbours);         
				
					// now iterate the neighbours container, and get the neighbour vertex co-ordinates

					for (int j=0;j<neighbours->GetNumberOfIds();j++)
					{
						if (neighbours->GetId(j) < map2D->GetNumberOfPoints()) {
							index2D = qf->_3d_2_2d_mapping[neighbours->GetId(j)]; 
							if (index2D > 0 && index2D < map2D->GetNumberOfPoints()) {
								map2D->GetPoint(index2D, n_coord_neighbour_2d);
								map3D_2->GetPoint(neighbours->GetId(j), n_coord_neighbour_3d);
								
								total_dist_2d += GetEuclideanDistanceIn3D(n_coord_self_2d, n_coord_neighbour_2d);
								total_dist_3d += GetEuclideanDistanceIn3D(n_coord_self_3d, n_coord_neighbour_3d);
								total_neighbours++;
							}
						}
					}
					
					if (total_neighbours > 0) { 
						avg_dist_2d = total_dist_2d/total_neighbours;
						avg_dist_3d = total_dist_3d/total_neighbours;
						avg_neighbour_distances[i] = factor*avg_dist_2d/avg_dist_3d; 
						//cout << avg_dist << endl;
					}
					else { 
						//cout << "Point " << i << " has no neighbour " << endl;
					}
					
					double p_complete = 100*((double)i/(double)map3D_2->GetNumberOfPoints());
					int p_complete_int = round(p_complete); 
					if (p_complete_int > 0 && p_complete_int > prev_p_complete_int && p_complete_int%10 == 0) { 
						n_percent+=10;
						cout << n_percent << " % completed\n"; 
					}
					prev_p_complete_int = p_complete_int;
					//cout << "Point = " << i << ", average = " << avg_dist << endl;
					neighbours->Reset();      // reset container for next fetch
					// do something else
				}
		}
	}
}



void OutputAccuracyMap(char* mesh_2d_fn, char* mesh_3d_fn, char* uvMap_fn, char* uvPointIDmax_fn, char* output_fn, double factor)
{
	QuickFlatener* qf = new QuickFlatener(); 
	CallBackDataContainer* cb_data1 = new CallBackDataContainer(); 
	CallBackDataContainer* cb_data2 = new CallBackDataContainer(); 

	vtkPolyData* _2D_mesh, *_3D_mesh; 
	qf->ReadUVMappingFile(uvMap_fn, uvPointIDmax_fn); 
	//qf->DisplayMapping();

	_2D_mesh = vtkPolyData::New(); 
	_3D_mesh = vtkPolyData::New(); 
	
	qf->ReadPolyData(mesh_2d_fn, _2D_mesh); 
	qf->ReadPolyData(mesh_3d_fn, _3D_mesh); 

	int total_points_in_3d_mesh = _3D_mesh->GetNumberOfPoints(); 
	int total_points_in_2d_mesh = _2D_mesh->GetNumberOfPoints();

	double* avg_neighbour_distances_3D = new double[total_points_in_3d_mesh]; 
	double* avg_neighbour_distances_2D = new double[total_points_in_2d_mesh];
	
	//AverageDistanceToNeighbour(_2D_mesh, avg_neighbour_distances_2D); 
	cout << "Finished average distances to neighbour in 2D mesh\n";
	AverageDistanceToNeighbour(_2D_mesh, _3D_mesh, qf, avg_neighbour_distances_3D, factor); 
	cout << "\nFinished Computing average distances to neighbour in 3D mesh\n";
	
	


	//vtkSmartPointer<vtkFloatArray> new_scalars = vtkFloatArray::SafeDownCast(_2D_mesh->GetPointData()->GetScalars());
	vtkFloatArray* new_scalars = vtkFloatArray::New();
	

	for (int i=0;i<_2D_mesh->GetNumberOfPoints();i++)
	{
		new_scalars->InsertNextTuple1(-1);
	}
	
	int index3D;
	double ratio_dist_3d_to_2d, avg_distance_to_neighbour_of_corresponding_point_3D;
	
	for (int i=0;i<_2D_mesh->GetNumberOfPoints();i++)
	{ 
		 index3D = qf->_2d_2_3d_mapping[i]; 
		 avg_distance_to_neighbour_of_corresponding_point_3D = avg_neighbour_distances_3D[index3D]; 
		// ratio_dist_3d_to_2d = avg_distance_to_neighbour_of_corresponding_point_3D/avg_neighbour_distances_2D[i]; 
		  ratio_dist_3d_to_2d = avg_distance_to_neighbour_of_corresponding_point_3D; 
		// cout << "Distance = " << ratio_dist_3d_to_2d << endl;
		 new_scalars->SetTuple1(i,ratio_dist_3d_to_2d); 	
	}
	
	_2D_mesh->GetPointData()->SetScalars(new_scalars);
	vtkSmartPointer<vtkPolyDataWriter> writer = vtkSmartPointer<vtkPolyDataWriter>::New(); 
	writer->SetFileName(output_fn); 
	writer->SetInput(_2D_mesh); 
	writer->Update();
	
}

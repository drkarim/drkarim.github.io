<source lang="cpp"> void graphCutClassifier::setNWeight(int i, int j, int k, int n_index) { int maxX, maxY, maxZ; double pos[3], v, n_intensity, intensity, neighbour_cost; maxX = atrial_wall.GetX(); maxY = atrial_wall.GetY(); maxZ = atrial_wall.GetZ();

intensity = atrial_wall.Get(i,j,k); if (i+1 > 0 && i+1 < maxX && g_isvisited.Get(i+1,j,k) == NOT_VISITED) { pos[0] = i+1; pos[1] = j; pos[2] = k; v = planes->FunctionValue(pos); if (isPointInsideBox(pos[0], pos[1], pos[2], planes, &atrial_wall) && atrial_wall.Get(i+1,j,k) > 0 && g_index.Get(i+1, j, k) > 0) { n_intensity = atrial_wall.Get(i+1, j, k); neighbour_cost = exp(-pow((intensity - n_intensity), 2)/(2*pow(N_REGULATOR, 2))); neighbour_cost *= _SCALING_FACTOR; config_graph->add_edge(n_index, g_index.Get(i+1, j, k), neighbour_cost, neighbour_cost); } }

if (i-1 > 0 && i-1 < maxX && g_isvisited.Get(i-1,j,k) == NOT_VISITED) { pos[0] = i-1; pos[1] = j; pos[2] = k; v = planes->FunctionValue(pos); if (isPointInsideBox(pos[0], pos[1], pos[2], planes, &atrial_wall) && atrial_wall.Get(i-1,j,k) > 0 && g_index.Get(i-1, j, k) > 0) { n_intensity = atrial_wall.Get(i-1, j, k); neighbour_cost = exp(-pow((intensity - n_intensity), 2)/(2*pow(N_REGULATOR, 2))); neighbour_cost *= _SCALING_FACTOR; //if (abs(intensity - n_intensity) < 20 && abs(intensity - n_intensity) > 0) cout << "neighbour_cost less than diff = " << abs(intensity - n_intensity) << ",cost = " << neighbour_cost << endl; config_graph->add_edge(n_index, g_index.Get(i-1, j, k), neighbour_cost, neighbour_cost); } }

if (j+1 > 0 && j+1 < maxY && g_isvisited.Get(i,j+1,k) == NOT_VISITED) { pos[0] = i; pos[1] = j+1; pos[2] = k; v = planes->FunctionValue(pos); if (isPointInsideBox(pos[0], pos[1], pos[2], planes, &atrial_wall) && atrial_wall.Get(i,j+1,k) > 0 && g_index.Get(i, j+1, k) > 0) { n_intensity = atrial_wall.Get(i, j+1, k); neighbour_cost = exp(-pow((intensity - n_intensity), 2)/(2*pow(N_REGULATOR, 2))); neighbour_cost *= _SCALING_FACTOR; //if (abs(intensity - n_intensity) < 20 && abs(intensity - n_intensity) > 0) cout << "neighbour_cost less than diff = " << abs(intensity - n_intensity) << ",cost = " << neighbour_cost << endl; config_graph->add_edge(n_index, g_index.Get(i, j+1, k), neighbour_cost, neighbour_cost); } }

if (j-1 > 0 && j-1 < maxY && g_isvisited.Get(i,j-1,k) == NOT_VISITED) { pos[0] = i; pos[1] = j-1; pos[2] = k; v = planes->FunctionValue(pos); if (isPointInsideBox(pos[0], pos[1], pos[2], planes, &atrial_wall) && atrial_wall.Get(i,j-1,k) > 0 && g_index.Get(i, j-1, k) > 0) { n_intensity = atrial_wall.Get(i, j-1, k); neighbour_cost = exp(-pow((intensity - n_intensity), 2)/(2*pow(N_REGULATOR, 2))); neighbour_cost *= _SCALING_FACTOR; //if (abs(intensity - n_intensity) < 20 && abs(intensity - n_intensity) > 0) cout << "neighbour_cost less than diff = " << abs(intensity - n_intensity) << ",cost = " << neighbour_cost << endl; config_graph->add_edge(n_index, g_index.Get(i, j-1, k),neighbour_cost, neighbour_cost); } }

if (k+1 > 0 && k+1 < maxZ && g_isvisited.Get(i,j,k+1) == NOT_VISITED) { pos[0] = i; pos[1] = j; pos[2] = k+1; v = planes->FunctionValue(pos); if (isPointInsideBox(pos[0], pos[1], pos[2], planes, &atrial_wall) && atrial_wall.Get(i,j,k+1) > 0 && g_index.Get(i, j, k+1) > 0) { n_intensity = atrial_wall.Get(i, j, k+1); neighbour_cost = exp(-pow((intensity - n_intensity), 2)/(2*pow(N_REGULATOR, 2))); neighbour_cost *= _SCALING_FACTOR; //if (abs(intensity - n_intensity) < 20 && abs(intensity - n_intensity) > 0) cout << "neighbour_cost less than diff = " << abs(intensity - n_intensity) << ",cost = " << neighbour_cost << endl; config_graph->add_edge(n_index, g_index.Get(i, j, k+1), neighbour_cost, neighbour_cost); } }

if (k-1 > 0 && k-1 < maxZ && g_isvisited.Get(i,j,k-1) == NOT_VISITED) { pos[0] = i; pos[1] = j; pos[2] = k-1; v = planes->FunctionValue(pos); if (isPointInsideBox(pos[0], pos[1], pos[2], planes, &atrial_wall) && atrial_wall.Get(i,j,k-1) > 0 && g_index.Get(i, j, k-1) >0) { n_intensity = atrial_wall.Get(i, j, k-1); neighbour_cost = exp(-pow((intensity - n_intensity), 2)/(2*pow(N_REGULATOR, 2))); neighbour_cost *= _SCALING_FACTOR; // if (abs(intensity - n_intensity) < 20 && abs(intensity - n_intensity) > 0) cout << "neighbour_cost less than diff = " << abs(intensity - n_intensity) << ",cost = " << neighbour_cost << endl; config_graph->add_edge(n_index, g_index.Get(i, j, k-1), neighbour_cost, neighbour_cost); } }

g_isvisited.Put(i,j,k, VISITED); } </source>

Segmentation overlap

From Rashed Karim Wiki

Jump to: navigation, search

Usage

Only for comparing outlines drawn for t slices where t < n, n = total slices in image. 

Usage: overlap <manual_outline.hdr> <auto_outline.hdr> <auto_x_outline.hdr> <num_slices> <slice1> <slice2> .. <slice n>

Code 

#include <irtkImage.h>
#include <irtkErosion.h>
#include <vector>
using namespace std; 

int main(int argc, char **argv)
{
       char *filename1, *filename2, *filename3, *output, *slice_number;
       irtkGreyPixel *ptr1, *ptr2;
	   int num_slice; 
	   vector<int> slice_numbers; 
	
	   if (argc <4)
	   {
		   cerr << "Usage: overlap <manual_outline.hdr> <auto_outline.hdr> <auto_x_outline.hdr> <num_slices> <slice1> <slice2> .. <slice n>" << endl; 
		   exit(1); 
	   }
       // get files to subtract
       filename1 = argv[1];
       argv++;
       argc--;
	   filename2 = argv[1];
	   argv++;
       argc--;
       filename3 = argv[1]; 
	   argv++;
       argc--;
	   slice_number = argv[1]; 
	   argv++;
	   argc--;
	   num_slice = atoi(slice_number); 
	   for (int i=0;i<num_slice;i++)
	   {
		   slice_numbers.push_back(atoi(argv[1])); 
		   argv++;
		   argc--;
	   }
			   

       // read image
       irtkGreyImage seg, seg_x, f1, f2, f3, seg_erode, seg_x_erode, input1, input2, input3; 
	   double overlap_percent, overlap_percent_x; 

       double d;
       f1.Read(filename1);	
	   seg.Read(filename2); 
	   seg_x.Read(filename3); 
	   seg_erode = seg; seg_x_erode = seg; 
	   // compute boundary of segmentation 
	    irtkErosion<irtkGreyPixel> erosion;
		erosion.SetInput(&seg);
		erosion.SetOutput(&seg_erode);		// my auto seg
		erosion.Run(); // 1 iteration 

		erosion.SetInput(&seg_x);
		erosion.SetOutput(&seg_x_erode);		// auto seg x
		erosion.Run();	// 1 iteration 
		//seg_x_erode.Write("erosion_x.hdr"); 
		//seg_erode.Write("erosion.hdr"); 
		f2 = seg; f3 = seg; 
		for (int i=0;i<seg.GetX();i++)
		{
			for (int j=0;j<seg.GetY();j++)
			{
				for (int k=0;k<seg.GetZ();k++)
				{
					f2.Put(i,j,k,0); f3.Put(i,j,k,0); 
				}

			}
		}
	   
		for (int i=0;i<seg.GetX();i++)
		{
			for (int j=0;j<seg.GetY();j++)
			{
				for (int k=0;k<seg.GetZ();k++)
				{
					if (seg.Get(i,j,k) > 0 && seg_erode.Get(i,j,k) == 0)
						f2.Put(i,j,k,1);
					
					if (seg_x.Get(i,j,k) > 0 && seg_x_erode.Get(i,j,k) == 0)
						f3.Put(i,j,k,1); 
				}
			}
		}
		f2.Write("mri_auto_seg_outline.hdr");  f3.Write("mri_auto_seg_x_outline.hdr");
			   
	   //input2 = input1; 
	   int x=20, y=40, z=30, r1=10, r2=20;
	   int circle_offset_x = 0; // the amount u want the second circle to go into circle 1

	   int maxX, maxY, maxZ,n=1, overlap=0, total=0, overlap_x=0; 
	   bool overlapfound = false, overlapfound_x = false; 
	   maxX = f1.GetX(); maxY = f1.GetY(); maxZ = f1.GetZ(); 
	
	   for (int v=0;v<slice_numbers.size();v++)
	   {
		   // running through each slice at a time 
		   
		   input1 = f1.GetRegion(0,0,slice_numbers[v], maxX, maxY, slice_numbers[v]+1); 
		   input2 = f2.GetRegion(0,0,slice_numbers[v], maxX, maxY, slice_numbers[v]+1); 
		   input3 = f3.GetRegion(0,0,slice_numbers[v], maxX, maxY, slice_numbers[v]+1);
		   overlap=0; overlap_x=0; total=0;

			for (int i=0;i<input1.GetX();i++)
			{
				for (int j=0;j<input1.GetY();j++)
				{
					overlapfound = false; 
					overlapfound_x = false;		// reset
					if (input1.Get(i,j,0) >0)
					{
						// n here is the overlap pixel threshold
						for (int a=-n;a<=n && !(overlapfound && overlapfound_x);a++)
						{
							for (int b=-n;b<=n && !(overlapfound && overlapfound_x);b++)
							{
								if (i+a >= 0 && i+a <maxX && j+b >= 0 && j+b < maxY && !(a==0 &&  b==0))
								{
									if (input2.Get(i+a,j+b,0) > 0 && !overlapfound)
									{
										overlap++; 
										overlapfound = true; 
									}

									if (input3.Get(i+a, j+b, 0) > 0 && !overlapfound_x)
									{
										overlap_x++; 
										overlapfound_x = true; 
									}
								}
								
							}
						}
						total++; 
						
					}
					
				}
			}
	   
				  
			  
			// input2.Write(output);
			overlap_percent=0, overlap_percent_x=0; 
			overlap_percent = (double)overlap/(double)total; 
			overlap_percent_x = (double)overlap_x/(double)total;
			overlap_percent*=100; 
			overlap_percent_x*=100; 
			cout << "Slice no. " << slice_numbers[v] << ":: auto = " << overlap_percent << ", auto_x = " << overlap_percent_x << "\n\n"; 
	   }	// next slice

	   //cout << "Min (X,Y,Z) = " << minX <<", " << minY << ", " << minZ << endl << "Max (X, Y, Z) = " << 
		   //maxX << ", " << maxY << ", " << maxZ << endl;
}
